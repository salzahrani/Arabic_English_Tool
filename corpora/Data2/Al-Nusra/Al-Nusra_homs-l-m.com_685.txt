ئۇيغۇرچە
سۈرە پۇرقان - عدد الآيات 77
تَبَارَكَ الَّذِي نَزَّلَ الْفُرْقَانَ عَلَىٰ عَبْدِهِ لِيَكُونَ لِلْعَالَمِينَ نَذِيرًا ( 1 )
پۈتۈن جاھان ئەھلىنى (اﷲ نىڭ ئازابىدىن) ئاگاھلاندۇرغۇچى بولۇشى ئۈچۈن، بەندىسىگە (يەنى مۇھەممەد ئەلەيھىسسالامغا) ھەق بىلەن باتىلنى ئايرىغۇچى قۇرئاننى نازىل قىلغان اﷲ نىڭ بەرىكىتى بۈيۈكتۇر
الَّذِي لَهُ مُلْكُ السَّمَاوَاتِ وَالْأَرْضِ وَلَمْ يَتَّخِذْ وَلَدًا وَلَمْ يَكُن لَّهُ شَرِيكٌ فِي الْمُلْكِ وَخَلَقَ كُلَّ شَيْءٍ فَقَدَّرَهُ تَقْدِيرًا ( 2 )
ئاسمانلارنىڭ ۋە زېمىننىڭ پادىشاھلىقى اﷲ قا خاستۇر، اﷲ نىڭ بالىسى يوقتۇر، پادىشاھلىقتا شېرىكىمۇ يوقتۇر، اﷲ ھەممە نەرسىنى ياراتتى، ئاندىن (ئۆزىنىڭ خاھىشى بويىچە) ئۇنىڭغا مۇناسىپ ئۆلچەمنى بەلگىلىدى
وَاتَّخَذُوا مِن دُونِهِ آلِهَةً لَّا يَخْلُقُونَ شَيْئًا وَهُمْ يُخْلَقُونَ وَلَا يَمْلِكُونَ لِأَنفُسِهِمْ ضَرًّا وَلَا نَفْعًا وَلَا يَمْلِكُونَ مَوْتًا وَلَا حَيَاةً وَلَا نُشُورًا ( 3 )
مۇشرىكلار اﷲ نى قويۇپ ھېچ نەرسىنى يارىتالمايدىغان ۋە ئۆزلىرى يارىتىلغان ئىلاھلار (يەنى بۇتلار) غا ئىبادەت قىلدى. ئۇلار (يەنى بۇتلار) ئۆزلىرىدىن بىرەر زىياننى دەپئى قىلالمايدۇ. ئۆزلىرىگە بىرەر پايدىنى كەلتۈرەلمەيدۇ، (باشقىلارنىڭ) جېنىنى ئېلىشقا، (باشقىلارغا) ھايات بېغىشلاشقا ياكى (ئۆلۈكلەرنى) تىرىلدۈرۈشكە قادىر بولالمايدۇ
وَقَالَ الَّذِينَ كَفَرُوا إِنْ هَٰذَا إِلَّا إِفْكٌ افْتَرَاهُ وَأَعَانَهُ عَلَيْهِ قَوْمٌ آخَرُونَ ۖ فَقَدْ جَاءُوا ظُلْمًا وَزُورًا ( 4 )
كاپىرلار: «بۇ (يەنى قۇرئان) پەقەت ئۇ (يەنى مۇھەممەد) ئۆزى توقۇغان يالغاندۇر، ئۇنىڭغا ئۇنى توقۇشقا باشقا بىر قەۋم (يەنى ئەھلى كىتاب) ياردەم بەرگەن» دەيدۇ، ئۇلار (پەيغەمبەر ئەلەيھىسسالامغا بوھتان چاپلاش بىلەن ئۇنىڭغا) زۇلۇم قىلدى ۋە يالغاننى چاپلىدى
وَقَالُوا أَسَاطِيرُ الْأَوَّلِينَ اكْتَتَبَهَا فَهِيَ تُمْلَىٰ عَلَيْهِ بُكْرَةً وَأَصِيلًا ( 5 )
ئۇلار ئېيتتى: «(بۇ) بۇرۇنقىلارنىڭ ئەپسانىلىرىدۇركى، ئۇلارنى ئۇ يازدۇرۇۋالغان، (ھىپزى قىلىۋېلىش ئۈچۈن) ئۇنىڭغا ئەتىگەن – ئاخشامدا ئوقۇپ بېرىلىۋاتىدۇ»
قُلْ أَنزَلَهُ الَّذِي يَعْلَمُ السِّرَّ فِي السَّمَاوَاتِ وَالْأَرْضِ ۚ إِنَّهُ كَانَ غَفُورًا رَّحِيمًا ( 6 )
ئېيتقىنكى، «ئۇنى ئاسمانلاردىكى ۋە زېمىندىكى سىرنى بىلىپ تۇرىدىغان اﷲ نازىل قىلدى، اﷲ ھەقىقەتەن ناھايىتى مەغپىرەت قىلغۇچىدۇر، (بەندىلەرگە) ناھايىتى مېھرىباندۇر»
وَقَالُوا مَالِ هَٰذَا الرَّسُولِ يَأْكُلُ الطَّعَامَ وَيَمْشِي فِي الْأَسْوَاقِ ۙ لَوْلَا أُنزِلَ إِلَيْهِ مَلَكٌ فَيَكُونَ مَعَهُ نَذِيرًا ( 7 )
مۇشرىكلار (مۇھەممەد ئەلەيھىسسالامنى مەسخىرە قىلىپ) ئېيتتى: «بازارلاردا (بىز تاماق يېگەندەك) تاماق يەيدىغان، (بىز ماڭغاندەك) مېڭىپ يۈرىدىغان بۇ قانداق پەيغەمبەر؟ ئۇنىڭ بىلەن بىللە ئاگاھلاندۇرغۇچى بولۇش ئۈچۈن اﷲ ئۇنىڭغا نېمىشقا بىر پەرىشتە ئەۋەتمىدى
أَوْ يُلْقَىٰ إِلَيْهِ كَنزٌ أَوْ تَكُونُ لَهُ جَنَّةٌ يَأْكُلُ مِنْهَا ۚ وَقَالَ الظَّالِمُونَ إِن تَتَّبِعُونَ إِلَّا رَجُلًا مَّسْحُورًا ( 8 )
يا ئۇنىڭغا بىر خەزىنە بېرىلمىدى، يا ئۇنىڭغا مېۋىلىرىنى يەيدىغان بىر باغ بېرىلمىدى؟» زالىملار (يەنى كاپىرلار): «سىلەر پەقەت سېھىرلەنگەن بىر ئادەمگە ئەگىشىۋاتىسىلەر» دېدى
انظُرْ كَيْفَ ضَرَبُوا لَكَ الْأَمْثَالَ فَضَلُّوا فَلَا يَسْتَطِيعُونَ سَبِيلًا ( 9 )
مۇشرىكلارنىڭ سېنىڭ ئۈچۈن نۇرغۇن مىساللارنى كەلتۈرگەنلىكىگە قارىغىن، ئۇلار ئازدى، توغرا يول تاپالمايدۇ
تَبَارَكَ الَّذِي إِن شَاءَ جَعَلَ لَكَ خَيْرًا مِّن ذَٰلِكَ جَنَّاتٍ تَجْرِي مِن تَحْتِهَا الْأَنْهَارُ وَيَجْعَل لَّكَ قُصُورًا ( 10 )
اﷲ نىڭ بەرىكىتى بۈيۈكتۇركى، ئەگەر ئۇ خالىسا ساڭا ئۇلارنىڭ ئېيتقانلىرىدىنمۇ ياخشى، ئاستىدىن ئۆستەڭلار ئېقىپ تۇرىدىغان باغلارنى ۋە چوڭ سارايلارنى ئاتا قىلاتتى
بَلْ كَذَّبُوا بِالسَّاعَةِ ۖ وَأَعْتَدْنَا لِمَن كَذَّبَ بِالسَّاعَةِ سَعِيرًا ( 11 )
ياق، ئۇلار قىيامەتنى ئىنكار قىلدى، قىيامەتنى ئىنكار قىلغانلارغا بىز دوزاخنى تەييارلىدۇق
إِذَا رَأَتْهُم مِّن مَّكَانٍ بَعِيدٍ سَمِعُوا لَهَا تَغَيُّظًا وَزَفِيرًا ( 12 )
دوزاخ ئۇلارنى يىراقتىن كۆرگەن چاغدا، ئۇلار دوزاخنىڭ غەزەپتىن قاينىغان ۋە (ئېشەكتەك) ھاڭرىغان ئاۋازنى ئاڭلايدۇ
وَإِذَا أُلْقُوا مِنْهَا مَكَانًا ضَيِّقًا مُّقَرَّنِينَ دَعَوْا هُنَالِكَ ثُبُورًا ( 13 )
ئۇلار زەنجىر بىلەن باغلانغان ھالدا دوزاخنىڭ تار يېرىگە تاشلانغان چاغدا، بۇ يەردە ئۇلار (ئۆزلىرىنىڭ) ئۆلۈمىنى تىلەيدۇ
لَّا تَدْعُوا الْيَوْمَ ثُبُورًا وَاحِدًا وَادْعُوا ثُبُورًا كَثِيرًا ( 14 )
(ئۇلارغا) بۈگۈن سىلەر بىر ئۆلۈمنى ئەمەس، تالاي ئۆلۈملەرنى تىلەڭلار (دېيىلىدۇ)
قُلْ أَذَٰلِكَ خَيْرٌ أَمْ جَنَّةُ الْخُلْدِ الَّتِي وُعِدَ الْمُتَّقُونَ ۚ كَانَتْ لَهُمْ جَزَاءً وَمَصِيرًا ( 15 )
ئۇلارغا ئېيتقىنكى، «شۇ (دوزاخ) ياخشىمۇ يا تەقۋادارلارغا ۋەدە قىلىنغان، (كىرگەن كىشى) مەڭگۈ قالىدىغان جەننەتتە ياخشىمۇ؟» جەننەت تەقۋادارلارغا بېرىلگەن مۇكاپاتتۇر ۋە ئۇلارنىڭ ئاخىرى بارىدىغان جايىدۇر
لَّهُمْ فِيهَا مَا يَشَاءُونَ خَالِدِينَ ۚ كَانَ عَلَىٰ رَبِّكَ وَعْدًا مَّسْئُولًا ( 16 )
ئۇلار جەننەتتە خالىغان نېمەتلەردىن بەھرىمەن بولىدۇ، جەننەتتە مەڭگۈ قالىدۇ، بۇ پەرۋەردىگارىڭدىن ئىشقا ئاشۇرۇش تىلىنىدىغان ۋەدىدۇر
وَيَوْمَ يَحْشُرُهُمْ وَمَا يَعْبُدُونَ مِن دُونِ اللَّهِ فَيَقُولُ أَأَنتُمْ أَضْلَلْتُمْ عِبَادِي هَٰؤُلَاءِ أَمْ هُمْ ضَلُّوا السَّبِيلَ ( 17 )
شۇ كۈندە اﷲ ئۇلارغا قوشۇپ، ئۇلار اﷲ نى قويۇپ، چوقۇنغان نەرسىلىرىنى يىغىدۇ. ئاندىن اﷲ (ئۇلارغا): «مېنىڭ بۇ بەندىلىرىمنى سىلەر ئازدۇردۇڭلارمۇ؟ يا (توغرا) يولدىن ئۇلار ئۆزلىرى ئازدىمۇ؟» دەيدۇ
قَالُوا سُبْحَانَكَ مَا كَانَ يَنبَغِي لَنَا أَن نَّتَّخِذَ مِن دُونِكَ مِنْ أَوْلِيَاءَ وَلَٰكِن مَّتَّعْتَهُمْ وَآبَاءَهُمْ حَتَّىٰ نَسُوا الذِّكْرَ وَكَانُوا قَوْمًا بُورًا ( 18 )
(ئىبادەت قىلىنغۇچىلار) ئېتىدۇ: «ئى اﷲ! سەن پاكتۇرسەن، سېنى قويۇپ باشقىلارنى دوست تۇتۇش (يەنى سەندىن باشقىغا چوقۇنۇش) بىزگە (ۋە سېنىڭ مەخلۇقاتىڭدىن ھېچ ئەھەدىگە) لايىق ئەمەس، لېكىن سەن ئۇلارنى ۋە ئۇلارنىڭ ئاتا – بوۋىلىرىنى شۇ قەدەر نېمەتكە چۈمدۈردۇڭكى، ئۇلار سېنى ياد ئېتىشىنى ئۇنتۇدى. ئۇلار ھالاك بولغۇچى قەۋم بولدى»
فَقَدْ كَذَّبُوكُم بِمَا تَقُولُونَ فَمَا تَسْتَطِيعُونَ صَرْفًا وَلَا نَصْرًا ۚ وَمَن يَظْلِم مِّنكُمْ نُذِقْهُ عَذَابًا كَبِيرًا ( 19 )
شۈبھىسىزكى، ئىبادەت قىلىنغۇچىلار سىلەرنىڭ (ئۇلارنى مەبۇد دېگەن) سۆزۈڭلارنى ئىنكار قىلدى، (ئى كۇففارلار!) سىلەر (ئۆزۈڭلاردىن ئازابنى) دەپئى قىلىشقىمۇ ۋە (بۇ بالادىن ئۆزۈڭلارنى قۇتۇلدۇرۇشقا) ياردەم بېرىشكىمۇ قادىر بولالمايسىلەر، سىلەردىن كىمكى (اﷲ قا شېرىك كەلتۈرۈش بىلەن ئۆزىگە) زۇلۇم قىلىدىكەن، ئۇنىڭغا بىز (ئاخىرەتتە) قاتتىق ئازابنى تېتىتىمىز
وَمَا أَرْسَلْنَا قَبْلَكَ مِنَ الْمُرْسَلِينَ إِلَّا إِنَّهُمْ لَيَأْكُلُونَ الطَّعَامَ وَيَمْشُونَ فِي الْأَسْوَاقِ ۗ وَجَعَلْنَا بَعْضَكُمْ لِبَعْضٍ فِتْنَةً أَتَصْبِرُونَ ۗ وَكَانَ رَبُّكَ بَصِيرًا ( 20 )
بىز سەندىن ئىلگىرى ئەۋەتكەن پەيغەمبەرلەرنىڭ ھەممىسىلا ئەلۋەتتە تاماق يەيتتى، بازارلاردا مېڭىپ يۈرەتتى، بەزىڭلارنى بەزىڭلار بىلەن سىنىدۇق، سەۋر قىلامسىلەر؟ (يەنى سەۋر قىلىڭلار) پەرۋەردىگارىڭ (سەۋر قىلغۇچىلار بىلەن سەۋر قىلمىغۇچىلارنى) كۆرۇپ تۇرغۇچىدۇر
وَقَالَ الَّذِينَ لَا يَرْجُونَ لِقَاءَنَا لَوْلَا أُنزِلَ عَلَيْنَا الْمَلَائِكَةُ أَوْ نَرَىٰ رَبَّنَا ۗ لَقَدِ اسْتَكْبَرُوا فِي أَنفُسِهِمْ وَعَتَوْا عُتُوًّا كَبِيرًا ( 21 )
بىزگە مۇلاقات بولۇشنى ئۈمىد قىلمايدىغانلار (يەنى مۇشرىكلار): «نېمىشقا بىزگە پەرىشتىلەر چۈشۈرۈلمەيدۇ، ياكى (بىز نېمىشقا) پەرۋەردىگارىمىزنى كۆرمەيمىز» دېيىشتى. شۈبھىسىزكى ئۇلار ئۆزلىرىنى چوڭ تۇتتى ۋە تولىمۇ ھەددىدىن ئاشتى
يَوْمَ يَرَوْنَ الْمَلَائِكَةَ لَا بُشْرَىٰ يَوْمَئِذٍ لِّلْمُجْرِمِينَ وَيَقُولُونَ حِجْرًا مَّحْجُورًا ( 22 )
ئۇلار (يەنى مۇشرىكلار) (ئۆزلىرىنىڭ جېنىنى ئېلىش ئۈچۈن چۈشكەن) پەرىشتىلەرنى كۆرگەن كۈنى گۇناھكارلارغا خۇش خەۋەر بولمايدۇ، پەرىشتىلەر ئۇلارغا: «(سىلەرگە جەننەت) ھارامدۇر» دەيدۇ
وَقَدِمْنَا إِلَىٰ مَا عَمِلُوا مِنْ عَمَلٍ فَجَعَلْنَاهُ هَبَاءً مَّنثُورًا ( 23 )
بىز ئۇلارنىڭ قىلغان ياخشى ئەمەللىرىنى بىر تەرەپ قىلىپ، ئۇنى ئۇچۇپ يۈرگەن توزاندەك قىلىۋېتىمىز، (يەنى ئۇلار ئىمانسىىز بولغانلىقتىن، قىلغان ياخشى ئەمەللىرىنى يوققا چىقىرىمىز)
أَصْحَابُ الْجَنَّةِ يَوْمَئِذٍ خَيْرٌ مُّسْتَقَرًّا وَأَحْسَنُ مَقِيلًا ( 24 )
ئەھلى جەننەت بۇ كۈندە ئەڭ ياخشى جايدا ۋە ئەڭ ياخشى ئارامگاھتا بولىدۇ
وَيَوْمَ تَشَقَّقُ السَّمَاءُ بِالْغَمَامِ وَنُزِّلَ الْمَلَائِكَةُ تَنزِيلًا ( 25 )
شۇ كۈندە ئاسمان بۇلۇت بىلەن بىرلىكتە يېرىلىدۇ، پەرىشتىلەر (بەندىلەرنىڭ نامە - ئەمالىنى ئېلىپ) ئارقىمۇئارقا چۈشىدۇ
الْمُلْكُ يَوْمَئِذٍ الْحَقُّ لِلرَّحْمَٰنِ ۚ وَكَانَ يَوْمًا عَلَى الْكَافِرِينَ عَسِيرًا ( 26 )
پادىشاھلىق بۇ كۈندە مەرھەمەتلىك اﷲ قا خاستۇر، بۇ كۈن كاپىرلارغا قېيىندۇر
وَيَوْمَ يَعَضُّ الظَّالِمُ عَلَىٰ يَدَيْهِ يَقُولُ يَا لَيْتَنِي اتَّخَذْتُ مَعَ الرَّسُولِ سَبِيلًا ( 27 )
شۇ كۈنى زالىم (يەنى كاپىر) ئىككى قولىنى چىشلەپ: «ئىسىت! پەيغەمبەر بىلەن (نىجاتلىق) يولىنى تۇتسامچۇ، ئىسىت! پالانىنى دوست تۇتمىغان بولسامچۇ؟ قۇرئان ماڭا يەتكەندىن كېيىن، ئۇ (يەنى پالانى) مېنى قۇرئاندىن، شەك - شۈبھىسىزكى، ئازدۇردى» دەيدۇ، شەيتان ئىنساننى (ئازدۇرۇپ بولۇپ) تاشلىۋېتىدۇ
يَا وَيْلَتَىٰ لَيْتَنِي لَمْ أَتَّخِذْ فُلَانًا خَلِيلًا ( 28 )
شۇ كۈنى زالىم (يەنى كاپىر) ئىككى قولىنى چىشلەپ: «ئىسىت! پەيغەمبەر بىلەن (نىجاتلىق) يولىنى تۇتسامچۇ، ئىسىت! پالانىنى دوست تۇتمىغان بولسامچۇ؟ قۇرئان ماڭا يەتكەندىن كېيىن، ئۇ (يەنى پالانى) مېنى قۇرئاندىن، شەك - شۈبھىسىزكى، ئازدۇردى» دەيدۇ، شەيتان ئىنساننى (ئازدۇرۇپ بولۇپ) تاشلىۋېتىدۇ
لَّقَدْ أَضَلَّنِي عَنِ الذِّكْرِ بَعْدَ إِذْ جَاءَنِي ۗ وَكَانَ الشَّيْطَانُ لِلْإِنسَانِ خَذُولًا ( 29 )
شۇ كۈنى زالىم (يەنى كاپىر) ئىككى قولىنى چىشلەپ: «ئىسىت! پەيغەمبەر بىلەن (نىجاتلىق) يولىنى تۇتسامچۇ، ئىسىت! پالانىنى دوست تۇتمىغان بولسامچۇ؟ قۇرئان ماڭا يەتكەندىن كېيىن، ئۇ (يەنى پالانى) مېنى قۇرئاندىن، شەك - شۈبھىسىزكى، ئازدۇردى» دەيدۇ، شەيتان ئىنساننى (ئازدۇرۇپ بولۇپ) تاشلىۋېتىدۇ
وَقَالَ الرَّسُولُ يَا رَبِّ إِنَّ قَوْمِي اتَّخَذُوا هَٰذَا الْقُرْآنَ مَهْجُورًا ( 30 )
پەيغەمبەر ئېيتتى: «ئى پەرۋەردىگارىم! شۈبھىسىزكى، مېنىڭ قەۋمىم بۇ قۇرئاننى تاشلاندۇق قىلىپ قويدى»
وَكَذَٰلِكَ جَعَلْنَا لِكُلِّ نَبِيٍّ عَدُوًّا مِّنَ الْمُجْرِمِينَ ۗ وَكَفَىٰ بِرَبِّكَ هَادِيًا وَنَصِيرًا ( 31 )
شۇنىڭدەك (يەنى ساڭا قەۋمىڭنىڭ مۇشرىكلىرىدىن دۈشمەنلەرنى قىلغاندەك) ھەربىر پەيغەمبەرگە بىر قىسىم گۇناھكارلارنى ئۇنىڭ دۈشمىنى قىلدۇق، پەرۋەردىگارىڭ يول كۆرسەتكۈچى ۋە ياردەمچى بولۇشقا ساڭا يېتەرلىكتۇر
وَقَالَ الَّذِينَ كَفَرُوا لَوْلَا نُزِّلَ عَلَيْهِ الْقُرْآنُ جُمْلَةً وَاحِدَةً ۚ كَذَٰلِكَ لِنُثَبِّتَ بِهِ فُؤَادَكَ ۖ وَرَتَّلْنَاهُ تَرْتِيلًا ( 32 )
كاپىرلار: «قۇرئان نېمىشقا ئۇنىڭغا (يەنى مۇھەممەد ئەلەيھىسسالامغا) بىر قېتىمدىلا نازىل قىلىنمىدى؟» دېدى. سېنىڭ دىلىڭنى مۇستەھكەم قىلىش ئۈچۈن، ئۇنى پارچە - پارچە نازىل قىلدۇق، ئۇنى ئايرىم – ئايرىم نازىل قىلدۇق
وَلَا يَأْتُونَكَ بِمَثَلٍ إِلَّا جِئْنَاكَ بِالْحَقِّ وَأَحْسَنَ تَفْسِيرًا ( 33 )
ئۇلار (يەنى مۇشرىكلار) قانداق بىر سوئالنى تاشلىمىسۇن، (ئۇنىڭغا قارىتا) بىز ساڭا ھەق جاۋابنى ۋە ھەممىدىن گۈزەل چۈشەندۇرۇشنى نازىل قىلدۇق
الَّذِينَ يُحْشَرُونَ عَلَىٰ وُجُوهِهِمْ إِلَىٰ جَهَنَّمَ أُولَٰئِكَ شَرٌّ مَّكَانًا وَأَضَلُّ سَبِيلًا ( 34 )
ئۇلار جەھەننەمگە يۈزلىرى بىلەن سۈرۈلىدۇ، ئۇلارنىڭ جايى ئەڭ يامان، يولى ئەڭ ئازغۇندۇر
وَلَقَدْ آتَيْنَا مُوسَى الْكِتَابَ وَجَعَلْنَا مَعَهُ أَخَاهُ هَارُونَ وَزِيرًا ( 35 )
شۈبھىسىزكى، بىز مۇساغا كىتاب (يەنى تەۋراتنى) ئاتا قىلدۇق ھەمدە ئۇنىڭ قېرىندىشى ھارۇننى ئۇنىڭغا ياردەمچى قىلدۇق
فَقُلْنَا اذْهَبَا إِلَى الْقَوْمِ الَّذِينَ كَذَّبُوا بِآيَاتِنَا فَدَمَّرْنَاهُمْ تَدْمِيرًا ( 36 )
ئۇلارغا: «بىزنىڭ ئايەتلىرىمىزنى ئىنكار قىلغان قەۋمگە بېرىڭلار» دېدۇق، (ئۇ قەۋم كۇفرىدا چىڭ تۇرغانلىقتىن) ئۇلارنى دەھشەتلىك تۈردە ھالاك قىلدۇق
وَقَوْمَ نُوحٍ لَّمَّا كَذَّبُوا الرُّسُلَ أَغْرَقْنَاهُمْ وَجَعَلْنَاهُمْ لِلنَّاسِ آيَةً ۖ وَأَعْتَدْنَا لِلظَّالِمِينَ عَذَابًا أَلِيمًا ( 37 )
نۇھنىڭ قەۋمى پەيغەمبەرلەرنى ئىنكار قىلغان چاغدا ئۇلارنى غەرق قىلدۇق ھەمدە ئۇلارنى كىشىلەرگە ئىبرەت قىلدۇق، زالىملارغا (ئاخىرەتتە) قاتتىق ئازاب تەييارلىدۇق
وَعَادًا وَثَمُودَ وَأَصْحَابَ الرَّسِّ وَقُرُونًا بَيْنَ ذَٰلِكَ كَثِيرًا ( 38 )
ئاد قەۋمىنى، سەمۇد قەۋمىنى، رەس ئاھالىسىنى ۋە ئۇلارنىڭ ئارىسىدىكى نۇرغۇن ئۈممەتلەرنى ھەم (ھالاك قىلدۇق)
وَكُلًّا ضَرَبْنَا لَهُ الْأَمْثَالَ ۖ وَكُلًّا تَبَّرْنَا تَتْبِيرًا ( 39 )
ھەربىر قەۋمگە نۇرغۇن مىساللارنى بايان قىلدۇق، (ۋەز - نەسىھەت ئۈنۈم بەرمىگەنلىكتىن) ھەربىر قەۋمنى ھالاك قىلدۇق
وَلَقَدْ أَتَوْا عَلَى الْقَرْيَةِ الَّتِي أُمْطِرَتْ مَطَرَ السَّوْءِ ۚ أَفَلَمْ يَكُونُوا يَرَوْنَهَا ۚ بَلْ كَانُوا لَا يَرْجُونَ نُشُورًا ( 40 )
ئۇلار ھەقىقەتەن (شامغا بېرىپ كېلىشتە) يامان يامغۇر (يەنى تاش) ياغدۇرۇلغان شەھەردىن ئۆتىدۇ، ئەجەبا ئۇلار ئۇنى كۆرۈپ (ئىبرەت ئالمامدۇ؟) ياق (يەنى ئىبرەت ئالمايدۇ)، ئۇلار (قىيامەت كۈنى) قايتا تىرىلىشنى ئۈمىد قىلمايدۇ
وَإِذَا رَأَوْكَ إِن يَتَّخِذُونَكَ إِلَّا هُزُوًا أَهَٰذَا الَّذِي بَعَثَ اللَّهُ رَسُولًا ( 41 )
ئۇلار سېنى كۆرسە پەقەت مەسخىرە قىلىۋېلىپ: «اﷲ پەيغەمبەرقىلىپ ئەۋەتكەن مۇشۇمۇ؟ ئەگەر دىنىمىزدا چىڭ تۇرمىساق (مۇھەممەد) بىزنى ئىلاھلىرىمىزندىن ئازدۇرۇۋەتكىلى تاس قالاتتى» (دېدى). كىمنىڭ يولىنىڭ ئەڭ خاتا ئىكەنلىكىنى ئۇلار (ئاخىرەتتە) ئازابنى كۆرگەن چاغدا بىلىدۇ
إِن كَادَ لَيُضِلُّنَا عَنْ آلِهَتِنَا لَوْلَا أَن صَبَرْنَا عَلَيْهَا ۚ وَسَوْفَ يَعْلَمُونَ حِينَ يَرَوْنَ الْعَذَابَ مَنْ أَضَلُّ سَبِيلًا ( 42 )
ئۇلار سېنى كۆرسە پەقەت مەسخىرە قىلىۋېلىپ: «اﷲ پەيغەمبەرقىلىپ ئەۋەتكەن مۇشۇمۇ؟ ئەگەر دىنىمىزدا چىڭ تۇرمىساق (مۇھەممەد) بىزنى ئىلاھلىرىمىزندىن ئازدۇرۇۋەتكىلى تاس قالاتتى» (دېدى). كىمنىڭ يولىنىڭ ئەڭ خاتا ئىكەنلىكىنى ئۇلار (ئاخىرەتتە) ئازابنى كۆرگەن چاغدا بىلىدۇ
أَرَأَيْتَ مَنِ اتَّخَذَ إِلَٰهَهُ هَوَاهُ أَفَأَنتَ تَكُونُ عَلَيْهِ وَكِيلًا ( 43 )
ئېيتىپ باقساڭچۇ؟ نەپسى خاھىشىنى ئىلاھ قىلىۋالغان ئادەمگە (نەپسى خاھىشىغا ئەگىشىشتىن ئۇنى ساقلاش ئۈچۈن) ھامىي بولالامسەن
أَمْ تَحْسَبُ أَنَّ أَكْثَرَهُمْ يَسْمَعُونَ أَوْ يَعْقِلُونَ ۚ إِنْ هُمْ إِلَّا كَالْأَنْعَامِ ۖ بَلْ هُمْ أَضَلُّ سَبِيلًا ( 44 )
ئۇلارنىڭ تولىسىنى (گېپىڭنى) ئاڭلايدىغان ياكى چۈشىنىدىغان ئادەملەر دەپ گۇمان قىلامسەن؟ ئۇلار (چۈشەنمەسلىكتە) پەقەت چاھارپايلارغا ئوخشاشتۇر، بەلكى ئۇلاردىنمۇ بەتتەر گۇمراھتۇر
أَلَمْ تَرَ إِلَىٰ رَبِّكَ كَيْفَ مَدَّ الظِّلَّ وَلَوْ شَاءَ لَجَعَلَهُ سَاكِنًا ثُمَّ جَعَلْنَا الشَّمْسَ عَلَيْهِ دَلِيلًا ( 45 )
پەرۋەردىگارىڭنىڭ سايىنى قانداق سوزغانلىقىنى كۆرمىدىڭمۇ؟ ئەگەر خالىسا ئۇنى ئەلۋەتتە مۇقىم قىلاتتى، ئاندىن قۇياشنى سايىگە دەلىل قىلدۇق (يەنى كۈننىڭ چىققانلىقىنى سايىنىڭ بارلىقىغا دەلىل قىلدۇق)
ثُمَّ قَبَضْنَاهُ إِلَيْنَا قَبْضًا يَسِيرًا ( 46 )
ئاندىن ئۇنى ئاستا – ئاستا يوقاتتۇق
وَهُوَ الَّذِي جَعَلَ لَكُمُ اللَّيْلَ لِبَاسًا وَالنَّوْمَ سُبَاتًا وَجَعَلَ النَّهَارَ نُشُورًا ( 47 )
اﷲ سىلەرگە كېچىنى لىباس، ئۇيقۇنى (بەدىنىڭلار ئۈچۈن) راھەت قىلدى، كۈندۈزنى (تىرىكچىلىك ئۈچۈن زېمىندا) يېيىلىدىغان ۋاقىت قىلدى
وَهُوَ الَّذِي أَرْسَلَ الرِّيَاحَ بُشْرًا بَيْنَ يَدَيْ رَحْمَتِهِ ۚ وَأَنزَلْنَا مِنَ السَّمَاءِ مَاءً طَهُورًا ( 48 )
اﷲ رەھمىتىنى (يەنى يامغۇرنى) ياغدۇرۇش ئالدىدا شامالنى خۇش خەۋەر قىلىپ ئەۋەتتى، يامغۇر بىلەن ئۆلۈك زېمىننى تىرىلدۈرۈش ۋە بىز ياراتقان ھايۋانلارنى، نۇرغۇن ئىنسانلارنى سۇ بىلەن تەمىنلەش ئۈچۈن، بۇلۇتتىن پاك سۇنى چۈشۈرۈپ بەردۇق
لِّنُحْيِيَ بِهِ بَلْدَةً مَّيْتًا وَنُسْقِيَهُ مِمَّا خَلَقْنَا أَنْعَامًا وَأَنَاسِيَّ كَثِيرًا ( 49 )
اﷲ رەھمىتىنى (يەنى يامغۇرنى) ياغدۇرۇش ئالدىدا شامالنى خۇش خەۋەر قىلىپ ئەۋەتتى، يامغۇر بىلەن ئۆلۈك زېمىننى تىرىلدۈرۈش ۋە بىز ياراتقان ھايۋانلارنى، نۇرغۇن ئىنسانلارنى سۇ بىلەن تەمىنلەش ئۈچۈن، بۇلۇتتىن پاك سۇنى چۈشۈرۈپ بەردۇق
وَلَقَدْ صَرَّفْنَاهُ بَيْنَهُمْ لِيَذَّكَّرُوا فَأَبَىٰ أَكْثَرُ النَّاسِ إِلَّا كُفُورًا ( 50 )
ئۇلارنى ۋەز - نەسىھەت ئالسۇن دەپ، يامغۇرنى ئۇلارنىڭ ئارىسىدا تەقسىم قىلدۇق (يەنى گاھ ئۇ يەرگە، گاھ بۇ يەرگە ياغدۇردۇق)، ئىنسانلارنىڭ تولىسى (اﷲ قا) كۇفرىلىق قىلىدۇ
وَلَوْ شِئْنَا لَبَعَثْنَا فِي كُلِّ قَرْيَةٍ نَّذِيرًا ( 51 )
ئەگەر خالىساق ئەلۋەتتە ھەر شەھەرگە بىر ئاگاھلاندۇرغۇچى (يەنى پەيغەمبەر) ئەۋەتەتتۇق
فَلَا تُطِعِ الْكَافِرِينَ وَجَاهِدْهُم بِهِ جِهَادًا كَبِيرًا ( 52 )
شۇنىڭ ئۈچۈن كاپىرلارغا ئىتائەت قىلمىغىن، قۇرئان ئارقىلىق ئۇلارغا قارشى (پاكىتلارنى ئوتتۇرىغا قويۇپ) بارلىق كۈچۈڭ بىلەن كۈرەش قىلغىن
وَهُوَ الَّذِي مَرَجَ الْبَحْرَيْنِ هَٰذَا عَذْبٌ فُرَاتٌ وَهَٰذَا مِلْحٌ أُجَاجٌ وَجَعَلَ بَيْنَهُمَا بَرْزَخًا وَحِجْرًا مَّحْجُورًا ( 53 )
اﷲ (قۇدرىتى بىلەن) بىرى تاتلىق بولغان دەريانى، بىرى تۇزلۇق بولغان دېڭىزنى (ئۇلار بىر - بىرىگە تۇتىشاڭغۇ بولغان ھالدا (قويۇۋەتتى) ئارلىشىپ كەتمەسلىكى ئۈچۈن (ئۇلارنىڭ ئارىسىدا پەردە، توسما بەرپا قىلدى
وَهُوَ الَّذِي خَلَقَ مِنَ الْمَاءِ بَشَرًا فَجَعَلَهُ نَسَبًا وَصِهْرًا ۗ وَكَانَ رَبُّكَ قَدِيرًا ( 54 )
اﷲ ئىنساننى ئابىمەنىدىن ياراتتى، ئۇنى بالىلار نىسبەت بېرىلىدىغان ئەر ۋە قۇدا - باجىلىق پەيدا قىلىدىغان ئايال (دىن ئىبارەت ئىككى خىل قىلىپ) ياراتتى، پەرۋەردىگارىڭ ھەممىگە قادىردۇر
وَيَعْبُدُونَ مِن دُونِ اللَّهِ مَا لَا يَنفَعُهُمْ وَلَا يَضُرُّهُمْ ۗ وَكَانَ الْكَافِرُ عَلَىٰ رَبِّهِ ظَهِيرًا ( 55 )
ئۇلار (يەنى مۇشرىكلار) اﷲ نى قويۇپ ئۇلارغا پايدىمۇ يەتكۈزەلمەيدىغان، زىيانمۇ يەتكۈزەلمەيدىغان نەرسىلەرگە چوقۇنىدۇ، كاپىر پەرۋەردىگارىغا ئاسىيلىق قىلىش بىلەن (شەيتانغا) ياردەم بەرگۈچىدۇر
وَمَا أَرْسَلْنَاكَ إِلَّا مُبَشِّرًا وَنَذِيرًا ( 56 )
سېنى پەقەت (جەننەت بىلەن) خۇش خەۋەر بەرگۈچى، (دوزاختىن) ئاگاھلاندۇرغۇچى قىلىپ ئەۋەتتۇق
قُلْ مَا أَسْأَلُكُمْ عَلَيْهِ مِنْ أَجْرٍ إِلَّا مَن شَاءَ أَن يَتَّخِذَ إِلَىٰ رَبِّهِ سَبِيلًا ( 57 )
ئېيتقىنكى، «مەن (اﷲ نىڭ ئەمرىنى) تەبلىغ قىلغانلىقىمغا سىلەردىن ھېچ ھەق تەلەپ قىلمايمەن، پەقەت (مېنىڭ تىلەيدىغىنىم شۇكى، مېنىڭ دىنىمغا ئەگىشىش بىلەن (پەرۋەردىگارى تەرەپكە يول ئېلىشنى خالايدىغانلار) يول ئالسۇن)»
وَتَوَكَّلْ عَلَى الْحَيِّ الَّذِي لَا يَمُوتُ وَسَبِّحْ بِحَمْدِهِ ۚ وَكَفَىٰ بِهِ بِذُنُوبِ عِبَادِهِ خَبِيرًا ( 58 )
(ھەممە ئىشىڭدا) ئۆلمەيدىغان مەڭگۈ ھايات اﷲ قا يۆلەنگىن، ئۇنىڭغا ھەمد ئېيتىش بىلەن اﷲ نى پاك دەپ ئېتىقاد قىلغىن، ئۇ بەندىلەرنىڭ گۇناھلىرىنى تولۇق بىلىشتە يېتەرلىكتۇر
الَّذِي خَلَقَ السَّمَاوَاتِ وَالْأَرْضَ وَمَا بَيْنَهُمَا فِي سِتَّةِ أَيَّامٍ ثُمَّ اسْتَوَىٰ عَلَى الْعَرْشِ ۚ الرَّحْمَٰنُ فَاسْأَلْ بِهِ خَبِيرًا ( 59 )
اﷲ ئاسمانلارنى زېمىننى ۋە ئۇلارنىڭ ئارىسىدىكى نەرسىلەرنى ئالتە كۈندە ياراتتى، ئاندىن ئەرش ئۈستىدە (ئۆزىگە لايىق رەۋىشتە) قارار ئالدى، اﷲ ناھايىتى مەرھەمەتلىكتۇر، بۇنى (يەنى اﷲ نىڭ بۈيۈكلۈكىنى) بىلىدىغان ئادەمدىن سورىغىن (ساڭا اﷲ نىڭ سۈپەتلىرىنى ئېيتىپ بېرىدۇ)
وَإِذَا قِيلَ لَهُمُ اسْجُدُوا لِلرَّحْمَٰنِ قَالُوا وَمَا الرَّحْمَٰنُ أَنَسْجُدُ لِمَا تَأْمُرُنَا وَزَادَهُمْ نُفُورًا ۩ ( 60 )
ئۇلارغا (يەنى مۇشرىكلارغا): «رەھمانغا (يەنى مەرھەمەتلىك اﷲ قا) سەجدە قىلىڭلار!» دېيىلسە، ئۇلار: «رەھمان دېگەن نېمە؟ سەن بىزنى بۇيرۇغان نەرسىگە سەجدە قىلامدۇق؟» دەيدۇ. بۇ سۆز ئۇلارنى (ئىماندىن) تېخىمۇ يىراقلاشتۇرىدۇ
تَبَارَكَ الَّذِي جَعَلَ فِي السَّمَاءِ بُرُوجًا وَجَعَلَ فِيهَا سِرَاجًا وَقَمَرًا مُّنِيرًا ( 61 )
ئاسماندا بۇرۇجلەرنى، يېنىپ تۇرىدىغان چىراقنى (يەنى كۈننى) ۋە نۇرلۇق ئاينى ياراتقان اﷲ نىڭ بەرىكىتى بۈيۈكتۇر
وَهُوَ الَّذِي جَعَلَ اللَّيْلَ وَالنَّهَارَ خِلْفَةً لِّمَنْ أَرَادَ أَن يَذَّكَّرَ أَوْ أَرَادَ شُكُورًا ( 62 )
اﷲ (ئۆزىنىڭ قۇدرىتىنى) ئەسلىمەكچى بولغانلار ياكى (ئۆزىنىڭ نېمىتىگە) شۈكۈر قىلماقچى بولغانلار ئۈچۈن كېچە بىلەن كۈندۈزنى ئالمىشىپ تۇرىدىغان قىلىپ بەردى
وَعِبَادُ الرَّحْمَٰنِ الَّذِينَ يَمْشُونَ عَلَى الْأَرْضِ هَوْنًا وَإِذَا خَاطَبَهُمُ الْجَاهِلُونَ قَالُوا سَلَامًا ( 63 )
مەرھەمەتلىك اﷲ نىڭ (ياخشى كۆرىدىغان) بەندىلىرى زېمىندا ئۆزلىرىنى تۆۋەن تۇتۇپ تەمكىنلىك بىلەن ماڭىدۇ، نادانلار ئۇلارغا (ياقتۇرمايدىغان) سۆز قىلسا، ئۇلار: «سىلەرگە ئامانلىق تىلەيمىز» دەيدۇ (يەنى گۇناھ بولمايدىغان سۆزلەرنى قىلىدۇ)
وَالَّذِينَ يَبِيتُونَ لِرَبِّهِمْ سُجَّدًا وَقِيَامًا ( 64 )
ئۇلار كېچىلەرنى پەرۋەردىگارىغا سەجدە قىلىش ۋە قىيامدا تۇرۇش بىلەن) يەنى ناماز ئوقۇش بىلەن) ئۆتكۈزىدۇ
وَالَّذِينَ يَقُولُونَ رَبَّنَا اصْرِفْ عَنَّا عَذَابَ جَهَنَّمَ ۖ إِنَّ عَذَابَهَا كَانَ غَرَامًا ( 65 )
ئۇلار (يەنى اﷲ ياخشى كۆرىدىغان بەندىلەر) ئېيتىدۇ: «پەرۋەردىگارىمىز! بىزدىن جەھەننەم ئازابىنى دەپئى قىلغىن، جەھەننەمنىڭ ئازابى ھەقىقەتەن (سېنىڭ دۈشمەنلىرىڭدىن) ئايرىلمايدۇ
إِنَّهَا سَاءَتْ مُسْتَقَرًّا وَمُقَامًا ( 66 )
جەھەننەم ھەقىقەتەن يامان قارارگاھتۇر، يامان جايدۇر»
وَالَّذِينَ إِذَا أَنفَقُوا لَمْ يُسْرِفُوا وَلَمْ يَقْتُرُوا وَكَانَ بَيْنَ ذَٰلِكَ قَوَامًا ( 67 )
ئۇلار (يەنى اﷲ ياخشى كۆرىدىغان بەندىلەر) خىراجەت قىلغاندا، ئىسراپچىلىقمۇ قىلمايدۇ، بېخىللىقمۇ قىلمايدۇ، ئوتتۇراھال خىراجەت قىلىدۇ
وَالَّذِينَ لَا يَدْعُونَ مَعَ اللَّهِ إِلَٰهًا آخَرَ وَلَا يَقْتُلُونَ النَّفْسَ الَّتِي حَرَّمَ اللَّهُ إِلَّا بِالْحَقِّ وَلَا يَزْنُونَ ۚ وَمَن يَفْعَلْ ذَٰلِكَ يَلْقَ أَثَامًا ( 68 )
ئۇلار اﷲ قا ئىككىنچى بىر مەبۇدنى شېرىك قىلمايدۇ، اﷲ ھارام قىلغان ناھەق ئادەم ئۆلتۈرۈش ئىشىنى قىلمايدۇ، زىنا قىلمايدۇ، كىمكى بۇ (گۇناھلار) نى قىلىدىكەن، (ئاخىرەتتە) ئۇ جازاغا ئۇچرايدۇ
يُضَاعَفْ لَهُ الْعَذَابُ يَوْمَ الْقِيَامَةِ وَيَخْلُدْ فِيهِ مُهَانًا ( 69 )
قىيامەت كۈنى ئۇنىڭغا ھەسسىلەپ ئازاب قىلىنىدۇ، ئۇ مەڭگۈ ئازاب ئىچىدە خارلانغان ھالدا قالىدۇ
إِلَّا مَن تَابَ وَآمَنَ وَعَمِلَ عَمَلًا صَالِحًا فَأُولَٰئِكَ يُبَدِّلُ اللَّهُ سَيِّئَاتِهِمْ حَسَنَاتٍ ۗ وَكَانَ اللَّهُ غَفُورًا رَّحِيمًا ( 70 )
پەقەت (ئۇلارنىڭ ئىچىدىن) (بۇ دۇنيادىكى چېغىدا) تەۋبە قىلغان، ئىمان ئېيتقان ۋە ياخشى ئەمەللەرنى قىلغانلارلا بۇ ھالدا قالمايدۇ، اﷲ ئۇلارنىڭ گۇناھلىرىنى ياخشىلىققا ئالماشتۇرىدۇ، اﷲ تولىمۇ مەغپىرەت قىلغۇچىدۇر ۋە مەرھەمەت قىلغۇچىدۇر
وَمَن تَابَ وَعَمِلَ صَالِحًا فَإِنَّهُ يَتُوبُ إِلَى اللَّهِ مَتَابًا ( 71 )
كىمكى (گۇناھلىرىغا) تەۋبە قىلىدىكەن ۋە ئەمەلىنى تۈزەيدىكەن، ئۇ اﷲ قا يۈزلەنگەن بولىدۇ (يەنى اﷲ ئۇ نىڭ تەۋبىسىنى قوبۇل قىلىپ، ئۇنىڭدىن رازى بولىدۇ)
وَالَّذِينَ لَا يَشْهَدُونَ الزُّورَ وَإِذَا مَرُّوا بِاللَّغْوِ مَرُّوا كِرَامًا ( 72 )
ئۇلار (يەنى اﷲ ياخشى كۆرىدىغان بەندىلەر) يالغان گۇۋاھلىق بەرمەيدۇ، يامان سۆزنى ئاڭلاپ قالغان چاغدا ئالىيجانابلىق بىلەن ئۆتۈپ كېتىدۇ
وَالَّذِينَ إِذَا ذُكِّرُوا بِآيَاتِ رَبِّهِمْ لَمْ يَخِرُّوا عَلَيْهَا صُمًّا وَعُمْيَانًا ( 73 )
ئۇلارغا پەرۋەردىگارىنىڭ ئايەتلىرى بىلەن ۋەز - نەسىھەت قىلىنسا، گاس، كور بولۇۋالمايدۇ (يەنى ئۇنى بېرىلىپ ئىخلاس بىلەن ئاڭلايدۇ)
وَالَّذِينَ يَقُولُونَ رَبَّنَا هَبْ لَنَا مِنْ أَزْوَاجِنَا وَذُرِّيَّاتِنَا قُرَّةَ أَعْيُنٍ وَاجْعَلْنَا لِلْمُتَّقِينَ إِمَامًا ( 74 )
ئۇلار: «ئى پەرۋەردىگارىمىز! بىزگە ئاياللىرىمىز ۋە ئەۋلاتلىرىمىز ئارقىلىق شادلىق بېغىشلىشىڭنى (يەنى بىزگە ساڭا ئىتائەتمەن پەرزەنت ئاتا قىلىشىڭنى) تىلەيمىز، بىزنى تەقۋادارلارنىڭ پېشىۋاسى (يەنى تەقۋادارلارنىڭ نەمۇنىسى، ياخشىلىققا دەۋەت قىلغۇچى) قىلغىن» دەيدۇ
أُولَٰئِكَ يُجْزَوْنَ الْغُرْفَةَ بِمَا صَبَرُوا وَيُلَقَّوْنَ فِيهَا تَحِيَّةً وَسَلَامًا ( 75 )
ئەنە شۇلار سەۋرلىك بولغانلىقى ئۈچۈن جەننەت بىلەن مۇكاپاتلىنىدۇ، ئۇلار جەننەتتە (پەرىشتىلەر تەرىپىدىن قىلىنغان) دۇئا ۋە سالام بىلەن قارشى ئېلىنىدۇ
خَالِدِينَ فِيهَا ۚ حَسُنَتْ مُسْتَقَرًّا وَمُقَامًا ( 76 )
ئۇلار جەننەتتە مەڭگۈ قالىدۇ، جەننەت نېمىدېگەن گۈزەل قارارگاھ! نېمىدېگەن گۈزەل جاي!
قُلْ مَا يَعْبَأُ بِكُمْ رَبِّي لَوْلَا دُعَاؤُكُمْ ۖ فَقَدْ كَذَّبْتُمْ فَسَوْفَ يَكُونُ لِزَامًا ( 77 )
ئېيتقىنكى، «ئەگەر سىلەرنىڭ دۇئايىڭلار بولمىسا، پەرۋەردىگارىم سىلەرگە پەرۋا قىلمايدۇ، (ئى كاپىرلار! پەيغەمبەرنى) ئىنكار قىلدىڭلار، (ئاخىرەتتە) سىلەر قۇتۇلالمايدىغان ئازابقا دۇچار بولۇسىلەر»
كىتابلار
مەشھۇر ئۆلىمالارنى ئەيىبلىمەسلىك توغرىسىدا «مەشھۇر ئۆلىمالارنى ئەيىبلىمەسلىك توغرىسىدا» ناملىق بۇ ئەسەردە ، ئىسلام ئۈممىتى تەرىپىدىن ئېتىراپ قىلىنغان ئىماملارنىڭ ھېچبىرى قەستەن ھالدا پەيغەمبەر ئەلەيھىسسالامنىڭ مەيلى كىچىك، مەيلى چوڭ بىر سۈننىتىگە قارشى ھەرىكەت قىلمايدىغانلىقى، ئۇلار ھەر تۈرلۈك شۈبھە ۋە ئىككېلىنىشلەردىن يىراق ھالدا پەيغەمبەر ئەلەيھىسسالامغا ئەگىشىشنىڭ ۋاجىبلىقى، پەيغەمبەر ئەلەيھىسسالامدىن باشقا ھەر قانداق كىشىنىڭ سۆزىنىڭ توغرا-خاتاسىنىڭ بولىدىغانلىقى، بۇ ئىماملارنىڭ بىرىنىڭ سەھىھ ھەدىسكە زىت بىر سۆزى بولۇپ قالسا، ئۇنىڭ مەزكۇر ھەدىسنى تەرك قىلىش خۇسۇسىدا جەزمەن قوبۇل قىلغىلى بولىدىغان بىرەر سەۋەبنى دەلىل ئىسپات بىلەن ئوتتۇرىغا قويىدىغانلىقى ۋە شۇ ئارقىلىق باشقىلارنىڭ بۈيۈك ئۆلىمالارغا تىل ئۇزۇتىشىنىڭ ئالدىنى ئالىدىغانلىقى قاتارلىقلار توغرىسىدا چۈشەنچە بىرىدۇ.

پەزىلەتلىك ئەمەللەر «پەزىلەتلىك ئەمەللەر» ناملىق بۇ كىتاپتا ئاپتۇر ئىسلام دىنىدىكى ئىبادەتلەر ئەقىلغا تايانماستىن ؛بەلكى قۇرئان ۋە ھەدىسگە ئاساسلىنىدىغانلىقىنى بايان قىلىش بىلەن بىرگە، بەندىنى ئاللاھقا يىقىنلاشتۇرىدىغان، ئىنساننىڭ روھى ۋە مەنىۋىيىتىگە ۋاستسىز تەسىر كۆرسىتىدىغان ئەمەللەرنى قۇرئان ۋە ساغلام ھەدىسلەرگە ئاساسلىنىپ تۇرۇپ چۈشەنچە بېىرىدۇ.

مۇتتەقىيلەر سەيلىگاھى «ياخشىلار باغچىسى» نىڭ شەرھى مۇتتەقىيلەر سەيلىگاھى «ياخشىلار باغچىسى» نىڭ شەرھى ناملىق بۇ ئەسەر بىر گۇرۇپ ئۆلىمالار تەرىپىدىن شەرھىلەنگەن بولۇپ، كىتاب ئۆزىنىڭ پايدىلىق ۋە ئىلمى قىممىتىنىڭ يۇقىرىلىقى بىلەن ھەممىگە مەلۇم.
يازما : شەيخ مۇستەپا بەغا - شەيخ مۇھەممەد ئەمىن لۇتپى - شەيخ ئەلى شەربىجى - شەيخ مۇھيىددىن مەستۇ
تەكشۇرۇپ بېكىتكۈچى : ن.تەمكىنى
تەرجىمە قىلغۈچى : مۇھەممەد سالىھ قاراجىم
نەشرى ماتىرياللىرىدىن : ئۇيغۇر تەرجىمە مەركىزى
