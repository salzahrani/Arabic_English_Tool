ދިވެހި
Sorah Yunus ( Jonah ) - Verses Number 109
الر ۚ تِلْكَ آيَاتُ الْكِتَابِ الْحَكِيمِ ( 1 )
ا ل ر (މި އަކުރުތަކުގެ مراد ދެނެވޮޑިގެންވަނީ اللَّه އެވެ.) މިއީ حكمة ތެރި ފޮތުގެ (އެބަހީ: قرآن ގެ) آية ތަކެވެ.
أَكَانَ لِلنَّاسِ عَجَبًا أَنْ أَوْحَيْنَا إِلَىٰ رَجُلٍ مِّنْهُمْ أَنْ أَنذِرِ النَّاسَ وَبَشِّرِ الَّذِينَ آمَنُوا أَنَّ لَهُمْ قَدَمَ صِدْقٍ عِندَ رَبِّهِمْ ۗ قَالَ الْكَافِرُونَ إِنَّ هَٰذَا لَسَاحِرٌ مُّبِينٌ ( 2 )
މީސްތަކުންނަށް إنذار ކުރެއްވުމަށާއި، إيمان ވެއްޖެ މީސްތަކުންނަށް އެއުރެންގެ ވެރިރަސްކަލާނގެ حضرة ގައި އިސްކަމާއި، މަތިވެރި مقام ހުރިކަމުގެ އުފާވެރި خبر ދެއްވުމަށް، ތިމަންރަސްކަލާނގެ އެއުރެންގެ ތެރެއިން ފިރިހެނަކަށް، وحى އިއްވެވިކަން މީސްތަކުންނަށް عجائب އަކަށްވީހެއްޔެވެ؟ كافر ވި މީހުން ބުނެފޫއެވެ. ހަމަކަށަވަރުން، މިއީ (އެބަހީ: محمد ގެފާނީ) ފާޅުވެ بيان ވެގެންވާ ޖާދޫގަރެކެވެ.
إِنَّ رَبَّكُمُ اللَّهُ الَّذِي خَلَقَ السَّمَاوَاتِ وَالْأَرْضَ فِي سِتَّةِ أَيَّامٍ ثُمَّ اسْتَوَىٰ عَلَى الْعَرْشِ ۖ يُدَبِّرُ الْأَمْرَ ۖ مَا مِن شَفِيعٍ إِلَّا مِن بَعْدِ إِذْنِهِ ۚ ذَٰلِكُمُ اللَّهُ رَبُّكُمْ فَاعْبُدُوهُ ۚ أَفَلَا تَذَكَّرُونَ ( 3 )
ހަމަކަށަވަރުން، ތިޔަބައިމީހުންގެ ވެރިރަސްކަލާނގެއީ، اللَّه އެވެ. އެކަލާނގެއީ، އުޑުތަކާއި ބިން ހަ ދުވަހުގެ ތެރޭގައި ހެއްދެވި ރަސްކަލާނގެއެވެ. ދެން عرش ގެ މައްޗަށް އެކަލާނގެ استواء ވެވޮޑިގެންވިއެވެ. އެކަލާނގެ (އެންމެހާ) ކަންތައްތައް މިންވަރު ކުރައްވައި تدبير ކުރައްވަތެވެ. އެކަލާނގެ إذن ފުޅަށް ފަހުގައި މެނުވީ (އެއްވެސް މީހަކަށް شفاعة ތެރިވާނޭ) شفاعة ތެރިޔަކު ނުވެއެވެ. އެއީ ތިޔަބައިމީހުންގެ ވެރި اللَّه އެވެ. ފަހެ، ތިޔަބައިމީހުން އެކަލާނގެއަށް އަޅުކަން ކުރާށެވެ! ފަހެ، ތިޔަބައިމީހުން ހަނދުމަ ނުކުރަނީހެއްޔެވެ؟
إِلَيْهِ مَرْجِعُكُمْ جَمِيعًا ۖ وَعْدَ اللَّهِ حَقًّا ۚ إِنَّهُ يَبْدَأُ الْخَلْقَ ثُمَّ يُعِيدُهُ لِيَجْزِيَ الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ بِالْقِسْطِ ۚ وَالَّذِينَ كَفَرُوا لَهُمْ شَرَابٌ مِّنْ حَمِيمٍ وَعَذَابٌ أَلِيمٌ بِمَا كَانُوا يَكْفُرُونَ ( 4 )
ތިޔަބައިމީހުން އެންމެން رجوع ވުން ވަނީ، އެކަލާނގެ حضرة އަށެވެ. (މިއީ) اللَّه ގެ حق وعد ފުޅެވެ. ހަމަކަށަވަރުން، ނެތުމުގެ ތެރެއިން خلق ތަކުން އަލަށް އުފައްދަވައި، ދެން އަލުން އެ خلق ތަކުން إعادة ކުރައްވަނީ، އެކަލާނގެއެވެ. (އެއީ) إيمان ވެ صالح عمل ތައް ކުޅަމީހުންނަށް عدل ވެރި ގޮތުން ޖަޒާ ދެއްވުމަށްޓަކައެވެ. އަދި كافر ވި މީހުންނަށް އެއުރެން كافر ވެ އުޅުނުކަމުގެ ސަބަބުން، ކެކިކެކި ހުންނަ ހޫނުފެނުގެ ބުއިމަކާއި، ވޭންދެނިވި عذاب ހުށްޓެވެ.
هُوَ الَّذِي جَعَلَ الشَّمْسَ ضِيَاءً وَالْقَمَرَ نُورًا وَقَدَّرَهُ مَنَازِلَ لِتَعْلَمُوا عَدَدَ السِّنِينَ وَالْحِسَابَ ۚ مَا خَلَقَ اللَّهُ ذَٰلِكَ إِلَّا بِالْحَقِّ ۚ يُفَصِّلُ الْآيَاتِ لِقَوْمٍ يَعْلَمُونَ ( 5 )
އެކަލާނގެއީ، އަލިކަމެއްކަމުގައި އިރު ލައްވައި، ނޫރެއްކަމުގައި ހަނދު ލެއްވި ރަސްކަލާނގެއެވެ. އަދި އެ ހަނދު ހިނގުމަށް منـزل ތަކެއް މިންވަރުކުރެއްވިއެވެ. (އެއީ) ތިޔަބައިމީހުން އަހަރުތަކުގެ عدد އާއި (މައްސަރުތަކާއި ދުވަސްތަކުގެ) حساب ދެނެގަތުމަށްޓަކައެވެ. حق ގޮތުން މެނުވީ، اللَّه، އެތަކެތި ނަހައްދަވައެވެ. އެނގޭބަޔަކަށް އެކަލާނގެ آية ތައް تفصيل ކޮށްދެއްވަތެވެ.
إِنَّ فِي اخْتِلَافِ اللَّيْلِ وَالنَّهَارِ وَمَا خَلَقَ اللَّهُ فِي السَّمَاوَاتِ وَالْأَرْضِ لَآيَاتٍ لِّقَوْمٍ يَتَّقُونَ ( 6 )
ހަމަކަށަވަރުން، ރޭދުވާ ތަފާތުވުމުގައްޔާއި އުޑުތަކާއި، ބިމުގައި اللَّه ހެއްދެވި (އެންމެހައި) ތަކެތީގައި، تقوى ވެރިވާ ބަޔަކަށް دليل ތަކެއްވާކަން ކަށަވަރެވެ.
إِنَّ الَّذِينَ لَا يَرْجُونَ لِقَاءَنَا وَرَضُوا بِالْحَيَاةِ الدُّنْيَا وَاطْمَأَنُّوا بِهَا وَالَّذِينَ هُمْ عَنْ آيَاتِنَا غَافِلُونَ ( 7 )
ހަމަކަށަވަރުން، ރޭދުވާ ތަފާތުވުމުގައްޔާއި އުޑުތަކާއި، ބިމުގައި اللَّه ހެއްދެވި (އެންމެހައި) ތަކެތީގައި، تقوى ވެރިވާ ބަޔަކަށް دليل ތަކެއްވާކަން ކަށަވަރެވެ.
أُولَٰئِكَ مَأْوَاهُمُ النَّارُ بِمَا كَانُوا يَكْسِبُونَ ( 8 )
ހަމަަށަވަރުން، ތިމަންރަސްކަލާނގެއާ ބައްދަލުވާން ނޭދޭ، އަދި ދުނިޔޭގެ ދިރިއުޅުމަށްރުހި، އަދި އެކަމަށް ހިތްހަމަޖައްސައިގެންތިބެ ތިމަންރަސްކަލާނގެ آية ތަކާމެދު غافل ވި މީހުން (ދަންނާށެވެ!) އެއުރެން ތިބެނިވި ތަނަކީ، ނަރަކައެވެ. (އެއީ) އެއުރެން ހޯދައިއުޅުނު عمل ތަކުގެ ސަބަބުންނެވެ.
إِنَّ الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ يَهْدِيهِمْ رَبُّهُم بِإِيمَانِهِمْ ۖ تَجْرِي مِن تَحْتِهِمُ الْأَنْهَارُ فِي جَنَّاتِ النَّعِيمِ ( 9 )
ހަމަކަށަވަރުން، إيمان ވެ، صالح عمل ތައް ކުޅަމީހުންނަށް، އެއުރެންގެ إيمان ކަމުގެ ސަބަބުން، އެއުރެންގެ ވެރިރަސްކަލާނގެ އެއުރެންނަށް ތެދުމަގު ދައްކަވާނެތެވެ. نعمة ގެ ސުވަރުގެތަކުގައި، އެއުރެންގެ ދަށުން ފެންކޯރުތައް ހިނގަހިނގާ ހުންނާނެތެވެ.
دَعْوَاهُمْ فِيهَا سُبْحَانَكَ اللَّهُمَّ وَتَحِيَّتُهُمْ فِيهَا سَلَامٌ ۚ وَآخِرُ دَعْوَاهُمْ أَنِ الْحَمْدُ لِلَّهِ رَبِّ الْعَالَمِينَ ( 10 )
އެތާނގައި، އެއުރެންގެ دعاء އަކީ [سبحانك اللّهم] އެވެ. އޭ ހައްދަވާ ބޮޑުކުރެއްވި ރަސްކަލާކޯއެވެ! އިބަރަސްކަލާނގެއަށް އަޅަމެން تسبيح ދަންނަވަމެވެ. އަދި އެތާނގައި، އެއުރެންގެ تحيّة އަކީ، [سلام] އެވެ. އަދި އެއުރެންގެ دعاء ގެ ނިންމުމަކީ، [الحمد للَّه رب العالمين] އެވެ. حمد ހުރީ عالم ތަކުގެ ވެރި اللَّه އަށެވެ.
وَلَوْ يُعَجِّلُ اللَّهُ لِلنَّاسِ الشَّرَّ اسْتِعْجَالَهُم بِالْخَيْرِ لَقُضِيَ إِلَيْهِمْ أَجَلُهُمْ ۖ فَنَذَرُ الَّذِينَ لَا يَرْجُونَ لِقَاءَنَا فِي طُغْيَانِهِمْ يَعْمَهُونَ ( 11 )
ހެޔޮކަމާއި، ثواب އަވަސްކޮށްދެއްވުމަށް މީސްތަކުން އެދޭހިނދު، އެކަން އަވަސްކޮށްދެއްވާ ފަދައިން، اللَّه އެއުރެންނަށް ނުބަޔާއި، عقاب އަވަސްކޮށްދެއްވާނަމަ، އެއުރެންގެ أجل ޖެހި، އެހިނދު އެހިނދަށް ނިމިއްޖައީހެވެ. ފަހެ، (قيامة ދުވަހުން) ތިމަންރަސްކަލާނގެއާ ބައްދަލުވާން ނޭދޭމީހުން އެއުރެންގެ ފުރެދުމުގައި، ދެފުށު އެޅިއެޅިތިބޭ حال، އެއުރެން ދޫކޮށްލައްވައި، ބައިތިއްބަވާހުށީމެވެ.
وَإِذَا مَسَّ الْإِنسَانَ الضُّرُّ دَعَانَا لِجَنبِهِ أَوْ قَاعِدًا أَوْ قَائِمًا فَلَمَّا كَشَفْنَا عَنْهُ ضُرَّهُ مَرَّ كَأَن لَّمْ يَدْعُنَا إِلَىٰ ضُرٍّ مَّسَّهُ ۚ كَذَٰلِكَ زُيِّنَ لِلْمُسْرِفِينَ مَا كَانُوا يَعْمَلُونَ ( 12 )
އަދި އިންސާނާއަށް އުނދަގުލެއް ޖެހިއްޖެނަމަ، އޮށޯވެ އޮވެގެން ނުވަތަ އިށީނދެ އިނދެގެން، ނުވަތަ ކޮޅަށް ހުރެގެން، ތިމަންރަސްކަލާނގެ حضرة އަށް دعاء ކުރެތެވެ. ފަހެ، ތިމަންރަސްކަލާނގެ އެމީހެއްގެ ކިބައިން އެ އުނދަގޫ ފިއްލަވައިފުމުން، އޭނާ (އެދާ މަގުން) ހިނގައިދެއެވެ. އުނދަގުލެއްޖެހިގެން ތިމަންރަސްކަލާނގެ حضرة އަށް دعاء ވެސް ނުކުރާކަހަލައެވެ. كافر ންނަށް އެއުރެން ކޮށްއުޅުނު عمل ތައް ޒީނަތްތެރިި ކުރައްވާފައިވަނީ އެފަދައިންނެވެ.
وَلَقَدْ أَهْلَكْنَا الْقُرُونَ مِن قَبْلِكُمْ لَمَّا ظَلَمُوا ۙ وَجَاءَتْهُمْ رُسُلُهُم بِالْبَيِّنَاتِ وَمَا كَانُوا لِيُؤْمِنُوا ۚ كَذَٰلِكَ نَجْزِي الْقَوْمَ الْمُجْرِمِينَ ( 13 )
އަދި ހަމަކަށަވަރުން، ތިޔަބައިމީހުންގެ ކުރީގެ أمّة ތަކުގެ މީހުންގެ ގާތަށް معجزات ތަކާއިގެން رسول ން ވަޑައިގަތުމުން، އެއުރެން އަނިޔާވެރިވީހިނދު، އެ أمّة ތައް ހަލާކު ކުރައްވައި ނެތިކުރެއްވީމެވެ. އެއުރެންނީ إيمان ވާނޭ ބަޔަކު ކަމުގައި ނުވެތެވެ. ކުށްވެރިވި މީހުންނަށް، ތިމަންރަސްކަލާނގެ ޖަޒާ ދެއްވަނީ އެފަދައިންނެވެ.
ثُمَّ جَعَلْنَاكُمْ خَلَائِفَ فِي الْأَرْضِ مِن بَعْدِهِمْ لِنَنظُرَ كَيْفَ تَعْمَلُونَ ( 14 )
ދެން އެއުރެންގެ ފަހުން ބިމުގައި خليفة އިން ކަމުގައި ތިމަންރަސްކަލާނގެ ތިޔަބައިމީހުން ލެއްވީމެވެ. އެއީ ތިޔަބައިމީހުން ކަންތައްކުރަނީ ކޮންފަދައަކުންތޯ ބެއްލެވުމަށްޓަކައެވެ.
وَإِذَا تُتْلَىٰ عَلَيْهِمْ آيَاتُنَا بَيِّنَاتٍ ۙ قَالَ الَّذِينَ لَا يَرْجُونَ لِقَاءَنَا ائْتِ بِقُرْآنٍ غَيْرِ هَٰذَا أَوْ بَدِّلْهُ ۚ قُلْ مَا يَكُونُ لِي أَنْ أُبَدِّلَهُ مِن تِلْقَاءِ نَفْسِي ۖ إِنْ أَتَّبِعُ إِلَّا مَا يُوحَىٰ إِلَيَّ ۖ إِنِّي أَخَافُ إِنْ عَصَيْتُ رَبِّي عَذَابَ يَوْمٍ عَظِيمٍ ( 15 )
އަދި އެއުރެންނަށް ތިމަންރަސްކަލާނގެ آية ތައް بيان ވެގެންވާ ގޮތުގައި ކިޔަވައިދެއްވާހިނދު، ތިމަންރަސްކަލާނގެއާ ބައްދަލުވުމަށް ނޭދޭމީހުން ބުނެތެވެ. ތިޔަނޫން އެހެން قرآن އެއް ގެނެސްދެއްވާށެވެ! ނުވަތަ އެ قرآن އަށް ބަދަލުގެނެސް ދެއްވާށެވެ! ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިމަންކަލޭގެފާނު އަމިއްލަފުޅަށް އެ ބަދަލުކުރުމަކީ، ތިމަންކަލޭގެފާނަށް އޮތްކަމެއް ނޫނެވެ. ތިމަންކަލޭގެފާނަށް އިއްވަވާ وحى އަށް މެނުވީ، ތިމަންކަލޭގެފާނު تبع ވެވަޑައެއް ނުގަންނަވަމެވެ. ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެއަށް ތިމަންކަލޭގެފާނު އުރެދިއްޖެނަމަ، ބޮޑުވެގެންވާ ދުވަހެއްގެ عذاب އަށް ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނު ބިރުވެތިވެވަޑައިގަންނަވަމެވެ.
قُل لَّوْ شَاءَ اللَّهُ مَا تَلَوْتُهُ عَلَيْكُمْ وَلَا أَدْرَاكُم بِهِ ۖ فَقَدْ لَبِثْتُ فِيكُمْ عُمُرًا مِّن قَبْلِهِ ۚ أَفَلَا تَعْقِلُونَ ( 16 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! اللَّه އިރާދަކުރެއްވިނަމަ، ތިމަންކަލޭގެފާނު އެ آية ތައް ތިޔަބައިމީހުންނަށް ކިޔައެއް ނުދެއްވީމުހެވެ. އަދި އެކަލާނގެ ތިޔަބައިމީހުންނަށް އެ آية ތަކެއް ނުވެސް އެންގެވީހެވެ. ފަހެ، މީގެކުރިން ތިޔަބައިމީހުންގެ ތެރޭގައި އެތައްދުވަހެއް ވަންދެން (އެބަހީ: ސާޅީސް އަހަރު ވަންދެން) ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނު ހުންނެވީމެވެ. (މިބާވަތުގެ ވާހަކައެއް ނުދައްކަވަމެވެ.) ފަހެ، ތިޔަބައިމީހުންނަކަށް ނުވިސްނޭހެއްޔެވެ؟
فَمَنْ أَظْلَمُ مِمَّنِ افْتَرَىٰ عَلَى اللَّهِ كَذِبًا أَوْ كَذَّبَ بِآيَاتِهِ ۚ إِنَّهُ لَا يُفْلِحُ الْمُجْرِمُونَ ( 17 )
اللَّه އާމެދު ހަދައިގެން ދޮގެއް ބުނިމީހަކަށްވުރެ، ނުވަތަ އެކަލާނގެ آية ތައް ދޮގުކުޅަ މީހަކަށްވުރެ، އަނިޔާވެރިކަންބޮޑު މީހަކީ ފަހެ، ކާކުހެއްޔެވެ؟ ހަމަކަށަވަރުން، ކުށްކުރާ މީހުން ނުދިނޭނެތެވެ.
وَيَعْبُدُونَ مِن دُونِ اللَّهِ مَا لَا يَضُرُّهُمْ وَلَا يَنفَعُهُمْ وَيَقُولُونَ هَٰؤُلَاءِ شُفَعَاؤُنَا عِندَ اللَّهِ ۚ قُلْ أَتُنَبِّئُونَ اللَّهَ بِمَا لَا يَعْلَمُ فِي السَّمَاوَاتِ وَلَا فِي الْأَرْضِ ۚ سُبْحَانَهُ وَتَعَالَىٰ عَمَّا يُشْرِكُونَ ( 18 )
އެއުރެން اللَّه ފިޔަވައި އަޅުކަންކުރަނީ، އެއުރެންނަށް ގެއްލުމެއްވެސް ނުދެވޭ، އަދި އެއުރެންނަށް މަންފާއެއްވެސް ނުދެވޭ ތަކެއްޗަށެވެ. އަދި އެއުރެން ބުނެތެވެ. މި ތަކެއްޗަކީ، اللَّه ގެ حضرة ގައި ތިމަންމެންނަށް شفاعة ތެރިވާނޭ ތަކެއްޗެވެ. ކަލޭގެފާނު ވިދާޅުވާށެވެ! އުޑުތަކުގައިވެސް، އަދި ބިމުގައިވެސް اللَّه ދެނެވޮޑިގެންނުވާ ކަމެއްގެ ވާހަކައިން ތިޔަބައިމީހުން އެކަލާނގެއަށް خبر ދެނީހެއްޔެވެ؟ އެއުރެން އެ شريك ކުރާ ތަކެތިން އެކަލާނގެ ހުސްطاهر ވެ މަތިވެރިވެވޮޑިގެންވެއެވެ.
وَمَا كَانَ النَّاسُ إِلَّا أُمَّةً وَاحِدَةً فَاخْتَلَفُوا ۚ وَلَوْلَا كَلِمَةٌ سَبَقَتْ مِن رَّبِّكَ لَقُضِيَ بَيْنَهُمْ فِيمَا فِيهِ يَخْتَلِفُونَ ( 19 )
އަދި މީސްތަކުން (އެންމެފުރަތަމަ) އެއް أمّة އެއްކަމުގައި މެނުވީ ނުވެއެވެ. ދެން (އެއަށްފަހު) އެއުރެން خلاف ވޫއެވެ. ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ حضرة ން (ޖަޒާދެއްވުން قيامة ދުވަހަށް ފަސްކުރެއްވުމުގެ) ބަސްފުޅެއް އިސްވެފައި ނުވާނަމަ، އެކަންތަކާ ދޭތެރޭ އެއުރެން خلاف ވާ ކަންތަކާމެދު (ދުނިޔެމަތީގައި) އެއުރެންގެ މެދުގައި ނިޔާކުރެއްވީހެވެ.
وَيَقُولُونَ لَوْلَا أُنزِلَ عَلَيْهِ آيَةٌ مِّن رَّبِّهِ ۖ فَقُلْ إِنَّمَا الْغَيْبُ لِلَّهِ فَانتَظِرُوا إِنِّي مَعَكُم مِّنَ الْمُنتَظِرِينَ ( 20 )
އަދި އެއުރެން ބުނެތެވެ. އެކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ حضرة ން އެކަލޭގެފާނަށް معجزة އެއް ބާވައިލައްވާފައި ނުވަނީ ކީއްވެގެންބާވައެވެ! ފަހެ، ކަލޭގެފާނު ވިދާޅުވާށެވެ! غيب ތައް ދެނެވޮޑިގަތުންވާކަން ކަށަވަރީ، اللَّه އަށެވެ. ފަހެ، ތިޔަބައިމީހުން انتظار ކުރާށެވެ! ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނުވެސް ތިޔަބައިމީހުންނާއެކު انتظار ކުރާ މީހުންގެ ތެރެއިން ވާހުށީމެވެ.
وَإِذَا أَذَقْنَا النَّاسَ رَحْمَةً مِّن بَعْدِ ضَرَّاءَ مَسَّتْهُمْ إِذَا لَهُم مَّكْرٌ فِي آيَاتِنَا ۚ قُلِ اللَّهُ أَسْرَعُ مَكْرًا ۚ إِنَّ رُسُلَنَا يَكْتُبُونَ مَا تَمْكُرُونَ ( 21 )
އަދި މީސްތަކުންނަށް ތަދުމަޑު حال އެއް ޖެހުމަށްފަހު، އެއުރެންނަށް ތިމަންރަސްކަލާނގެ رحمة ލައްވައި، ތިޔާގިކަމުގެ ރަހަ ދައްކަވާހިނދު، ތިމަންރަސްކަލާނގެ آية ތައް އެއުރެން ދޮގުކޮށް މަކަރު ހަދަތެވެ. ކަލޭގެފާނު ވިދާޅުވާށެވެ! އެއުރެންގެ މަކަރު ރައްދުކުރެއްވުމުގައި، اللَّه އަވަސްވެވޮޑިގަތުން ބޮޑެވެ. ހަމަކަށަވަރުން، ތިމަންރަސްކަލާނގެ ملائكة ން ތިޔަބައިމީހުން ހަދާ މަކަރުތައް ލިޔުއްވަތެވެ.
هُوَ الَّذِي يُسَيِّرُكُمْ فِي الْبَرِّ وَالْبَحْرِ ۖ حَتَّىٰ إِذَا كُنتُمْ فِي الْفُلْكِ وَجَرَيْنَ بِهِم بِرِيحٍ طَيِّبَةٍ وَفَرِحُوا بِهَا جَاءَتْهَا رِيحٌ عَاصِفٌ وَجَاءَهُمُ الْمَوْجُ مِن كُلِّ مَكَانٍ وَظَنُّوا أَنَّهُمْ أُحِيطَ بِهِمْ ۙ دَعَوُا اللَّهَ مُخْلِصِينَ لَهُ الدِّينَ لَئِنْ أَنجَيْتَنَا مِنْ هَٰذِهِ لَنَكُونَنَّ مِنَ الشَّاكِرِينَ ( 22 )
އެކަލާނގެއީ، އެއްގަމާއި، ކަނޑުގައި، ތިޔަބައިމީހުން ދަތުރުކުރާގޮތް ހިންމަވައިދެއްވާ ރަސްކަލާނގެއެވެ. ދެން ތިޔަަބައިމީހުން، ނަވަށް އަރައި ނަލަ ރޯޅިއަކާއެކު އެއުރެންނާއިގެން އެ ނައު ދުއްވައިގަތުމުން، އެކަމާ އެއުރެން އުފާކޮށްކޮށް ތިއްބައި، އެ ނަވަށް ޖެހުނުގަދަ ކޮޅިގަނޑަކުން ހުރިހާ ފަރާތަކުން އެއުރެންނާ ދިމާޔަށް ރާޅުތައް އައިސް އެއުރެން ހަލާކުވެދާނެހެން ހީވީމާ، އެކަލާނގެއަށް إيمان ކަން خالص ކުރާ حال ގައި، اللَّه އަށް އެއުރެން دعاء ކުރެތެވެ. މިކަމުން، އިބަރަސްކަލާނގެ އަޅަމެން ސަލާމަތްކުރައްވައިފިނަމަ، ހަމަކަށަވަރުން، شكر ކުރާ މީހުންގެ ތެރެއިން އަޅަމެން ވާހުށީމުއެވެ.
فَلَمَّا أَنجَاهُمْ إِذَا هُمْ يَبْغُونَ فِي الْأَرْضِ بِغَيْرِ الْحَقِّ ۗ يَا أَيُّهَا النَّاسُ إِنَّمَا بَغْيُكُمْ عَلَىٰ أَنفُسِكُم ۖ مَّتَاعَ الْحَيَاةِ الدُّنْيَا ۖ ثُمَّ إِلَيْنَا مَرْجِعُكُمْ فَنُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ ( 23 )
ދެން އެކަލާނގެ އެއުރެން ސަލާމަތްކުރެއްވި ހިނދު، އެހާރުން، حق އަކާނުލައި، އެއުރެން ބިމުގައި فساد އުފައްދައި ދެކޮޅުވެރިވެތެވެ. އޭ މީސްތަކުންނޭވެ! ތިޔަބައިމީހުންގެ ދެކޮޅުވެރިކަމުގެ ގެއްލުން ހުރިކަންކަށަވަރީ، ހަމަ ތިޔަބައިމީހުންގެ އަމިއްލަ نفس ތަކަށެވެ. ދުނިޔޭގެ ދިރިއުޅުމުގެ ވަގުތީ އުފަލެއްގެ ގޮތުގައެވެ. ދެން ތިޔަބައިމީހުން رجوع ވާހުށީ، ތިމަންރަސްކަލާނގެ حضرة އަށެވެ. އޭރުން ތިޔަބައިމީހުން ކޮށްއުޅުނު ކަންތަކުގެ خبر، ތިމަންރަސްކަލާނގެ ތިޔަބައިމީހުންނަށް އަންގަވާހުށީމެވެ.
إِنَّمَا مَثَلُ الْحَيَاةِ الدُّنْيَا كَمَاءٍ أَنزَلْنَاهُ مِنَ السَّمَاءِ فَاخْتَلَطَ بِهِ نَبَاتُ الْأَرْضِ مِمَّا يَأْكُلُ النَّاسُ وَالْأَنْعَامُ حَتَّىٰ إِذَا أَخَذَتِ الْأَرْضُ زُخْرُفَهَا وَازَّيَّنَتْ وَظَنَّ أَهْلُهَا أَنَّهُمْ قَادِرُونَ عَلَيْهَا أَتَاهَا أَمْرُنَا لَيْلًا أَوْ نَهَارًا فَجَعَلْنَاهَا حَصِيدًا كَأَن لَّمْ تَغْنَ بِالْأَمْسِ ۚ كَذَٰلِكَ نُفَصِّلُ الْآيَاتِ لِقَوْمٍ يَتَفَكَّرُونَ ( 24 )
ދުނިޔޭގެ ދިރިއުޅުމުގެ مثال ވާކަން ކަށަވަރީ، ތިމަންރަސްކަލާނގެ އުޑުން ވެއްސެވި ފެނެއްގެ مثال ކަމުގައެވެ. ފަހެ، އިންސާނުންނާއި، نعم ސޫފި ކައިއުޅޭ ބާވަތްތަކުގެ ތެރެއިން ބިމުގެ ގަސްގަހާގެހި، އެ ވާރޭފެނާއެކުވެ، އެންމެފަހުން އެ ބިމަށް ރީތިކަމާއި، އާރޯކަންލިބި (މާމެލާމެއްޔާއި މޭވާތަކުން) ޒީނަތްތެރިވެ، އެ ބިމުގެ أهل ވެރިން އެ ތަކެތީގެ މައްޗަށް ބާރު ހިންގުމަށް ކުޅަދާނަވެގެންވާކަމަށް ހީކޮށްގެން ތިއްބައި (އެތަކެތި ހަލާކުވެދިއުމަށް) ރޭގަނޑެއްގައި، ނުވަތަ ދުވާލެއްގައި، ތިމަންރަސްކަލާނގެ أمر ފުޅު އައެވެ. ފަހެ، އެހިނދު ކޮށާލައިފައިވާ ގޮވާންތަކެއް ފަދައިން، ތިމަންރަސްކަލާނގެ އެތަކެތި ލެއްވީމެވެ. އެތަކެތި އިއްޔެ ނުވެސް ނެތްކަހަލައެވެ. ފިކުރުކުރާ ބަޔަކަށް، ތިމަންރަސްކަލާނގެ آية ތައް تفصيل ކޮށްދެއްވަނީ އެފަދައިންނެވެ.
وَاللَّهُ يَدْعُو إِلَىٰ دَارِ السَّلَامِ وَيَهْدِي مَن يَشَاءُ إِلَىٰ صِرَاطٍ مُّسْتَقِيمٍ ( 25 )
اللَّه ގޮވައިލައްވަނީ، ސަލާމަތުގެ ގޮވައްޗަށެވެ. (އެބަހީ: ސުވަރުގެއަށެވެ.) އަދި އެކަލާނގެ އިރާދަކުރައްވާ ބަޔަކަށް، ސީދާވެގެންވާ ތެދުމަގަށް އެކަލާނގެ މަގު ދައްކަވަތެވެ.
لِّلَّذِينَ أَحْسَنُوا الْحُسْنَىٰ وَزِيَادَةٌ ۖ وَلَا يَرْهَقُ وُجُوهَهُمْ قَتَرٌ وَلَا ذِلَّةٌ ۚ أُولَٰئِكَ أَصْحَابُ الْجَنَّةِ ۖ هُمْ فِيهَا خَالِدُونَ ( 26 )
ހެޔޮ عمل ކުޅަމީހުންނަށް، އެންމެހެޔޮ ޖަޒާ ހުށްޓެވެ. (އެއީ ސުވަރުގެއެވެ.) އަދި އޭގެއިތުރަށްވެސް ހުށްޓެވެ. (އެއީ اللَّه ގެ ދީދާރުފުޅު ދެއްކެވުމެވެ.) އަދި އެއުރެންގެ މޫނުތަކަކަށް ހިރަފުހެއް ނުވަތަ ދެރަކަމެއް ނާރާނެތެވެ. އެއުރެންނަކީ، ސުވަރުގޭގެ أهل ވެރިންނެވެ. އެެއުރެން އެތާނގައި ދެމިތިބޭ ހުށްޓެވެ.
وَالَّذِينَ كَسَبُوا السَّيِّئَاتِ جَزَاءُ سَيِّئَةٍ بِمِثْلِهَا وَتَرْهَقُهُمْ ذِلَّةٌ ۖ مَّا لَهُم مِّنَ اللَّهِ مِنْ عَاصِمٍ ۖ كَأَنَّمَا أُغْشِيَتْ وُجُوهُهُمْ قِطَعًا مِّنَ اللَّيْلِ مُظْلِمًا ۚ أُولَٰئِكَ أَصْحَابُ النَّارِ ۖ هُمْ فِيهَا خَالِدُونَ ( 27 )
އަދި ނުބައި عمل ތައް ކުރި މީހުންނަށްހުރީ، ކޮންމެ ނުބައިކަމަކަށް އެފަދަ ޖަޒާއެވެ. އަދި އެއުރެންގެ މައްޗަށް ފެތުރިފައިވަނީ ނިކަމެތިކަމެވެ. اللَّه ގެ عذاب ން ރައްކާތެރިކުރާނޭ އެއްވެސް ފަރާތެއް އެއުރެންނަކަށް ނުވެއެވެ. އެއުރެންގެ މޫނުތައް، ކަޅު އަނދިރި ރޭގަނޑުން ބައެއްލައްވައި ނިވާކުރައްވާފައިވާ ކަހަލައެވެ. އެއުރެންނަކީ، ނަރަކައިގެ أهل ވެރިންނެވެ. އެއުރެން އެތާނގައި ދެމިތިބޭހުށްޓެވެ.
وَيَوْمَ نَحْشُرُهُمْ جَمِيعًا ثُمَّ نَقُولُ لِلَّذِينَ أَشْرَكُوا مَكَانَكُمْ أَنتُمْ وَشُرَكَاؤُكُمْ ۚ فَزَيَّلْنَا بَيْنَهُمْ ۖ وَقَالَ شُرَكَاؤُهُم مَّا كُنتُمْ إِيَّانَا تَعْبُدُونَ ( 28 )
ތިމަންރަސްކަލާނގެ އެއުރެން އެންމެން އެއްތަނަކަށް އެއްކުރައްވާ ދުވަސް (ހަނދުމަކުރާށެވެ!) ދެން شريك ކުޅަ މީހުންނަށް، ތިމަންރަސްކަލާނގެ وحى ކުރައްވާހުށީމެވެ. ތިޔަބައިމީހުންނާއި، ތިޔަބައިމީހުން شريك ކުޅަ ތަކެތި ތިބިތާނގައި ތިބޭށެވެ! ފަހެ، އެއުރެންގެ މެދުގައިވާ ގުޅުންތައް ތިމަންރަސްކަލާނގެ ވަނީ ކަނޑުއްވައި ވަކިކުރައްވާފައެވެ. އަދި އެއުރެން شريك ކުޅަ ތަކެތި ބުނާނެތެވެ. ތިޔަބައިމީހުން އަޅުކަންކޮށް އުޅުނީ ތިމަންމެންނަކަށް ނޫނެވެ.
فَكَفَىٰ بِاللَّهِ شَهِيدًا بَيْنَنَا وَبَيْنَكُمْ إِن كُنَّا عَنْ عِبَادَتِكُمْ لَغَافِلِينَ ( 29 )
ފަހެ، ތިމަންމެންނާއި، ތިޔަބައިމީހުންނާ ދެމެދުގައި ހެތްކަކަށް اللَّه ފުދެއެވެ. ހަމަކަށަވަރުން، ތިޔަބައިމީހުން ކުޅަ އަޅުކަމުން ތިމަންމެންވަނީ غافل ވެގެންނެވެ.
هُنَالِكَ تَبْلُو كُلُّ نَفْسٍ مَّا أَسْلَفَتْ ۚ وَرُدُّوا إِلَى اللَّهِ مَوْلَاهُمُ الْحَقِّ ۖ وَضَلَّ عَنْهُم مَّا كَانُوا يَفْتَرُونَ ( 30 )
އެތާނގައި ކޮންމެ نفس އަކަށްމެ، އެ نفس އެއް އިސްކުޅަ عمل އެއްގެގޮތް އެނގޭނެތެވެ. އަދި އެއުރެންގެ އެހީތެރިވަންތަ، حق ވަންތަ اللَّه ގެ حضرة އަށް އެއުރެން ރައްދުކުރައްވާނެތެވެ. އަދި އެއުރެން ދޮގުން އުފައްދައިގެން އުޅުނު ކަންތައްތައް، އެއުރެންގެ ކިބައިން ގެއްލިގެން ހިނގައިދާނެތެވެ.
قُلْ مَن يَرْزُقُكُم مِّنَ السَّمَاءِ وَالْأَرْضِ أَمَّن يَمْلِكُ السَّمْعَ وَالْأَبْصَارَ وَمَن يُخْرِجُ الْحَيَّ مِنَ الْمَيِّتِ وَيُخْرِجُ الْمَيِّتَ مِنَ الْحَيِّ وَمَن يُدَبِّرُ الْأَمْرَ ۚ فَسَيَقُولُونَ اللَّهُ ۚ فَقُلْ أَفَلَا تَتَّقُونَ ( 31 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިޔަބައިމީހުންނަށް އުޑުންނާއި، ބިމުން رزق ދެއްވަނީ ކާކުހެއްޔެވެ؟ ނުވަތަ އިވުމާއި ފެނުން ملك ވެގެންވަނީ ކާކަށްހެއްޔެވެ؟ އަދި މަރުވެފައިވާ ތަކެތިން، ދިރޭތަކެތި ނެރުއްވައި، ދިރޭތަކެތިން މަރުވެފައިވާ ތަކެތި ނެރުއްވަނީ ކާކުހެއްޔެވެ؟ އަދި އެންމެހާ ކަންތައް تدبير ކުރައްވަނީ ކާކުހެއްޔެވެ؟ ފަހެ، އެއުރެން ބުނާނެތެވެ. (އެއީ) اللَّه އެވެ. ދެން ކަލޭގެފާނު ވިދާޅުވާށެވެ! ފަހެ، ތިޔަބައިމީހުން (އެކަލާނގެއަށް) تقوى ވެރި ނުވަނީހެއްޔެވެ؟
فَذَٰلِكُمُ اللَّهُ رَبُّكُمُ الْحَقُّ ۖ فَمَاذَا بَعْدَ الْحَقِّ إِلَّا الضَّلَالُ ۖ فَأَنَّىٰ تُصْرَفُونَ ( 32 )
ފަހެ، އެއީ اللَّه އެވެ. ތިޔަބައިމީހުންގެ ވެރި، حق ވަންތަ ރަސްކަލާނގެއެވެ. ފަހެ، حق ގޮތަށްފަހުގައި ވަނީ މަގުފުރެދުންނޫން ކޮންއެއްޗެއް ހެއްޔެވެ؟ (އެބަހީ: حق ގޮތުން އެއްކިބާވެއްޖެނަމަ، ދެން ދެވޭނީ ފުރެދުމުގެ ތެރެއަށެވެ.) ފަހެ، ތިޔަބައިމީހުން (حق ގޮތް ފިޔަވައި) އެނބުރިދަނީ، ކޮންފަދައަކުން ހެއްޔެވެ؟
كَذَٰلِكَ حَقَّتْ كَلِمَتُ رَبِّكَ عَلَى الَّذِينَ فَسَقُوا أَنَّهُمْ لَا يُؤْمِنُونَ ( 33 )
فاسق ވި މީހުންގެ މައްޗަށް، އެއުރެން إيمان ނުވާނެކަމަށް ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ كلمة ފުޅު ހަމައެފަދައިންވަނީ حق ވެގެންނެވެ.
قُلْ هَلْ مِن شُرَكَائِكُم مَّن يَبْدَأُ الْخَلْقَ ثُمَّ يُعِيدُهُ ۚ قُلِ اللَّهُ يَبْدَأُ الْخَلْقَ ثُمَّ يُعِيدُهُ ۖ فَأَنَّىٰ تُؤْفَكُونَ ( 34 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިޔަބައިމީހުން شريك ކުރާ ފަރާތްތަކުގެ ތެރޭގައި (ނެތުމުގެތެރެއިން) خلق ތަކުން އަލަށް އުފައްދައި، ދެން އަލުން އެ خلق ތަކުން إعادة ކުރާ މީހަކު ވޭހެއްޔެވެ؟ ކަލޭގެފާނު ވިދާޅުވާށެވެ! (ނެތުމުގެތެރެއިން) خلق ތަކުން އަލަށް އުފައްދަވައި، ދެން އަލުން އެ خلق ތަކުން إعادة ކުރައްވަނީ اللَّه އެވެ. ފަހެ، ތިޔަބައިމީހުން (إيمان ކަމުން) އެއްކިބާވެގަންނަނީ ކޮންފަދައަކުން ހެއްޔެވެ؟
قُلْ هَلْ مِن شُرَكَائِكُم مَّن يَهْدِي إِلَى الْحَقِّ ۚ قُلِ اللَّهُ يَهْدِي لِلْحَقِّ ۗ أَفَمَن يَهْدِي إِلَى الْحَقِّ أَحَقُّ أَن يُتَّبَعَ أَمَّن لَّا يَهِدِّي إِلَّا أَن يُهْدَىٰ ۖ فَمَا لَكُمْ كَيْفَ تَحْكُمُونَ ( 35 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިޔަބައިމީހުން شريك ކުރާ ފަރާތްތަކުގެ ތެރޭގައި حق އަށް މަގުދައްކާ މީހަކު ވޭހެއްޔެވެ؟ ކަލޭގެފާނު ވިދާޅުވާށެވެ! حق އަށް މަގުދައްކަވަނީ اللَّه އެވެ. ފަހެ، تبع ވުން حق ކަންބޮޑީ حق އަށް މަގުދައްކާ ފަރާތަކާ ހެއްޔެވެ؟ ނުވަތަ މަގުދައްކަވައިގެން މެނުވީ ތެދުމަގު ލިބިނުގަނެވޭ މީހަކާ ހެއްޔެވެ؟ ފަހެ، ތިޔަބައިމީހުންނަށް ތިޔަ ތިބީ ކީއްވެފައި ހެއްޔެވެ؟ ތިޔަބައިމީހުން حكم ކުރަނީ ކޮންފަދައަކުން ހެއްޔެވެ؟
وَمَا يَتَّبِعُ أَكْثَرُهُمْ إِلَّا ظَنًّا ۚ إِنَّ الظَّنَّ لَا يُغْنِي مِنَ الْحَقِّ شَيْئًا ۚ إِنَّ اللَّهَ عَلِيمٌ بِمَا يَفْعَلُونَ ( 36 )
އެއުރެންކުރެ ގިނަމީހުން ހީއަކާމެނުވީ تبع އެއް ނުވެތެވެ. ހަމަކަށަވަރުން، ހީއަކީ حق ގެ ބަދަލުގައި އެއްވެސްކަމެއް ފުއްދައިދޭނޭ އެއްޗެއްނޫނެވެ. ހަމަކަށަވަރުން، اللَّه އީ، އެއުރެން ކުރާ ކަންތައް މޮޅަށް ދެނެވޮޑިގެންވާ ރަސްކަލާނގެއެވެ.
وَمَا كَانَ هَٰذَا الْقُرْآنُ أَن يُفْتَرَىٰ مِن دُونِ اللَّهِ وَلَٰكِن تَصْدِيقَ الَّذِي بَيْنَ يَدَيْهِ وَتَفْصِيلَ الْكِتَابِ لَا رَيْبَ فِيهِ مِن رَّبِّ الْعَالَمِينَ ( 37 )
اللَّه ނޫންފަރާތަކުން ހަދައިގެން ގެންނަ އެއްޗަކަށް މި قرآن ނުވެއެވެ. އެހެނެއްކަމަކު (އެ قرآن ބާވައިލެއްވީ) އޭގެކުރީގައިވާ ފޮތްތައް ތެދުކުރުމަށާއި، ފޮތުގައިވާ حكم ތައް تفصيل ކޮށްދެއްވުމަށެވެ. އެ ފޮތާމެދު شك އެއް ނެތެވެ. (އެ ފޮތް ބާވައިލެއްވީ) عالم ތަކުގެ ވެރިރަސްކަލާނގެ حضرة ންނެވެ.
أَمْ يَقُولُونَ افْتَرَاهُ ۖ قُلْ فَأْتُوا بِسُورَةٍ مِّثْلِهِ وَادْعُوا مَنِ اسْتَطَعْتُم مِّن دُونِ اللَّهِ إِن كُنتُمْ صَادِقِينَ ( 38 )
އަދި ކިއެއްތަ! (އެ قرآن (އެއީ، އެކަލޭގެފާނު ހައްދަވައިގެން އުޅުއްވާ އެއްޗެއްކަމަށް އެއުރެން ބުނަނީހެއްޔެވެ؟ ކަލޭގެފާނު ވިދާޅުވާށެވެ! ފަހެ، ތިޔަބައިމީހުން ތެދުވެރިން ކަމުގައިވަނީނަމަ، اللَّه ފިޔަވައި ތިޔަބައިމީހުންނަށް ކުޅަދާނަވީހައި ބަޔަކަށް ގޮވައިގެން ނަމަވެސް، އެއާ އެއްފަދަ سورة އެއް ނިކަން ގެނެސްބަލާށެވެ!
بَلْ كَذَّبُوا بِمَا لَمْ يُحِيطُوا بِعِلْمِهِ وَلَمَّا يَأْتِهِمْ تَأْوِيلُهُ ۚ كَذَٰلِكَ كَذَّبَ الَّذِينَ مِن قَبْلِهِمْ ۖ فَانظُرْ كَيْفَ كَانَ عَاقِبَةُ الظَّالِمِينَ ( 39 )
އަދި ކިއެއްތަ! އެ قرآن ގައިވާ އެއްޗެއް އެއުރެންނަށް ނޭނގެނީސް، އަދި އޭގެ މާނަ އެއުރެންނާ ހަމަޔަށް ނާންނަނީސް، އެއުރެން އެ قرآن ދޮގުކުރޫއެވެ. ހަމަ އެފަދައިން އެއުރެންގެ ކުރީގެ މީހުންވެސް (އެއުރެންގެ رسول ން) ދޮގުކުރޫއެވެ. ފަހެ، އަނިޔާވެރިންގެ ނިމުންވަނީ، ކޮންފަދައަކުންކަން ކަލޭގެފާނު ދެކެވަޑައި ގަންނަވާށެވެ!
وَمِنْهُم مَّن يُؤْمِنُ بِهِ وَمِنْهُم مَّن لَّا يُؤْمِنُ بِهِ ۚ وَرَبُّكَ أَعْلَمُ بِالْمُفْسِدِينَ ( 40 )
އެއުރެންގެ ތެރޭގައި އެ قرآن އަށް إيمان ވާ ބަޔަކުވެއެވެ. އަދި އެއުރެންގެ ތެރޭގައި އެ قرآن އަށް إيمان ނުވާ ބަޔަކުވެސްވެއެވެ. ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެއީ، فساد ކުރާމީހުން އެންމެމޮޅަށް ދެނެވޮޑިގެންވާ ރަސްކަލާނގެއެވެ.
وَإِن كَذَّبُوكَ فَقُل لِّي عَمَلِي وَلَكُمْ عَمَلُكُمْ ۖ أَنتُم بَرِيئُونَ مِمَّا أَعْمَلُ وَأَنَا بَرِيءٌ مِّمَّا تَعْمَلُونَ ( 41 )
އަދި އެއުރެން ކަލޭގެފާނު ދޮގުކޮށްފިނަމަ، ފަހެ، ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިމަންކަލޭގެފާނަށް، ތިމަންކަލޭގެފާނު ކުރެއްވި عمل އެކެވެ. އަދި ތިޔަބައިމީހުންނަށް ތިޔަބައިމީހުން ކުޅަ عمل އެކެވެ. ތިމަންކަލޭގެފާނު ކުރައްވާ عمل ތަކުން، ތިޔަބައިމީހުންވަނީ، ބަރީއަވެގެންނެވެ. އަދި ތިޔަބައިމީހުން ކުރާ عمل ތަކުން، ތިމަންކަލޭގެފާނު ވަނީ ބަރީއަވެގެންނެވެ.
وَمِنْهُم مَّن يَسْتَمِعُونَ إِلَيْكَ ۚ أَفَأَنتَ تُسْمِعُ الصُّمَّ وَلَوْ كَانُوا لَا يَعْقِلُونَ ( 42 )
އަދި އެއުރެންގެ ތެރޭގައި، ކަލޭގެފާނަށް ކަނުލައިއަހާ މީހުންވެއެވެ. ފަހެ، ބީރުބަޔަކަށް، އެއުރެންނަށް ނުވިސްނޭކަމުގައިވިޔަސް، ކަލޭގެފާނު އިއްވަވަނީތޯއެވެ؟
وَمِنْهُم مَّن يَنظُرُ إِلَيْكَ ۚ أَفَأَنتَ تَهْدِي الْعُمْيَ وَلَوْ كَانُوا لَا يُبْصِرُونَ ( 43 )
އަދި އެއުރެންގެ ތެރޭގައި ކަލޭގެފާނަށް ބަލާމީހުންވެއެވެ. ފަހެ، ކަނުބަޔަކަށް، އެއުރެންނަށް ނުފެންނަކަމުގައިވިޔަސް، ކަލޭގެފާނު މަގުދައްކަވަނީތޯއެވެ؟
إِنَّ اللَّهَ لَا يَظْلِمُ النَّاسَ شَيْئًا وَلَٰكِنَّ النَّاسَ أَنفُسَهُمْ يَظْلِمُونَ ( 44 )
ހަމަކަށަވަރުން، اللَّه، އެއްވެސްކަމަކުން މީސްތަކުންނަށް އަނިޔާއެއް ނުކުރައްވަތެވެ. އެހެނެއްކަމަކު، މީސްތަކުން އެއުރެންގެ އަމިއްލަ نفس ތަކަށް އަނިޔާކުރެތެވެ.
وَيَوْمَ يَحْشُرُهُمْ كَأَن لَّمْ يَلْبَثُوا إِلَّا سَاعَةً مِّنَ النَّهَارِ يَتَعَارَفُونَ بَيْنَهُمْ ۚ قَدْ خَسِرَ الَّذِينَ كَذَّبُوا بِلِقَاءِ اللَّهِ وَمَا كَانُوا مُهْتَدِينَ ( 45 )
އެކަލާނގެ އެއުރެން محشر އަށް ފޮނުއްވާ ދުވަހު، ދުވާލުން ساعة އެއްގެ މިންވަރަށް މެނުވީ (ދުނިޔޭގައި ނުވަތަ ކަށްވަޅުގައި) އެއުރެން ނުތިބޭކަހަލައެވެ. އެއުރެންގެ މެދުގައި تعارف ވާނެތެވެ. اللَّه އާ ބައްދަލުވާނޭކަން ދޮގުކުޅަ މީހުން، ގެއްލި ހަލާކުވެއްޖެކަން ކަށަވަރެވެ. އަދި އެއުރެންނީ، ތެދުމަގު ލިބިގެންވާ ބަޔަކު ކަމުގައިވެސް ނުމެވޭމެއެވެ.
وَإِمَّا نُرِيَنَّكَ بَعْضَ الَّذِي نَعِدُهُمْ أَوْ نَتَوَفَّيَنَّكَ فَإِلَيْنَا مَرْجِعُهُمْ ثُمَّ اللَّهُ شَهِيدٌ عَلَىٰ مَا يَفْعَلُونَ ( 46 )
އެއުރެންނަށް ތިމަންރަސްކަލާނގެ وعد ކުރެއްވި ބައެއްކަންތައް ކަލޭގެފާނަށް ދެއްކެވިޔަސް، ނުވަތަ (އޭގެކުރިން) ތިމަންރަސްކަލާނގެ ކަލޭގެފާނު މަރުގެންނެވިޔަސް، ފަހެ، އެއުރެންގެ رجوع ވުންވަނީ، ތިމަންރަސްކަލާނގެ حضرة އަށެވެ. ދެން އެއުރެން ކޮށްއުޅުނު ކަންތަކުގެ މައްޗަށް، اللَّه ހެކިވެވޮޑިގަންނަވާނެތެވެ.
وَلِكُلِّ أُمَّةٍ رَّسُولٌ ۖ فَإِذَا جَاءَ رَسُولُهُمْ قُضِيَ بَيْنَهُم بِالْقِسْطِ وَهُمْ لَا يُظْلَمُونَ ( 47 )
އަދި ކޮންމެ أمّة އަކަށްވެސް رسول އަކުވެއެވެ. ފަހެ، (قيامة ދުވަހުން) އެ أمّة އެއްގެ رسول އަކު ވަޑައިގަތުމުން، އެއުރެންގެ މެދުގައި عدل އަށް حكم ކުރައްވާނެތެވެ. އަދި އެއުރަންނަކަށް ބޭއިންސާފެއް ނުކުރައްވާހުށްޓެވެ.
وَيَقُولُونَ مَتَىٰ هَٰذَا الْوَعْدُ إِن كُنتُمْ صَادِقِينَ ( 48 )
އަދި އެއުރެން ބުނެތެވެ. ތިޔަބައިމީހުންނީ ތެދުބުނާ ބަޔަކު ކަމުގައިވާނަމަ، ތިޔަ (ބުނާ عذاب ގެ) وعد އަންނާނީ ކޮންއިރަކުން ހެއްޔެވެ؟
قُل لَّا أَمْلِكُ لِنَفْسِي ضَرًّا وَلَا نَفْعًا إِلَّا مَا شَاءَ اللَّهُ ۗ لِكُلِّ أُمَّةٍ أَجَلٌ ۚ إِذَا جَاءَ أَجَلُهُمْ فَلَا يَسْتَأْخِرُونَ سَاعَةً ۖ وَلَا يَسْتَقْدِمُونَ ( 49 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިމަންކަލޭގެފާނުގެ އަމިއްލަ نفس އަށް ގެއްލުމެއް ދިނުމަކީވެސް، އަދި މަންފާއެއް ދިނުމަކީވެސް، ތިމަންކަލޭގެފާނުގެ ބާރުގެ ދަށުގައިވާ ކަމެއްނޫނެވެ. اللَّه އިރާދަކުރެއްވިކަމެއް މެނުވީއެވެ. ކޮންމެ أمّة އަކަށްވެސް أجل އެއްވެއެވެ. އެއުރެންގެ أجل އަތުވެއްޖެނަމަ، އެންމެ ساعة އެއްގެ އިރުވެސް އެއުރެންނަކަށް ލަހެއްނުކުރައްވާނެތެވެ. އަދި އަވަހެއްވެސް ނުމެކުރައްވާނޭމެއެވެ.
قُلْ أَرَأَيْتُمْ إِنْ أَتَاكُمْ عَذَابُهُ بَيَاتًا أَوْ نَهَارًا مَّاذَا يَسْتَعْجِلُ مِنْهُ الْمُجْرِمُونَ ( 50 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ތިޔަބައިމީހުން ނިކަން ބުނެބަލާށެވެ! ތިޔަބައިމީހުންގެ ގާތަށް އެކަލާނގެ عذاب، ރޭގަނޑު (ނިދާފައި ތިއްބައި) ނުވަތަ ދުވާލު (މަސައްކަތުގައި ތިއްބައި) އަޔަސް (ތިޔަބައިމީހުންނަށް އޮތީ ކޮންފައިދާއެއް ހެއްޔެވެ؟) ކުށްވެރިން (ލަސްވެގެން) އަވަސްވުންއެދި އެއުޅެނީ އެ عذاب ގެ ތެރެއިން ކޮންކަމެއް ހެއްޔެވެ؟
أَثُمَّ إِذَا مَا وَقَعَ آمَنتُم بِهِ ۚ آلْآنَ وَقَدْ كُنتُم بِهِ تَسْتَعْجِلُونَ ( 51 )
އެ عذاب އައިސްޖެހުނީމާ، ދެން ތިޔަބައިމީހުން އެކަލާނގެއަށް إيمان ވީހެއްޔެވެ؟ (އަދި އަންގަވާނެތެވެ. ތިޔަބައިމީހުން ތިޔަ إيمان ވަނީ) މިހާރު ހެއްޔެވެ؟ ތިޔަބައިމީހުން (ފުރައްސާރައިގެ ގޮތުން) އެ عذاب (ނައިސް ލަސްވެގެން) އަވަސްވުންއެދި އުޅުނީމުއެވެ.
ثُمَّ قِيلَ لِلَّذِينَ ظَلَمُوا ذُوقُوا عَذَابَ الْخُلْدِ هَلْ تُجْزَوْنَ إِلَّا بِمَا كُنتُمْ تَكْسِبُونَ ( 52 )
ދެން އަނިޔާވެރިވި މީހުންނަށް އަންގަވާނެތެވެ. ދެމިހުންނަނިވި عذاب ގެ ރަހަ ތިޔަބައިމީހުން ލިބިގަންނާށެވެ! ތިޔަބައިމީހުން ހޯދައިއުޅުނު عمل ތަކުގެ ސަބަބުން މެނުވީ، ތިޔަބައިމީހުންނަށް ޖަޒާ ލިބޭހެއްޔެވެ؟
وَيَسْتَنبِئُونَكَ أَحَقٌّ هُوَ ۖ قُلْ إِي وَرَبِّي إِنَّهُ لَحَقٌّ ۖ وَمَا أَنتُم بِمُعْجِزِينَ ( 53 )
އެއީ، ހަމަ حق ތެދުތޯ خبر ދެއްވުމަށް، ކަލޭގެފާނުގެ ކިބައިން އެއުރެން އެދޭނެތެވެ. ކަލޭގެފާނު ވިދާޅުވާށެވެ! އާނއެކެވެ! ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ ގަންދެއްވައި ވިދާޅުވަމެވެ. ހަމަކަށަވަރުން، އެއީ، ހަމަ حق ތެދެވެ. އަދި ތިޔަބައިމީހުންނަކީ، އެކަލާނގެ ނުކުޅައްދައި ރެކިގެންދާނޭ ބަޔަކު ކަމުގައި ނުވެތެވެ.
وَلَوْ أَنَّ لِكُلِّ نَفْسٍ ظَلَمَتْ مَا فِي الْأَرْضِ لَافْتَدَتْ بِهِ ۗ وَأَسَرُّوا النَّدَامَةَ لَمَّا رَأَوُا الْعَذَابَ ۖ وَقُضِيَ بَيْنَهُم بِالْقِسْطِ ۚ وَهُمْ لَا يُظْلَمُونَ ( 54 )
އަނިޔާވެރި ކޮންމެ نفس އަކަށް، ބިމުގައިވާހާ ތަކެތި ލިބުނުނަމަ، (عذاب ން މިންޖުވުމަށް) އެތަކެތިން فدية ދިނީހެވެ. އަދި އެއުރެން عذاب ދެކޭހިނދު، ހިތާމަވެރިކަން ހިތުތެރޭގައި إحساس ކުރާނެތެވެ. އަދި އެއުރެންގެ މެދުގައި عدل އަށް ނިޔާކުރައްވާ ހުށްޓެވެ. އަދި އެއުރެންނަކަށް ބޭއިންސާފެއް ނުކުރައްވާ ހުށްޓެވެ.
أَلَا إِنَّ لِلَّهِ مَا فِي السَّمَاوَاتِ وَالْأَرْضِ ۗ أَلَا إِنَّ وَعْدَ اللَّهِ حَقٌّ وَلَٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ ( 55 )
ދަންނާށެވެ! ހަމަކަށަވަރުން، އުޑުތަކާއި، ބިމުގައިވާހާ ތަކެތި ملك ވެގެންވަނީ اللَّه އަށެވެ. ދަންނާށެވެ! ހަމަކަށަވަރުން، اللَّه ގެ وعد ފުޅަކީ حق ތެދެވެ. އެހެނެއްކަމަކު، އެއުރެންކުރެ ގިނަމީހުން (އެކަމެއް) ނުދަނެތެވެ.
هُوَ يُحْيِي وَيُمِيتُ وَإِلَيْهِ تُرْجَعُونَ ( 56 )
ދިރުއްވަނީވެސް، އަދި މަރުގަންނަވަނީވެސް އެކަލާނގެއެވެ. އަދި ތިޔަބައިމީހުން رجوع ކުރައްވާހުށީ، ހަމަ، އެކަލާނގެ حضرة އަށެވެ.
يَا أَيُّهَا النَّاسُ قَدْ جَاءَتْكُم مَّوْعِظَةٌ مِّن رَّبِّكُمْ وَشِفَاءٌ لِّمَا فِي الصُّدُورِ وَهُدًى وَرَحْمَةٌ لِّلْمُؤْمِنِينَ ( 57 )
އޭ މީސްތަކުންނޭވެ! ތިޔަބައިމީހުންގެ ވެރިރަސްކަލާނގެ حضرة ން ތިޔަބައިމީހުންގެ ގާތަށް وعظ އަކާއި، ހިތްތަކުގައިވާ ބަލިތަކަށް شفاء އަކާއި، مؤمن ތަކުންނަށް ތެދުމަގަކާއި، رحمة އެއް އަތުވެއްޖެކަން ކަށަވަރެވެ.
قُلْ بِفَضْلِ اللَّهِ وَبِرَحْمَتِهِ فَبِذَٰلِكَ فَلْيَفْرَحُوا هُوَ خَيْرٌ مِّمَّا يَجْمَعُونَ ( 58 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! اللَّه ގެ فضل ވަންތަ ކަމާއި، އެކަލާނގެ رحمة އަށްޓަކައި (އެބަހީ: إسلام دين އާއި قرآن އަށްޓަކައި) ފަހެ، އެއުރެން އުފާކުރާހުށިކަމެވެ! އެއުރެން (ދުނިޔެމަތީގައި) އެއްކުރާ ތަކެއްޗަށްވުރެ، އެކަން ހެޔޮކަން ބޮޑެވެ.
قُلْ أَرَأَيْتُم مَّا أَنزَلَ اللَّهُ لَكُم مِّن رِّزْقٍ فَجَعَلْتُم مِّنْهُ حَرَامًا وَحَلَالًا قُلْ آللَّهُ أَذِنَ لَكُمْ ۖ أَمْ عَلَى اللَّهِ تَفْتَرُونَ ( 59 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! اللَّه ތިޔަބައިމީހުންނަށް ބާވައިލެއްވި رزق އާމެދު، ތިޔަބައިމީހުން ނިކަން خبر ދީބަލާށެވެ! ފަހެ، ތިޔަބައިމީހުން އޭގެ ބައެއް ތަކެތި حرام ތަކެއްޗަށް އަނެއްބައި ތަކެތި حلال ތަކެއްޗަށް ހެދީމުއެވެ. ކަލޭގެފާނު ވިދާޅުވާށެވެ! اللَّه ތިޔަބައިމީހުންނަށް (އެކަމަށް) إذن ދެއްވީހެއްޔެވެ؟ އަދި ކިއެއްތަ! اللَّه އާމެދު ތިޔަބައިމީހުން ހަދައިގެން ދޮގެއް ބުނަނީއެވެ.
وَمَا ظَنُّ الَّذِينَ يَفْتَرُونَ عَلَى اللَّهِ الْكَذِبَ يَوْمَ الْقِيَامَةِ ۗ إِنَّ اللَّهَ لَذُو فَضْلٍ عَلَى النَّاسِ وَلَٰكِنَّ أَكْثَرَهُمْ لَا يَشْكُرُونَ ( 60 )
اللَّه އާމެދު ހަދައިގެން ދޮގުބުނާ މީހުން، قيامة ދުވަހު، ހީކުރަނީ ކީކޭހެއްޔެވެ؟ ހަމަކަށަވަރުން، اللَّه އީ، މީސްތަކުންގެ މައްޗަށް فضل ވަންތަވެރި ރަސްކަލާނގެކަން ކަށަވަރެވެ. އެހެނެއްކަމަކު، އެއުރެންކުރެ ގިނަމީހުން شكر އެއް ނުކުރެތެވެ.
وَمَا تَكُونُ فِي شَأْنٍ وَمَا تَتْلُو مِنْهُ مِن قُرْآنٍ وَلَا تَعْمَلُونَ مِنْ عَمَلٍ إِلَّا كُنَّا عَلَيْكُمْ شُهُودًا إِذْ تُفِيضُونَ فِيهِ ۚ وَمَا يَعْزُبُ عَن رَّبِّكَ مِن مِّثْقَالِ ذَرَّةٍ فِي الْأَرْضِ وَلَا فِي السَّمَاءِ وَلَا أَصْغَرَ مِن ذَٰلِكَ وَلَا أَكْبَرَ إِلَّا فِي كِتَابٍ مُّبِينٍ ( 61 )
ކަލޭގެފާނު ކޮންމެ ކަމެއްގައި އުޅުއްވިޔަސް، އަދި ކަލޭގެފާނު قرآن ން ކޮންމެ އެއްޗެއް ކިޔަވައިވިދާޅުވިޔަސް، އަދި ތިޔަބައިމީހުން ކޮންމެ عمل އެއް ކުރިޔަސް، އެ عمل ގެ ތެރޭގައި ތިޔަބައިމީހުން އުޅޭހިނދު، ތިޔަބައިމީހުންގެ މައްޗަށް ތިމަންރަސްކަލާނގެ ހެކިވެވޮޑިގެންވާ ކަމުގައި މެނުވީ ނުވަމެވެ. އަދި އުޑާއި، ބިމުގައިވާ ذرّ އެއްގެ މިންވަރުވެސް، އަދި އެއަށްވުރެ ކުޑަޔަސް، އަދި އެއަށްވުރެ ބޮޑަސް، ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެއަށް ވަންހަނާވެގެނެއްނުވެތެވެ. ބަޔާންވެގެންވާ ފޮތެއްގައި އެތަކެތި ލިޔުއްވާފައިމެނުވީ ނުވެތެވެ.
أَلَا إِنَّ أَوْلِيَاءَ اللَّهِ لَا خَوْفٌ عَلَيْهِمْ وَلَا هُمْ يَحْزَنُونَ ( 62 )
ދަންނާށެވެ! ހަމަކަށަވަރުން، اللَّه ގެ ولىّ ންގެ މައްޗަކަށް އެއްވެސް ބިރެއްނެތެވެ. އަދި އެއުރެންނީ ހިތާމަކުރާނޭ ބަޔަކު ކަމުގައިވެސް ނުމެވޭމެއެވެ.
الَّذِينَ آمَنُوا وَكَانُوا يَتَّقُونَ ( 63 )
އެއުރެންނަކީ إيمان ވެ، تقوى ވެރިވެ އުޅުނު މީހުންނެވެ.
لَهُمُ الْبُشْرَىٰ فِي الْحَيَاةِ الدُّنْيَا وَفِي الْآخِرَةِ ۚ لَا تَبْدِيلَ لِكَلِمَاتِ اللَّهِ ۚ ذَٰلِكَ هُوَ الْفَوْزُ الْعَظِيمُ ( 64 )
ދުނިޔޭގެ حياة ގައްޔާއި، آخرة ގައި އެއުރެންނަށް އުފާވެރިކަމުގެ خبر ހުށްޓެވެ. اللَّه ގެ كلمة ފުޅުތަކަށް ބަދަލުވުމެއް ނާންނަހުށްޓެވެ. ބޮޑުވެގެންވާ نصيب ލިބުމަކީ ހަމަ އެއީއެވެ.
وَلَا يَحْزُنكَ قَوْلُهُمْ ۘ إِنَّ الْعِزَّةَ لِلَّهِ جَمِيعًا ۚ هُوَ السَّمِيعُ الْعَلِيمُ ( 65 )
އެއުރެންގެ (ދޮގު) ވާހަކަތައް ކަލޭގެފާނު ހިތާމަވެރިކުރުވިޔަ ނުދެއްވާށެވެ! ހަމަކަށަވަރުން، ހުރިހާ عزّة އެއް ملك ވެގެންވަނީ اللَّه އަށެވެ. އެކަލާނގެއީ، މޮޅަށް އައްސަވާވޮޑިގެންވާ މޮޅަށް ދެނެވޮޑިގެންވާ ރަސްކަލާނގެއެވެ.
أَلَا إِنَّ لِلَّهِ مَن فِي السَّمَاوَاتِ وَمَن فِي الْأَرْضِ ۗ وَمَا يَتَّبِعُ الَّذِينَ يَدْعُونَ مِن دُونِ اللَّهِ شُرَكَاءَ ۚ إِن يَتَّبِعُونَ إِلَّا الظَّنَّ وَإِنْ هُمْ إِلَّا يَخْرُصُونَ ( 66 )
ދަންނާށެވެ! ހަމަކަށަވަރުން، އުޑުތަކުގައިވާހާ ބަޔަކާއި، ބިމުގައިވާހާ ބަޔެއް ملك ވެގެންވަނީ، اللَّه އަށެވެ. اللَّه ނޫންފަރާތަކަށް، އަޅުކަންކުރާ މީހުން (حقيقة ގައި) شريك ންނަކާ تبع އެއް ނުވެތެވެ. އެއުރެން އެ تبع ވަނީ، ހީއަކަށް ކަމުގައިނޫނީ ނުވެތެވެ. އަދި ދޮގުހަދާ ބަޔަކު ކަމުގައި ނޫނީ އެއުރެން ނުވެތެވެ.
هُوَ الَّذِي جَعَلَ لَكُمُ اللَّيْلَ لِتَسْكُنُوا فِيهِ وَالنَّهَارَ مُبْصِرًا ۚ إِنَّ فِي ذَٰلِكَ لَآيَاتٍ لِّقَوْمٍ يَسْمَعُونَ ( 67 )
އެކަލާނގެއީ، ތިޔަބައިމީހުން (އަރާމުކޮށް) ހަމަހިމޭންވުމަށް ތިޔަބައިމީހުންނަށްޓަކައި، ރޭގަނޑު ލެއްވި ރަސްކަލާނގެއެވެ. އަދި (ތިޔަބައިމީހުން ދިރިއުޅުމުގެ މަސައްކަތް ކުރުމަށް) ދުވާލު އަލިކޮށްދެއްވި ރަސްކަލާނގެއެވެ. ހަމަކަށަވަރުން، އެކަންތައްތަކުގައި، އަޑުއަހާ ބަޔަކަށް ހެކިތަކެއްވާކަން ކަށަވަރެވެ.
قَالُوا اتَّخَذَ اللَّهُ وَلَدًا ۗ سُبْحَانَهُ ۖ هُوَ الْغَنِيُّ ۖ لَهُ مَا فِي السَّمَاوَاتِ وَمَا فِي الْأَرْضِ ۚ إِنْ عِندَكُم مِّن سُلْطَانٍ بِهَٰذَا ۚ أَتَقُولُونَ عَلَى اللَّهِ مَا لَا تَعْلَمُونَ ( 68 )
اللَّه ދަރިކަލަކު ހިއްޕެވިކަމަށް އެއުރެން ބުނެފޫއެވެ. އެކަލާނގެ ހުސްطاهر ވަންތަކަމާއެވެ! އެކަލާނގެއީ، ފުދިވޮޑިގެންވާ ރަސްކަލާނގެއެވެ. އުޑުތަކުގައިވާހާ ތަކެއްޗާއި ބިމުގައިވާހާ ތަކެތި ملك ވެގެންވަނީ އެކަލާނގެއަށެވެ. ތިޔަބައިމީހުން ތިޔަބުނާ ބަހުގެ މައްޗަށް ތިޔަބައިމީހުންގެ ގާތުގައި އެއްވެސް دليل އެއް ނެތެވެ. ތިޔަބައިމީހުންނަށް، އެނގިގެންނުވާ އެއްޗެއް اللَّه އާމެދު ތިޔަބައިމީހުން ބުނަނީހެއްޔެވެ؟
قُلْ إِنَّ الَّذِينَ يَفْتَرُونَ عَلَى اللَّهِ الْكَذِبَ لَا يُفْلِحُونَ ( 69 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! ހަމަކަށަވަރުން، اللَّه އާމެދު ހަދައިގެން ދޮގުބުނާ މީހުން ނުދިނޭނެތެވެ.
مَتَاعٌ فِي الدُّنْيَا ثُمَّ إِلَيْنَا مَرْجِعُهُمْ ثُمَّ نُذِيقُهُمُ الْعَذَابَ الشَّدِيدَ بِمَا كَانُوا يَكْفُرُونَ ( 70 )
ދުނިޔޭގައި (ލިބޭ) ވަގުތީ އުފަލެކެވެ. ދެން އެއުރެން رجوع ވުން ވަނީ، ތިމަންރަސްކަލާނގެ حضرة އަށެވެ. ދެން އެއުރެން كافر ވެ އުޅުނުކަމުގެ ސަބަބުން، ގަދަފަދަވެގެންވާ عذاب ގެ ރަހަ، ތިމަންރަސްކަލާނގެ އެއުރެންނަށް ދައްކަވާހުށީމެވެ.
وَاتْلُ عَلَيْهِمْ نَبَأَ نُوحٍ إِذْ قَالَ لِقَوْمِهِ يَا قَوْمِ إِن كَانَ كَبُرَ عَلَيْكُم مَّقَامِي وَتَذْكِيرِي بِآيَاتِ اللَّهِ فَعَلَى اللَّهِ تَوَكَّلْتُ فَأَجْمِعُوا أَمْرَكُمْ وَشُرَكَاءَكُمْ ثُمَّ لَا يَكُنْ أَمْرُكُمْ عَلَيْكُمْ غُمَّةً ثُمَّ اقْضُوا إِلَيَّ وَلَا تُنظِرُونِ ( 71 )
نوح ގެފާނުގެ ވާހަކަ އެއުރެންނަށް ކަލޭގެފާނު ކިޔައިދެއްވާށެވެ! އެކަލޭގެފާނުގެ قوم ގެމީހުންނަށް ވިދާޅުވި ހިނދުގައެވެ. އޭ ތިމަންކަލޭގެފާނުގެ قوم ގެމީހުންނޭވެ! ތިމަންކަލޭގެފާނު ތިޔަބައިމީހުންގެ ތެރޭގައި ހުންނެވުމާއި، اللَّه ގެ آية ތަކުން ތިމަންކަލޭގެފާނު ތިޔަބައިމީހުންނަށް ހަނދާންކޮށް ދެއްވުން، ތިޔަބައިމީހުންގެ މައްޗަށް ބޮޑުކަމަކަށް ވެއްޖެނަމަ، ފަހެ، ތިމަންކަލޭގެފާނު، اللَّه އަށް وكيل ކުރައްވައިފީމެވެ. ދެން ތިޔަބައިމީހުންގެ ބައިވެރިންނާއެކު، ތިޔަބައިމީހުންގެ ކަންތަކާމެދު އެއްބަސްވާށެވެ! ދެން ތިޔަބައިމީހުންގެ ކަންތައް (ނިންމިގޮތް) ތިޔަބައިމީހުންނަށް އޮޅޭގޮތެއްގައި، ވަންހަނާވެފައި އޮވެގެނެއް ނުވާނެއެވެ. ދެން ތިޔަބައިމީހުން ނިންމިގޮތް، ތިމަންކަލޭގެފާނުގެ މައްޗަށް ހިންގާށެވެ! އަދި ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނަށް مهلة ނުދޭށެވެ!
فَإِن تَوَلَّيْتُمْ فَمَا سَأَلْتُكُم مِّنْ أَجْرٍ ۖ إِنْ أَجْرِيَ إِلَّا عَلَى اللَّهِ ۖ وَأُمِرْتُ أَنْ أَكُونَ مِنَ الْمُسْلِمِينَ ( 72 )
ފަހެ، ތިޔަބައިމީހުން ފުރަގަސްދީފިނަމަ (ދަންނާށެވެ!) ފަހެ، ތިޔަބައިމީހުންގެ ކިބަޔަކުން އުޖޫރައަކަށް ތިމަންކަލޭގެފާނު އެދިވަޑައެއް ނުގަންނަވަމެވެ. ތިމަންކަލޭގެފާނުގެ އުޖޫރަ، اللَّه ގެ حضرة ގައި މެނުވީ ނުވެތެވެ. އަދި ތިމަންކަލޭގެފާނަށް أمر ކުރައްވާފައިވަނީ، ތިމަންކަލޭގެފާނު مسلم ންގެ ތެރެއިން ވުމަށެވެ.
فَكَذَّبُوهُ فَنَجَّيْنَاهُ وَمَن مَّعَهُ فِي الْفُلْكِ وَجَعَلْنَاهُمْ خَلَائِفَ وَأَغْرَقْنَا الَّذِينَ كَذَّبُوا بِآيَاتِنَا ۖ فَانظُرْ كَيْفَ كَانَ عَاقِبَةُ الْمُنذَرِينَ ( 73 )
ފަހެ، އެއުރެން އެކަލޭގެފާނު ދޮގުކުރޫއެވެ. ދެން ތިމަންރަސްކަލާނގެ އެކަލޭގެފާނާއި، އެކަލޭގެފާނުގެ އަރިހުގައި ނާކޮޅުގައިވާ އެންމެން ސަލާމަތް ކުރެއްވީމެވެ. އަދި އޭގެފަހުން (ބިމުގައި) އުޅޭނެ ބަޔަކު ކަމުގައި އެއުރެން ލެއްވީމެވެ. އަދި ތިމަންރަސްކަލާނގެ آية ތައް ދޮގުކުރިމީހުން غرق ކުރެއްވީމެވެ. ފަހެ، إنذار ވެވުނު މީހުންގެ ނިމުންހުރީ، ކޮންފަދައަކުންކަން ކަލޭގެފާނު ދެކެވަޑައިގަންނަވާށެވެ!
ثُمَّ بَعَثْنَا مِن بَعْدِهِ رُسُلًا إِلَىٰ قَوْمِهِمْ فَجَاءُوهُم بِالْبَيِّنَاتِ فَمَا كَانُوا لِيُؤْمِنُوا بِمَا كَذَّبُوا بِهِ مِن قَبْلُ ۚ كَذَٰلِكَ نَطْبَعُ عَلَىٰ قُلُوبِ الْمُعْتَدِينَ ( 74 )
ދެން އެކަލޭގެފާނަށް ފަހުގައި ތިމަންރަސްކަލާނގެ رسول ންތަކެއް އެބޭކަލުންގެ قوم ގެމީހުންގެ ގާތަށް ފޮނުއްވީމެވެ. ފަހެ، بيان ވެގެންވާ ހެކިތައް އެއުރެންނަށް އެބޭކަލުން ގެނެސްދެއްވޫއެވެ. (ނަމަވެސް) ކުރިން އެއުރެން ދޮގުކުރި ކަންތަކަށް إيمان ވާ ބަޔަކު ކަމުގައި އެއުރެން ނުވިއެވެ. ހަމައެފަދައިން ދެކޮޅުވެރިވެ ހައްދުފަހަނަޅައިދާ މީހުންގެ ހިތްތަކުގައި ތިމަންރަސްކަލާނގެ ސިއްކަގައްސަވާހުށީމެވެ.
ثُمَّ بَعَثْنَا مِن بَعْدِهِم مُّوسَىٰ وَهَارُونَ إِلَىٰ فِرْعَوْنَ وَمَلَئِهِ بِآيَاتِنَا فَاسْتَكْبَرُوا وَكَانُوا قَوْمًا مُّجْرِمِينَ ( 75 )
ދެން އެބޭކަލުންނަށް ފަހުގައި، فرعون އާއި، އޭނާގެ ބޮޑުންގެ ގާތަށް، ތިމަންރަސްކަލާނގެ آية ތަކާއިގެން موسى ގެފާނާއި، هارون ގެފާނު ފޮނުއްވީމެވެ. ދެން އެއުރެން كبر ވެރިވެގަތޫއެވެ. އަދި ކުށްވެރި ބަޔަކު ކަމުގައި އެއުރެންވިއެވެ.
فَلَمَّا جَاءَهُمُ الْحَقُّ مِنْ عِندِنَا قَالُوا إِنَّ هَٰذَا لَسِحْرٌ مُّبِينٌ ( 76 )
ފަހެ، ތިމަންރަސްކަލާނގެ حضرة ން އެއުރެންގެ ގާތަށް حق އައުުމުން، އެއުރެން ބުނެފޫއެވެ. ހަމަކަށަވަރުން، މިއީ ބަޔާންވެގެންވާ ޖާދޫއެއްކަން ކަށަވަރެވެ.
قَالَ مُوسَىٰ أَتَقُولُونَ لِلْحَقِّ لَمَّا جَاءَكُمْ ۖ أَسِحْرٌ هَٰذَا وَلَا يُفْلِحُ السَّاحِرُونَ ( 77 )
موسى ގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންގެ ގާތަށް حق އައިހިނދު، އެ حق އަށް މިއީ ޖާދޫއެއްހޭ ތިޔަބައިމީހުން ބުނަނީހެއްޔެވެ؟ އަދި ޖާދޫހަދާ މީހުން ނުދިނޭހުށްޓެވެ.
قَالُوا أَجِئْتَنَا لِتَلْفِتَنَا عَمَّا وَجَدْنَا عَلَيْهِ آبَاءَنَا وَتَكُونَ لَكُمَا الْكِبْرِيَاءُ فِي الْأَرْضِ وَمَا نَحْنُ لَكُمَا بِمُؤْمِنِينَ ( 78 )
އެއުރެން ދެންނެވޫއެވެ. ކަލޭގެފާނު ތިޔަވަޑައިގަތީ، ތިމަންމެންގެ ކާބަފައިން އުޅޭތީ ތިމަންމެންނަށް ފެނުނުގޮތުން ތިމަންމެން އެއްކިބާކުރައްވައި ތިޔަ ދެބޭކަލުންނަށް ބިމުގައި ވެރިކަން ހޯއްދެވުމަށްތޯއެވެ؟ ތިޔަ ދެބޭކަލުންނަށް إيمان ވާ ބަޔަކު ކަމުގައި ތިމަންމެން ނުވަމުއެވެ.
وَقَالَ فِرْعَوْنُ ائْتُونِي بِكُلِّ سَاحِرٍ عَلِيمٍ ( 79 )
فرعون ބުންޏެވެ. މޮޅަށް ދަންނަ ހުރިހާ ޖާދޫގަރުން ތިމަންގެ ގާތަށް ގެނެސްދޭށެވެ!
فَلَمَّا جَاءَ السَّحَرَةُ قَالَ لَهُم مُّوسَىٰ أَلْقُوا مَا أَنتُم مُّلْقُونَ ( 80 )
ފަހެ، އެ ޖާދޫގަރުން އައިހިނދު އެއުރެންނަށް موسى ގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުން (ޖާދޫގެގޮތުން) އުކަންހުރި ތަކެތި އުކާށެވެ!
فَلَمَّا أَلْقَوْا قَالَ مُوسَىٰ مَا جِئْتُم بِهِ السِّحْرُ ۖ إِنَّ اللَّهَ سَيُبْطِلُهُ ۖ إِنَّ اللَّهَ لَا يُصْلِحُ عَمَلَ الْمُفْسِدِينَ ( 81 )
ފަހެ، އެއުރެން (އެތަކެތި) އުކިހިނދު، موسى ގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުން ތިޔަ ގެނައީ ޖާދޫއެވެ. ހަމަކަށަވަރުން، اللَّه، އެ ޖާދޫ باطل ކުރައްވާނެތެވެ. ހަމަކަށަވަރުން، فساد ކުރާ މީހުންގެ عمل، اللَّه إصلاح އެއް ނުކުރައްވާހުށްޓެވެ.
وَيُحِقُّ اللَّهُ الْحَقَّ بِكَلِمَاتِهِ وَلَوْ كَرِهَ الْمُجْرِمُونَ ( 82 )
އަދި ކުށްވެރިން ނުރުހުނަސް، اللَّه އެކަލާނގެ كلمة ފުޅުތަކުން حق ސާބިތުކުރައްވައި ފާޅުކުރައްވާނެތެވެ.
فَمَا آمَنَ لِمُوسَىٰ إِلَّا ذُرِّيَّةٌ مِّن قَوْمِهِ عَلَىٰ خَوْفٍ مِّن فِرْعَوْنَ وَمَلَئِهِمْ أَن يَفْتِنَهُمْ ۚ وَإِنَّ فِرْعَوْنَ لَعَالٍ فِي الْأَرْضِ وَإِنَّهُ لَمِنَ الْمُسْرِفِينَ ( 83 )
ފަހެ، موسى ގެފާނަށް އެކަލޭގެފާނުގެ قوم ގެ ބައެއް ދަރިކޮޅު މެނުވީ إيمان އެއް ނުވެތެވެ. (އެއީ) فرعون އާއި، އޭނާގެ ބޮޑުންނަށް ބިރުންނެވެ. އެއުރެންނަށް އަނިޔާކޮށް، دين ން އެއްކިބާކޮށްފާނެ ކަމަށްޓަކައެވެ. އަދި ހަމަކަށަވަރުން، فرعون އަކީ، ބިމުގައި ވަރަށް ބޮޑާވެގެން އުޅުނު މީހެކެވެ. އަދި ހަމަކަށަވަރުން، އޭނާވަނީ (އަނިޔާވެރިކަމާއި فساد ވެރިކަމުގައި) ހައްދުފަހަނަޅައިފައިވާ މީހުންގެ ތެރެއިންނެވެ.
وَقَالَ مُوسَىٰ يَا قَوْمِ إِن كُنتُمْ آمَنتُم بِاللَّهِ فَعَلَيْهِ تَوَكَّلُوا إِن كُنتُم مُّسْلِمِينَ ( 84 )
އަދި موسى ގެފާނު ވިދާޅުވިއެވެ. އޭ އަހުރެންގެ قوم ގެމީހުންނެވެ! ތިޔަބައިމީހުން اللَّه އަށް إيمان ވެގެންވާނަމަ، ފަހެ، އެކަލާނގެއަށް وكيل ކުރާށެވެ! ތިޔަބައިމީހުންނީ مسلم ން ކަމުގައި ވާނަމައެވެ.
فَقَالُوا عَلَى اللَّهِ تَوَكَّلْنَا رَبَّنَا لَا تَجْعَلْنَا فِتْنَةً لِّلْقَوْمِ الظَّالِمِينَ ( 85 )
ފަހެ، އެއުރެން ދެންނެވޫއެވެ. اللَّه އަށް އަޅަމެން وكيل ކޮށްފީމެވެ. އަޅަމެންގެ ވެރިރަސްކަލާކޯއެވެ! އަނިޔާވެރިވީ މީހުންނަށް ފިތުނައެއްކަމުގައި (އެބަހީ: އެއުރެންގެ އަނިޔާވެރިކަމުގެ ދަށުގައި) އަޅަމެން ނުލައްވާނދޭވެ!
وَنَجِّنَا بِرَحْمَتِكَ مِنَ الْقَوْمِ الْكَافِرِينَ ( 86 )
އަދި كافر ވި މީހުންގެ ކިބައިން، އިބަރަސްކަލާނގެ رحمة ފުޅާލައިގެން އަޅަމެން ސަލާމަތް ކުރައްވާނދޭވެ!
وَأَوْحَيْنَا إِلَىٰ مُوسَىٰ وَأَخِيهِ أَن تَبَوَّآ لِقَوْمِكُمَا بِمِصْرَ بُيُوتًا وَاجْعَلُوا بُيُوتَكُمْ قِبْلَةً وَأَقِيمُوا الصَّلَاةَ ۗ وَبَشِّرِ الْمُؤْمِنِينَ ( 87 )
ތިމަންރަސްކަލާނގެ موسى ގެފާނާއި، އެކަލޭގެފާނުގެ أخ (هارون ގެފާނު) އަށް وحى ކުރެއްވީމެވެ. ތިޔަ ދެބޭކަލުންގެ قوم ގެމީހުންނަށް مصر ގައި ގޭގެ ހައްދަވާށެވެ! އަދި ތިޔަބައިމީހުންގެ ގޭގެ، قبلة އަށް ބަހައްޓައި، ނަމާދު قائم ކުރާށެވެ! އަދި ކަލޭގެފާނު مؤمن ތަކުންނަށް (نصر ލިބޭނެކަމުގެ) އުފާވެރި خبر ދެއްވާށެވެ!
وَقَالَ مُوسَىٰ رَبَّنَا إِنَّكَ آتَيْتَ فِرْعَوْنَ وَمَلَأَهُ زِينَةً وَأَمْوَالًا فِي الْحَيَاةِ الدُّنْيَا رَبَّنَا لِيُضِلُّوا عَن سَبِيلِكَ ۖ رَبَّنَا اطْمِسْ عَلَىٰ أَمْوَالِهِمْ وَاشْدُدْ عَلَىٰ قُلُوبِهِمْ فَلَا يُؤْمِنُوا حَتَّىٰ يَرَوُا الْعَذَابَ الْأَلِيمَ ( 88 )
موسى ގެފާނު ދެންނެވިއެވެ. އަޅަމެންގެ ވެރިރަސްކަލާކޯއެވެ! ހަމަކަށަވަރުން، އިބަރަސްކަލާނގެ فرعون އަށާއި، އޭނާގެ ބޮޑުންނަށް ދުނިޔޭގެ ދިރިއުޅުމުގައި ޒީނަތްތެރިކަމާއި، މުދާތައް ދެއްވީމުއެވެ. އަޅަމެންގެ ވެރިރަސްކަލާކޯއެވެ! (އެތަކެތިން) އެއުރެން އިބަރަސްކަލާނގެ ތެދުމަގުން (މީސްތަކުން) އެއްކިބާކުރެތެވެ. އަޅަމެންގެ ވެރިރަސްކަލާކޯއެވެ! އެއުރެންގެ މުދާތައް ހަލާކު ކުރައްވައި، ނެތިކުރައްވާނދޭވެ! އަދި އެއުރެންގެ ހިތްތައް ހަރުކަށިކުރައްވައި، ސިއްކަގައްސަވާނދޭވެ! ފަހެ، އެއުރެންނަށް ވޭންދެނިވި عذاب ފެނިއްޖައުމަށް ދާނދެން، އެއުރެން إيمان އެއް ނުވާނެތެވެ.
قَالَ قَدْ أُجِيبَت دَّعْوَتُكُمَا فَاسْتَقِيمَا وَلَا تَتَّبِعَانِّ سَبِيلَ الَّذِينَ لَا يَعْلَمُونَ ( 89 )
އެކަލާނގެ وحى ކުރެއްވިއެވެ. ހަމަކަށަވަރުން، ތިޔަ ދެބޭކަލުންގެ دعاء އިޖާބަކުރައްވައިފީމެވެ. ފަހެ، ތިޔަ ދެބޭކަލުން ސީދާގޮތުގައި ތިއްބަވާށެވެ! ނުދަންނަ މީހުންގެ މަގަށް ތިޔަ ދެބޭކަލުން تبع ވެވަޑައި ނުގަންނަވާހުށިކަމެވެ!
وَجَاوَزْنَا بِبَنِي إِسْرَائِيلَ الْبَحْرَ فَأَتْبَعَهُمْ فِرْعَوْنُ وَجُنُودُهُ بَغْيًا وَعَدْوًا ۖ حَتَّىٰ إِذَا أَدْرَكَهُ الْغَرَقُ قَالَ آمَنتُ أَنَّهُ لَا إِلَٰهَ إِلَّا الَّذِي آمَنَتْ بِهِ بَنُو إِسْرَائِيلَ وَأَنَا مِنَ الْمُسْلِمِينَ ( 90 )
ކަނޑުހުރަސް ކުރައްވައި، ތިމަންރަސްކަލާނގެ إسـرآءيـل ގެދަރިން ގެންދެވީމެވެ. ފަހެ، فرعون އާއި، އޭނާގެ لشكر ދެކޮޅުވެރިކަމާއި، عداوة ތެރިކަމުގެ ގޮތުގައި އެއުރެންގެ ފަހަތުން ދިޔައޫއެވެ. ދެން އެ فرعون، غرق ވީހިނދު އޭނާ ބުންޏެވެ. إسـرآءيـل ގެދަރިން إيمان ވީ ކަލާނގެ މެނުވީ، އެހެންކަލަކު ނެތްކަމަށް އަހުރެން إيمان ވެއްޖައީމެވެ. އަދި އަހުރެން ވަނީ مسلم ންގެ ތެރެއިންނެވެ.
آلْآنَ وَقَدْ عَصَيْتَ قَبْلُ وَكُنتَ مِنَ الْمُفْسِدِينَ ( 91 )
(އެވަގުތު فرعون އަށް އެންގެވިއެވެ. ކަލޭ ތިޔަ إيمان ވީ) މިހާރު ހެއްޔެވެ؟ ކަލޭ މީގެކުރިންވަނީ އުރެދިފައެވެ. އަދި ކަލޭވީ، ފަސާދަކުރާ މީހުންގެ ތެރެއިންނެވެ.
فَالْيَوْمَ نُنَجِّيكَ بِبَدَنِكَ لِتَكُونَ لِمَنْ خَلْفَكَ آيَةً ۚ وَإِنَّ كَثِيرًا مِّنَ النَّاسِ عَنْ آيَاتِنَا لَغَافِلُونَ ( 92 )
ފަހެ، ތިމަންރަސްކަލާނގެ، މިއަދު ކަލޭގެ (ފުރާނަނެތް) ހަށިގަނޑު ސަލާމަތްކުރައްވާހުށީމެވެ. އެއީ، ކަލޭގެފަހުން އަންނަބަޔަކަށް، عبرة އެއް ކަމުގައި ކަލޭ ވުމަށެވެ. ހަމަކަށަވަރުން، މީސްތަކުންގެ ތެރެއިން ގިނަމީހުންވަނީ ތިމަންރަސްކަލާނގެ آية ތަކާމެދު (އެބަހީ: ތިމަންރަސްކަލާނގެ قدرة ތެރިކަމުގެ މައްޗަށް دليل ކުރާ ހެކިތަކާމެދު) غافل ވެގެންކަން ކަށަވަރެވެ.
وَلَقَدْ بَوَّأْنَا بَنِي إِسْرَائِيلَ مُبَوَّأَ صِدْقٍ وَرَزَقْنَاهُم مِّنَ الطَّيِّبَاتِ فَمَا اخْتَلَفُوا حَتَّىٰ جَاءَهُمُ الْعِلْمُ ۚ إِنَّ رَبَّكَ يَقْضِي بَيْنَهُمْ يَوْمَ الْقِيَامَةِ فِيمَا كَانُوا فِيهِ يَخْتَلِفُونَ ( 93 )
ހަމަކަށަވަރުން، إسـرآءيـل ގެދަރިންނަށް ރަނގަޅު މޮޅު منـزل އެއް (އެބަހީ: مصر އާއި شام) ތިމަންރަސްކަލާނގެ ދެއްކެވީމެވެ. އަދި ރަނގަޅު ތަކެތިން، ތިމަންރަސްކަލާނގެ އެއުރެންނަށް رزق ދެއްވީމެވެ. ދެން އެއުރެންނާ ހަމަޔަށް علم އަތުވެއްޖެއުމަށް ދާނދެން، އެއުރެން خلاف އެއް ނުވެތެވެ. ހަމަކަށަވަރުން، އެއުރެން خلاف ވެ އުޅުނު ކަންތަކާމެދު، ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ قيامة ދުވަހުން އެއުރެންގެ މެދުގައި ނިޔާކުރައްވާނެތެވެ.
فَإِن كُنتَ فِي شَكٍّ مِّمَّا أَنزَلْنَا إِلَيْكَ فَاسْأَلِ الَّذِينَ يَقْرَءُونَ الْكِتَابَ مِن قَبْلِكَ ۚ لَقَدْ جَاءَكَ الْحَقُّ مِن رَّبِّكَ فَلَا تَكُونَنَّ مِنَ الْمُمْتَرِينَ ( 94 )
ފަހެ، ތިމަންރަސްކަލާނގެ ކަލޭގެފާނަށް ބާވައިލެއްވި ތަކެއްޗާމެދު، ކަލޭގެފާނު شك އެއްގައިވާނަމަ، ކަލޭގެފާނުގެ ކުރިން ފޮތްކިޔައިއުޅުނު މީހުންނާ سؤال ކުރައްވާށެވެ! ހަމަކަށަވަރުން، ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ حضرة ން ކަލޭގެފާނަށް حق ތެދު އަތުވެއްޖެއެވެ. ފަހެ، شك ކުރާ މީހުންގެ ތެރެއިން، ކަލޭގެފާނު ވެވަޑައި ނުގަންނަވާ ހުށިކަމެވެ!
وَلَا تَكُونَنَّ مِنَ الَّذِينَ كَذَّبُوا بِآيَاتِ اللَّهِ فَتَكُونَ مِنَ الْخَاسِرِينَ ( 95 )
އަދި اللَّه ގެ آية ތައް ދޮގުކުރާ މީހުންގެ ތެރެއިން ކަލޭގެފާނު ވެވަޑައި ނުގަންނަވާހުށިކަމެވެ! ފަހެ، އޭރުން ކަލޭގެފާނު ވާހުށީ، ގެއްލި ހަލާކުވެގެންވާ މީހުންގެ ތެރެއިންނެވެ.
إِنَّ الَّذِينَ حَقَّتْ عَلَيْهِمْ كَلِمَتُ رَبِّكَ لَا يُؤْمِنُونَ ( 96 )
އަދި اللَّه ގެ آية ތައް ދޮގުކުރާ މީހުންގެ ތެރެއިން ކަލޭގެފާނު ވެވަޑައި ނުގަންނަވާހުށިކަމެވެ! ފަހެ، އޭރުން ކަލޭގެފާނު ވާހުށީ، ގެއްލި ހަލާކުވެގެންވާ މީހުންގެ ތެރެއިންނެވެ.
وَلَوْ جَاءَتْهُمْ كُلُّ آيَةٍ حَتَّىٰ يَرَوُا الْعَذَابَ الْأَلِيمَ ( 97 )
ހަމަކަށަވަރުން، އެއުރެންގެ މައްޗަށް (عذاب ހުރިކަމަށް) ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ ބަސްފުޅު حق ވެގެންވާ މީހުން، އެއުރެންގެ ގާތަށް ކޮންމެ آية އެއް އަޔަސް، އެއުރެންނަށް ވޭންދެނިވި عذاب ފެނިއްޖައުމަށް ދާނދެން، إيمان އެއް ނުވާނެތެވެ.
فَلَوْلَا كَانَتْ قَرْيَةٌ آمَنَتْ فَنَفَعَهَا إِيمَانُهَا إِلَّا قَوْمَ يُونُسَ لَمَّا آمَنُوا كَشَفْنَا عَنْهُمْ عَذَابَ الْخِزْيِ فِي الْحَيَاةِ الدُّنْيَا وَمَتَّعْنَاهُمْ إِلَىٰ حِينٍ ( 98 )
ފަހެ، ރަށެއްގެ މީހުން، އެއުރެންގެ إيمان ކަމުން މަންފާކުރާނޭ ގޮތުގައި، إيمان ވީނަމަ ކިހާރަނގަޅުހޭ! އެހެނެއްކަމަކު، يونس ގެފާނުގެ قوم ގެމީހުން إيمان ވުމުން، ތިމަންރަސްކަލާނގެ އެއުރެންގެ ކިބައިން ދުނިޔޭގެ حياة ގައި ނިކަމެތިކަމުގެ عذاب ދުރުކުރެއްވީމެވެ. އަދި އެއުރެންގެ އަޖަލު ޖެހިއްޖައުމަށް ދާނދެން، އެއުރެން އަރާމުގައި އުޅުމަށް ލެހެއްޓެވީމެވެ.
وَلَوْ شَاءَ رَبُّكَ لَآمَنَ مَن فِي الْأَرْضِ كُلُّهُمْ جَمِيعًا ۚ أَفَأَنتَ تُكْرِهُ النَّاسَ حَتَّىٰ يَكُونُوا مُؤْمِنِينَ ( 99 )
ކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ އިރާދަކުރައްވައިފިނަމަ، ބިމުގައި ވާހާ އެންމެން إيمان ވެއްޖައީހެވެ. ފަހެ، ކަލޭގެފާނު މީސްތަކުންނަށް އެއުރެން مؤمن ން ކަމުގައިވުމަށް مجبور ކުރައްވަނީތޯއެވެ؟
وَمَا كَانَ لِنَفْسٍ أَن تُؤْمِنَ إِلَّا بِإِذْنِ اللَّهِ ۚ وَيَجْعَلُ الرِّجْسَ عَلَى الَّذِينَ لَا يَعْقِلُونَ ( 100 )
އަދި اللَّه ގެ إذن ފުޅަށް މެނުވީ، إيمان ވުމެއް އެއްވެސް نفس އަކަށް ނެތެވެ. އަދި ނުވިސްނާ މީހުންނަށް، އެކަލާނގެ عذاب ދެއްވާހުށްޓެވެ.
قُلِ انظُرُوا مَاذَا فِي السَّمَاوَاتِ وَالْأَرْضِ ۚ وَمَا تُغْنِي الْآيَاتُ وَالنُّذُرُ عَن قَوْمٍ لَّا يُؤْمِنُونَ ( 101 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! އުޑުތަކާއި، ބިމުގައިވަނީ ކޮންއެއްޗެއްތޯ ތިޔަބައިމީހުން ބަލާށެވެ! އަދި إيمان ނުވާ ބަޔަކަށް آية ތަކުންނާއި، إنذار ކުރައްވާ رسول ންގެ ކިބަޔަކުން ކަމެއް ފުއްދައެއް ނުދެވޭނެތެވެ.
فَهَلْ يَنتَظِرُونَ إِلَّا مِثْلَ أَيَّامِ الَّذِينَ خَلَوْا مِن قَبْلِهِمْ ۚ قُلْ فَانتَظِرُوا إِنِّي مَعَكُم مِّنَ الْمُنتَظِرِينَ ( 102 )
ފަހެ، އެއުރެންގެ ކުރިން އިސްވެދިޔަ أمّة ތަކަށް ޖެހުނުފަދަ ކަންތައްތަކެއް ޖެހުމަށްމެނުވީ، އެއުރެން انتظار ކުރޭހެއްޔެވެ؟ ކަލޭގެފާނު ވިދާޅުވާށެވެ! ފަހެ، ތިޔަބައިމީހުން انتظار ކުރާށެވެ! ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނުވެސް، ތިޔަބައިމީހުންނާއެކު انتظار ކުރާ މީހުންގެ ތެރެއިންވަމެވެ.
ثُمَّ نُنَجِّي رُسُلَنَا وَالَّذِينَ آمَنُوا ۚ كَذَٰلِكَ حَقًّا عَلَيْنَا نُنجِ الْمُؤْمِنِينَ ( 103 )
ދެން ތިމަންރަސްކަލާނގެ رسول ންނާއި، إيمان ވި މީހުން ސަލާމަތް ކުރައްވަމެވެ. ހަމައެފަދައިން، مؤمن ން ދިންނަވައި ސަލާމަތް ކުރެއްވުމަކީ، ތިމަންރަސްކަލާނގެއަށް އޮތް حق އެކެވެ.
قُلْ يَا أَيُّهَا النَّاسُ إِن كُنتُمْ فِي شَكٍّ مِّن دِينِي فَلَا أَعْبُدُ الَّذِينَ تَعْبُدُونَ مِن دُونِ اللَّهِ وَلَٰكِنْ أَعْبُدُ اللَّهَ الَّذِي يَتَوَفَّاكُمْ ۖ وَأُمِرْتُ أَنْ أَكُونَ مِنَ الْمُؤْمِنِينَ ( 104 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! އޭ މީސްތަކުންނޭވެ! ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނުގެ دين އާމެދު شك އެއްގައި ވާނަމަވެސް، اللَّه ފިޔަވައި ތިޔަބައިމީހުން އަޅުކަންކުރާ ތަކެއްޗަކަށް، ތިމަންކަލޭގެފާނު އަޅުކަމެއް ނުކުރައްވާހުށީމެވެ. އެހެނެއްކަމަކު، ތިމަންކަލޭގެފާނު އަޅުކަންކުރައްވާހުށީ، اللَّه އަށެވެ. އެކަލާނގެއީ، ތިޔަބައިމީހުން މަރުގަންނަވާ ރަސްކަލާނގެއެވެ. އަދި ތިމަންކަލޭގެފާނަށް أمر ކުރައްވާފައިވަނީ، مؤمن ންގެ ތެރެއިން ވުމަށެވެ.
وَأَنْ أَقِمْ وَجْهَكَ لِلدِّينِ حَنِيفًا وَلَا تَكُونَنَّ مِنَ الْمُشْرِكِينَ ( 105 )
އަދި باطل މަގުން އެއްކިބާވެ، حق ގެ މަގަށް تبع ވެގެންވާ حال، ކަލޭގެފާނުގެ މޫނުފުޅު إسلام دين އާ ކުރިމަތިކުރައްވާށެވެ! އަދި ކަލޭގެފާނު مشرك ންގެ ތެރެއިން ވެވަޑައި ނުގަންނަވާހުށިކަމެވެ!
وَلَا تَدْعُ مِن دُونِ اللَّهِ مَا لَا يَنفَعُكَ وَلَا يَضُرُّكَ ۖ فَإِن فَعَلْتَ فَإِنَّكَ إِذًا مِّنَ الظَّالِمِينَ ( 106 )
އަދި اللَّه ފިޔަވައި ކަލޭގެފާނަށް މަންފާއެއް ނުދެވޭ، އަދި ކަލޭގެފާނަށް ގެއްލުމެއްވެސް ނުދެވޭ ތަކެއްޗަށް، ކަލޭގެފާނު އަޅުކަން ނުކުރައްވާށެވެ! ފަހެ، ކަލޭގެފާނު އެކަންކުރައްވައިފިނަމަ، ހަމަކަށަވަރުން، އޭރުން ކަލޭގެފާނު ވާނީ އަނިޔާވެރިވި މީހުންގެ ތެރެއިންނެވެ.
وَإِن يَمْسَسْكَ اللَّهُ بِضُرٍّ فَلَا كَاشِفَ لَهُ إِلَّا هُوَ ۖ وَإِن يُرِدْكَ بِخَيْرٍ فَلَا رَادَّ لِفَضْلِهِ ۚ يُصِيبُ بِهِ مَن يَشَاءُ مِنْ عِبَادِهِ ۚ وَهُوَ الْغَفُورُ الرَّحِيمُ ( 107 )
އަދި اللَّه ތިބާއަށް ގެއްލުމެއް މެދުވެރިކުރައްވާނަމަ، އެކަލާނގެ މެނުވީ، އެކަން ފިއްލަވާނޭ އެހެންފަރާތެއް ނުވެއެވެ. އަދި އެކަލާނގެ ކަލޭގެފާނަށް ހެޔޮކަމަކަށް އިރާދަކުރައްވައިފިނަމަ، އެކަލާނގެ فضل ވަންތަކަން އަނބުރާ ރައްދުކުރެވޭނެ ފަރާތެއް ނުވެތެވެ. އެ فضل ވަންތަކަން އެކަލާނގެ ދެއްވަނީ، އެކަލާނގެ އަޅުންގެ ތެރެއިން، އެކަލާނގެ އިރާދަކުރެއްވި މީހަކަށެވެ. އަދި އެކަލާނގެއީ، ގިނަގިނައިން ފާފަފުއްސަވާ رحيم ވަންތަ ރަސްކަލާނގެއެވެ.
قُلْ يَا أَيُّهَا النَّاسُ قَدْ جَاءَكُمُ الْحَقُّ مِن رَّبِّكُمْ ۖ فَمَنِ اهْتَدَىٰ فَإِنَّمَا يَهْتَدِي لِنَفْسِهِ ۖ وَمَن ضَلَّ فَإِنَّمَا يَضِلُّ عَلَيْهَا ۖ وَمَا أَنَا عَلَيْكُم بِوَكِيلٍ ( 108 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! އޭ މީސްތަކުންނޭވެ! ތިޔަބައިމީހުންގެ ވެރިރަސްކަލާނގެ حضرة ން، ތިޔަބައިމީހުންގެ ގާތަށް حق އަތުވެއްޖެކަން ކަށަވަރެވެ. ފަހެ، ތެދުމަގު ލިބިގެންފި މީހަކު ތެދުމަގު ލިބިގަންނަކަން ކަށަވަރީ، އެމީހެއްގެ އަމިއްލަ نفس އަށެވެ. އަދި މަގުފުރެދިއްޖެ މީހަކު މަގުފުރެދޭކަން ކަށަވަރީ، އެމީހެއްގެ އަމިއްލަ نفس ގެ މައްޗަށެވެ. އަދި ތިޔަބައިމީހުންގެ ކަންތައްތައް حوال ކުރައްވާފައިވާ وكيل އަކު ކަމުގައި، ތިމަންކަލޭގެފާނު ނުވަމެވެ.
وَاتَّبِعْ مَا يُوحَىٰ إِلَيْكَ وَاصْبِرْ حَتَّىٰ يَحْكُمَ اللَّهُ ۚ وَهُوَ خَيْرُ الْحَاكِمِينَ ( 109 )
އަދި ކަލޭގެފާނަށް وحى ކުރައްވާ އެއްޗަކަށް، ކަލޭގެފާނު تبع ވެވަޑައިގަންނަވާށެވެ! އަދި (އެއުރެންނާމެދު) اللَّه حكم ކުރައްވައިފުމަށް ދާނދެން (دين އަށް ގޮވައިލެއްވުމުގައި) ކެތްކުރައްވައިގެން ހުންނަވާށެވެ! އެކަލާނގެއީ، حكم ކުރައްވާ އެންމެ ހެޔޮ ރަސްކަލާނގެއެވެ.
