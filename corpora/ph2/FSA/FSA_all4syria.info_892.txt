أخبار دولية دويتشه قيليه تتحرى السلطات النمسا والسويد والنرويج وهولندا أمر عملاء محليين لمؤسسة قانونية تتخذ بنما مقرا وردت أسماؤهم تسريبات ضخمة للبيانات وذلك للتحقق حالات تهرب ضريبي محتملة يحقق جهاز مراقبة الأسواق المالية النمسا فيما إذا بنك رايفايسين الدولي وهايبو لاندسبنك فورارلبيرغ التزما بقواعد مكافحة غسل الأموال أن ورد اسماهما تسريبات وثائق واسعة النطاق متحدث الجهاز الاثنين الثالث أبريل نيسان إنه يتحرى الالتزام بإجراءات مثل فحص طبيعة التعاملات والمتورطين وأضاف أن الجهاز يحيل الحالات إلى السلطات الجنائية إذا لزم الأمر السويد اتصلت هيئة الرقابة المالية بالسلطات لوكسمبورغ لطلب معلومات ذات صلة بمزاعم أفادت بأن مجموعة نورديا البنكية ساعدت العملاء فتح حسابات ملاذات ضريبية الخارج ووردت المعلومات بشأن نورديا تسريب أكثر وثيقة ملفات مؤسسة موساك فونسيكا القانونية ومقرها بنما كريستر فوروشدت رئيس قسم مراقبة البنوك السويدية الكبرى هيئة الرقابة المالية نأخذ الأمر محمل الجد بشدة وغرمت نورديا كرونة دولار أيار عيوب تعاملها غسل الأموال قالت هيئة الرقابة المالية النرويج الاثنين إن الكشف مساعدة بنك دي بي النرويجي لعملاء تأسيس شركات معاملات خارجية جزر سيشيل يقوض الثقة القطاع المالي صحيفة أفتنبوستن نشرت تقارير أنشطة البنك سيشيل نقلاً الوثائق المسربة موساك فونسيكا مورتن بالتسرسن رئيس الهيئة بيان الترتيب لعملاء النحو ظهر يسهم إضعاف الثقة البنوك المعنية القطاع المالي وأضاف أن الهيئة ستطلب تفسيرا مجالس إدارات البنوك المعنية أمستردام قالت السلطات الهولندية إنها ستحقق المزاعم المتعلقة بهولندا الوثائق المسربة وزارة المالية الهولندية بيان الاثنين إن هيئة الضرائب ستحقق بهمة فيما إذا البيانات صلة بعمليات تهرب ضريبي متحدثة كاميرون الاستثمارات العائلية مسألة خاصة جانبها رفضت المتحدثة رئيس الوزراء البريطاني ديفيد كاميرون التعليق الاثنين إذا أسرة رئيس الوزراء لديها أموال مستثمرة صناديق للمعاملات الخارجية أوفشور طريق والده قائلة إن مسألة خاصة وثائق بنما كشف المستور شخصيات سياسية عربية ذكرت الوثائق أن للملك السعودي سلمان عبد العزيز دورا محدد شركتين مسجلتين جزر العذراء البريطانية حصلت الشركتان قرضين بقيمة مليوم دولار لشراء شقق فاخرة لندن ويخت وبينما تحدد الوثائق دور المللك السعودي يدير شبكة صناديق الاستثمار المعاملات الخارجية لتعزيز ثروة العائلة وردت المتحدثة كاميرون سؤال عما إذا بإمكانها تأكيد أن الأسرة تعد تستثمر أموالها الصناديق قائلة مسألة خاصة تركيزي ينصب تقوم الحكومة
