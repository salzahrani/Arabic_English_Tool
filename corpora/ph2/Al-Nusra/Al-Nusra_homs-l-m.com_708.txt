قُلُوبَنَا بَعْدَ إِذْ هَدَيْتَنَا وَهَبْ لَنَا مِن لَّدُنكَ رَحْمَةً ۚ إِنَّكَ أَنتَ الْوَهَّابُ اسان جا پالڻھار جڏھن تو اسان کي ھدايت ڪئي تنھن کان پوءِ اسان جي دلين کي ڏنگائي نه ڏي ۽ پاڻ وٽان اسان اللَّهُ بِذُنُوبِهِمْ ۗ وَاللَّهُ شَدِيدُ الْعِقَابِ اُنھن جو حال فرعونين جي ۽ اُنھن جي حال جھڙو آھي جي کانئن اڳ ھؤا اُنھن اسان جي آيتن کي ڪُوڙو ڀانيو پوءِ الله سندن گناھن سببان کين پڪڙيو ۽ الله سخت وَاللَّهُ عَلَىٰ كُلِّ شَيْءٍ قَدِيرٌ پيغمبر کين چؤ ته جيڪي اوھان جي دلين ۾ آھي سو جي لڪائيندؤ يا اُھو پڌرو ڪندؤ ته اُن کي الله ڄاڻندو آھي ۽ جيڪي آسمانن ۾ آھي ۽ جيڪي زمين ۾ آھي سوبه ڄاڻندو آھي ۽ الله چؤ ته جي اوھين الله کي دوست رکندا آھيو ته منھنجي تابعداري ڪريو ته الله اوھان کي دوست رکندو ۽ اوھان جا گناھ اوھان کي بخشيندو ۽ الله بخشڻھار مھربان آھي قُلْ أَطِيعُوا اللَّهَ وَالرَّسُولَ ۖ فَإِن الْمَوْتَىٰ بِإِذْنِ اللَّهِ ۖ وَأُنَبِّئُكُم بِمَا تَأْكُلُونَ وَمَا تَدَّخِرُونَ فِي بُيُوتِكُمْ ۚ إِنَّ فِي ذَٰلِكَ لَآيَةً لَّكُمْ إِن كُنتُم مُّؤْمِنِينَ ۽ اُن کي بني اسرائيلن ڏانھن پيغمبر ڪري موڪليندو عيسىٰ وڏو ٿيو ۽ چيائين ته مون اوھان جي پالڻھار وٽان نشاني آندي آھي ته آءٌ اوھان لاءِ مٽيءَ مان پکيءَ جي شڪل جھڙو بوتو بڻائيندس پوءِ منجھس ڦوڪيندس ته الله جي حُڪم سان پکي ٿيندو ۽ ماءُ ڄاول اَنڌي ۽ ڪُوڙھي کي ڇُٽائيندس ۽ الله جي حُڪم سان مُئي کي جياريندس ۽ جيڪي کائيندا آھيو ۽ جيڪي پنھنجن گھرن ۾ سانڀيندا آھيو سو اوھان کي ڏسيندس جيڪڏھن اوھين مڃيندڙ آھيو ته اوھان لاءِ اُن ۾ نشانيون آھن وَمُصَدِّقًا لِّمَا بَيْنَ يَدَيَّ مِنَ التَّوْرَاةِ وَلِأُحِلَّ لَكُم بَعْضَ الَّذِي حُرِّمَ عَلَيْكُمْ ۚ وَجِئْتُكُم بِآيَةٍ مِّن رَّبِّكُمْ فَاتَّقُوا اللَّهَ وَأَطِيعُونِ ۽ جيڪو مون کان اڳ توريت جو ڪتاب آيل آھي تنھن جو سچو ڪندڙ آھيان ۽ جيڪي شيون اوھان تي حرام ڪيل آھن تن مان ڪي اوھان لاءِ حلال ڪندس ۽ اوھان جي پالڻھار وٽان اوھان لاءِ نشانيون آنديون اٿم پوءِ الله کان ڊڄو ۽ مُنھنجي تابعداري ڪريو إِنَّ اللَّهَ رَبِّي وَرَبُّكُمْ فَاعْبُدُوهُ ۗ هَٰذَا صِرَاطٌ مُّسْتَقِيمٌ بيشڪ الله مُنھنجو پالڻھار ۽ اوھان جو پالڻھار آھي تنھنڪري سندس عبادت ڪريو اھا سنئين واٽ آھي فَلَمَّا أَحَسَّ عِيسَىٰ مِنْهُمُ الْكُفْرَ قَالَ مَنْ أَنصَارِي إِلَى اللَّهِ ۖ قَالَ الْحَوَارِيُّونَ نَحْنُ أَنصَارُ اللَّهِ آمَنَّا بِاللَّهِ وَاشْهَدْ بِأَنَّا مُسْلِمُونَ پوءِ جنھن مھل عيسىٰ کائن ڪُفر جا ڪم اکين سان ڏٺا تنھن مھل چيائين ته الله جي دين ڏانھن مُنھنجا واھرو ڪير آھن حوارين چيو ته اسين الله جي دين جا واھرو آھيون الله کي مڃيوسون ۽ شاھد ھُج ته اسين پَڪ مُسلمان آھيون رَبَّنَا آمَنَّا بِمَا أَنزَلْتَ وَاتَّبَعْنَا الرَّسُولَ فَاكْتُبْنَا مَعَ الشَّاهِدِينَ چيائون اسانجا پالڻھار جيڪي تو لاٿو سو مڃيوسون ۽ پيغمبر جي تابعداري ڪئي سون تنھنڪري اسانکي شاھدن ۾ لک وَمَكَرُوا وَمَكَرَ اللَّهُ ۖ وَاللَّهُ خَيْرُ الْمَاكِرِينَ ۽ يھودين رٿ رٿي ۽ الله ڳجھي رٿ رٿي ۽ الله چڱو ڳجھي رٿ رٿيندڙ آھي إِذْ قَالَ اللَّهُ يَا عِيسَىٰ إِنِّي مُتَوَفِّيكَ وَرَافِعُكَ إِلَيَّ وَمُطَهِّرُكَ مِنَ الَّذِينَ كَفَرُوا وَجَاعِلُ الَّذِينَ اتَّبَعُوكَ فَوْقَ الَّذِينَ كَفَرُوا إِلَىٰ يَوْمِ الْقِيَامَةِ ۖ ثُمَّ إِلَيَّ مَرْجِعُكُمْ فَأَحْكُمُ بَيْنَكُمْ فِيمَا كُنتُمْ فِيهِ تَخْتَلِفُونَ ياد ڪر جڏھن الله چيو ته عيسىٰ آءٌ توکي ماريندڙ ۽ پاڻ ڏانھن مٿاھون ڪندڙ آھيان ۽ جن ڪُفر ڪيو تن جي بھتانن کان توکي پاڪ ڪندڙ آھيان ۽ جن تنھنجي تابعداري ڪئي تن کي قيامت تائين ڪافرن تي مٿاھون غالب ڪندڙ آھيان وري مون ڏانھن اوھان جو موٽڻ آھي پوءِ اوھان جي وچ ۾ جنھن ڳالھ بابت اوھين تڪرار ڪندا آھيو تنھن بابت نبيرو ڪندس فَأَمَّا الَّذِينَ كَفَرُوا فَأُعَذِّبُهُمْ عَذَابًا شَدِيدًا فِي الدُّنْيَا وَالْآخِرَةِ وَمَا لَهُم مِّن نَّاصِرِينَ پوءِ اُنھن مان جن ڪُفر ڪيو تن کي دُنيا ۽ آخرت ۾ بلڪل سخت سزا ڏيندس ۽ اُنھن جو ڪو واھرو نه آھي وَأَمَّا الَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ فَيُوَفِّيهِمْ أُجُورَهُمْ ۗ وَاللَّهُ لَا يُحِبُّ الظَّالِمِينَ ۽ اُنھن مان جن ايمان آندو۽ چڱا ڪم ڪيا تن کي الله سندن پورھيو پُورو ڏيندو ۽ الله ظالمن کي دوست نه رکندو آھي ذَٰلِكَ نَتْلُوهُ عَلَيْكَ مِنَ الْآيَاتِ وَالذِّكْرِ الْحَكِيمِ اھو قصّو جو توتي پڙھون ٿا سو نشانين ۽ حِڪمت واري ڪتاب مان آھي إِنَّ مَثَلَ عِيسَىٰ عِندَ اللَّهِ كَمَثَلِ آدَمَ ۖ خَلَقَهُ مِن تُرَابٍ ثُمَّ قَالَ لَهُ كُن فَيَكُونُ الله وٽ عيسىٰ جي پيدا ڪرڻ جو مثال آدم جي مثال جھڙوئي آھي اُن آدم کي مِٽيءَ مان بڻايائين وري اُن چيائين ته ٿي پؤ ته ٿي پيو الْحَقُّ مِن رَّبِّكَ فَلَا تَكُن مِّنَ الْمُمْتَرِينَ اِھا ڳالھ تنھنجي پالڻھار وٽان سچي آھي تنھنڪري شڪ ڪندڙن مان نه ھُج سڏيون پوءِ زاريءَ سان دُعا گھرون پوءِ الله جي لَعنت ڪُوڙن تي ڪريون إِنَّ هَٰذَا لَهُوَ الْقَصَصُ الْحَقُّ ۚ وَمَا مِنْ إِلَٰهٍ إِلَّا اللَّهُ ۚ وَإِنَّ اللَّهَ لَهُوَ الْعَزِيزُ الْحَكِيمُ اِھو قِصو اللَّهِ ۚ فَإِن تَوَلَّوْا فَقُولُوا اشْهَدُوا بِأَنَّا مُسْلِمُونَ پيغمبر چؤ ته ڪتاب وارؤ ھڪ ڳالھ ڏانھن اچو جا اسان ۽ اوھان جي وچ ۾ ھڪ جھڙي آھي ته الله کانسواءِ ڪنھن جي عبادت نه ڪريون ۽ نڪي ساڻس ڪنھن جيڪڏھن وٽن تمام گھڻو مال امانت رکين ته اُھو سمورو توکي ورائي ڏيندا ۽ منجھائن ڪي اھڙا آھن جو جيڪڏھن وٽن ھِڪ مُھر امانت رکين ته مٿن سدائين بيھڻ کانسواءِ توکي ورائي نه ڏيندا اِھو نه ڏئڻ ھن سببان آھي جو رَبَّانِيِّينَ بِمَا كُنتُمْ تُعَلِّمُونَ الْكِتَابَ وَبِمَا كُنتُمْ تَدْرُسُونَ ڪنھن ماڻھوءَ کي نه جڳائيندو آھي جو الله کيس ڪتاب ۽ دانائي ۽ پيغمبري ڏئي اُھو وري ماڻھن کي چوي ته الله کانسواءِ مُنھنجا اللَّهُ مِيثَاقَ النَّبِيِّينَ لَمَا آتَيْتُكُم مِّن كِتَابٍ وَحِكْمَةٍ ثُمَّ جَاءَكُمْ رَسُولٌ مُّصَدِّقٌ لِّمَا مَعَكُمْ لَتُؤْمِنُنَّ بِهِ وَلَتَنصُرُنَّهُ ۚ قَالَ أَأَقْرَرْتُمْ وَأَخَذْتُمْ اوھين شاھد ھجو ۽ آءٌ اوھان سان گڏ شاھد آھيان فَمَن تَوَلَّىٰ بَعْدَ ذَٰلِكَ فَأُولَٰئِكَ هُمُ الْفَاسِقُونَ پوءِ جيڪي ھن پڃاڻان ڦرندا سي ئي بي دين آھن أَفَغَيْرَ دِينِ اللَّهِ يَبْغُونَ وَلَهُ غَنِيٌّ عَنِ الْعَالَمِينَ جنھن ۾ پڌريون نشانيون آھن ھڪڙي مقام ابراھيم آھي جو اُن ۾ ابراھيم جي پيرن جو نشان آھي ۽ جيڪو اُتي داخل ٿيو سو امن وارو آھي ۽ الله لاءِ خانه ڪعبه جو حج اُنھن ماڻھن تي لازم آھي جنھن کي ڏانھس واٽ وڃڻ جي سگھه آھي ۽ جنھن انڪار ڪيو ته الله کي پرواہ نه آھي ڇو ته الله جھانن کان بي پرواہ آھي قُلْ يَا أَهْلَ الْكِتَابِ لِمَ تَكْفُرُونَ بِآيَاتِ اللَّهِ وَاللَّهُ شَهِيدٌ عَلَىٰ مَا تَعْمَلُونَ پيغمبر کين چؤ ته ڪتاب وارؤ الله جي آيتن کي ڇونه ٿا مڃيو ۽ حقيقت ڪري جيڪي ڪندا آھيو تنھن تي الله حاضر آھي قُلْ يَا أَهْلَ الْكِتَابِ لِمَ تَصُدُّونَ عَن سَبِيلِ اللَّهِ مَنْ آمَنَ تَبْغُونَهَا عِوَجًا وَأَنتُمْ شُهَدَاءُ ۗ وَمَا اللَّهُ بِغَافِلٍ عَمَّا تَعْمَلُونَ پيغمبر چؤ ته ڪتاب وارؤ جيڪو ايمان آڻي تنھن کي الله جي واٽ کان اُن واٽ جي ڏنگائي گھُري ڇو ٿا جھليو ھن ھوندي جو اوھين ڄاڻندڙ آھيو ۽ جيڪي اوھين ڪندا آھيو تنھن کان الله بي خبر نه آھي يَا أَيُّهَا الَّذِينَ آمَنُوا إِن تُطِيعُوا فَرِيقًا مِّنَ الَّذِينَ أُوتُوا الْكِتَابَ يَرُدُّوكُم بَعْدَ إِيمَانِكُمْ كَافِرِينَ ايمان وارؤ جيڪڏھن ڪتاب وارن مان ڪنھن ٽوليءَ جي فرمانبرداري ڪندؤ ته اوھان جي ايمان آڻڻ کان پوءِ اوھان کي ڪافر ڪري ڦيرائيندا وَكَيْفَ تَكْفُرُونَ وَأَنتُمْ تُتْلَىٰ عَلَيْكُمْ آيَاتُ اللَّهِ وَفِيكُمْ رَسُولُهُ ۗ وَمَن يَعْتَصِم بِاللَّهِ فَقَدْ هُدِيَ إِلَىٰ صِرَاطٍ مُّسْتَقِيمٍ ۽ اوھين ھن ھوندي ڪئن ڪافر ٿا ٿيو جو الله جون آيتون اوھان تي پڙھجن ٿيون ۽ سندس پيغمبر اوھان ۾ آھي ۽ جيڪو الله جي دين کي چنبڙي پٺندو تنھن کي بيشڪ سڌيءَ واٽ ڏانھن رستو ڏيکاريو ويندو يَا أَيُّهَا الَّذِينَ آمَنُوا اتَّقُوا اللَّهَ حَقَّ تُقَاتِهِ وَلَا تَمُوتُنَّ إِلَّا وَأَنتُم مُّسْلِمُونَ مؤمنؤ الله کان اِئين ڊڄو جئن کانئس ڊڄڻ جو حق آھي ۽ اوھين مسلمان ھجڻ کانسواءِ نه مرجو وَاعْتَصِمُوا بِحَبْلِ اللَّهِ جَمِيعًا وَلَا تَفَرَّقُوا ۚ وَاذْكُرُوا نِعْمَتَ اللَّهِ عَلَيْكُمْ إِذْ كُنتُمْ أَعْدَاءً فَأَلَّفَ بَيْنَ قُلُوبِكُمْ فَأَصْبَحْتُم بِنِعْمَتِهِ إِخْوَانًا وَكُنتُمْ عَلَىٰ شَفَا حُفْرَةٍ مِّنَ النَّارِ فَأَنقَذَكُم مِّنْهَا ۗ كَذَٰلِكَ يُبَيِّنُ اللَّهُ لَكُمْ آيَاتِهِ لَعَلَّكُمْ تَهْتَدُونَ ۽ الله جي رسيءَ يعني قرآن کي سڀئي چنبڙي وٺو ۽ ڌاروڌار نه ٿيو ۽ پاڻ تي الله جي نعمت کي ياد ڪريو جو جڏھن پاڻ ۾ ويري ھيؤ پوءِ اوھان جي دلين ۾ ميلاپ وڌائين پوءِ سندس فضل سان ھڪ ٻئي جا ڦري ڀائر ٿيؤ ۽ اوھين دوزخ جي کڏ جي ڀَر تي ھيؤ پوءِ اوھان کي کانئس ڇڏايائين اھڙيءَ طرح الله اوھان لاءِ پنھنجيون نشانيون بيان ڪندو آھي ته مانَ اوھين ھدايت لھو وَلْتَكُن مِّنكُمْ أُمَّةٌ يَدْعُونَ إِلَى الْخَيْرِ وَيَأْمُرُونَ بِالْمَعْرُوفِ وَيَنْهَوْنَ عَنِ الْمُنكَرِ ۚ وَأُولَٰئِكَ هُمُ الْمُفْلِحُونَ ۽ اوھان ۾ اھڙي ٽولي ھئڻ گھرجي جي چڱائيءَ ڏانھن سڏين ۽ چڱن ڪمن جو حُڪم ڪن ۽ مَدن ڪمن کان جھلين ۽ اھي ئي ڇٽل آھن وَلَا تَكُونُوا كَالَّذِينَ تَفَرَّقُوا وَاخْتَلَفُوا مِن بَعْدِ مَا جَاءَهُمُ الْبَيِّنَاتُ ۚ وَأُولَٰئِكَ لَهُمْ عَذَابٌ عَظِيمٌ ۽ اوھين اُنھن جھڙا نه ٿيو جي فرقو فرقو ٿي ويا ۽ وٽن معجزا آيا اُن کانپوءِ بهتڪرار ڪيائون ۽ اُنھن لاءِ وڏو عذاب آھي يَوْمَ تَبْيَضُّ وُجُوهٌ وَتَسْوَدُّ وُجُوهٌ ۚ فَأَمَّا الَّذِينَ اسْوَدَّتْ وُجُوهُهُمْ أَكَفَرْتُم بَعْدَ إِيمَانِكُمْ فَذُوقُوا الْعَذَابَ بِمَا كُنتُمْ تَكْفُرُونَ اُنھيءَ ڏينھن جو ڪي مُنھن اڇا ٿيندا ۽ ڪي مُنھن ڪارا ٿيندا پوءِ جن جا مُنھن ڪارا ٿيندا چيو ويندن ته اوھين پنھنجي اِيمان آڻڻ کانپوءِ ڇو ڪافر ٿيؤ پوءِ اُنھيءَ سببان جو اوھين ڪافر ٿيا ھيؤ عذاب جو مزو چکو وَأَمَّا الَّذِينَ ابْيَضَّتْ وُجُوهُهُمْ فَفِي رَحْمَةِ اللَّهِ هُمْ فِيهَا خَالِدُونَ ۽ جن جا منھن اڇا ھوندا سي الله جي رحمت ھيٺ رھندا اُھي منجھس سدائين رھڻ وارا ھوندا تِلْكَ آيَاتُ اللَّهِ نَتْلُوهَا عَلَيْكَ بِالْحَقِّ ۗ وَمَا اللَّهُ يُرِيدُ ظُلْمًا لِّلْعَالَمِينَ پيغمبر اِھي الله جون آيتون آھن جي اسين توتي سچيءَ طرح پڙھون ٿا ۽ الله جھانن لاءِ ظُلم ڪرڻ نه گھرندو آھي وَلِلَّهِ مَا فِي السَّمَاوَاتِ وَمَا فِي الْأَرْضِ ۚ وَإِلَى اللَّهِ تُرْجَعُ الْأُمُورُ ۽ جيڪي آسمانن ۾ آھي ۽ جيڪي زمين ۾ آھي سو خاص الله جو آھي ۽ سڀ ڪم الله ڏانھن موٽايا ويندا كُنتُمْ خَيْرَ أُمَّةٍ أُخْرِجَتْ لِلنَّاسِ تَأْمُرُونَ بِالْمَعْرُوفِ وَتَنْهَوْنَ عَنِ الْمُنكَرِ وَتُؤْمِنُونَ بِاللَّهِ ۗ وَلَوْ آمَنَ أَهْلُ الْكِتَابِ لَكَانَ خَيْرًا لَّهُم ۚ مِّنْهُمُ الْمُؤْمِنُونَ وَأَكْثَرُهُمُ الْفَاسِقُونَ اوھين اھڙي ڀَلي اُمت ماڻھن لاءِ پيدا ٿيل آھيو جو چڱن ڪمن جو حُڪم ڪندا آھيو ۽ مدن ڪمن کان جھليندا آھيو ۽ الله کي مڃيندا آھيو ۽ جيڪڏھن ڪتاب وارا ايمان آڻين ھا ته ضرور کين ڀَلو ھُجي ھا منجھانئن ڪي مؤمن آھن ۽ منجھانئن گھڻا بي دين آھن لَن يَضُرُّوكُمْ إِلَّا أَذًى ۖ وَإِن يُقَاتِلُوكُمْ يُوَلُّوكُمُ الْأَدْبَارَ ثُمَّ لَا يُنصَرُونَ اِھي زباني اِيذاءُ کانسواءِ اوھان کي ڪڏھن ڏکوئي نه سگھندا ۽ جيڪڏھن اوھان سان وڙھندا ته اوھان کي پُٺي ڏيئي ڀڄندا وري مدد نه ڏبن ضُرِبَتْ عَلَيْهِمُ الذِّلَّةُ أَيْنَ مَا ثُقِفُوا إِلَّا بِحَبْلٍ مِّنَ اللَّهِ وَحَبْلٍ مِّنَ النَّاسِ وَبَاءُوا بِغَضَبٍ مِّنَ اللَّهِ وَضُرِبَتْ عَلَيْهِمُ الْمَسْكَنَةُ ۚ ذَٰلِكَ بِأَنَّهُمْ كَانُوا يَكْفُرُونَ بِآيَاتِ اللَّهِ وَيَقْتُلُونَ الْأَنبِيَاءَ بِغَيْرِ حَقٍّ ۚ ذَٰلِكَ بِمَا عَصَوا وَّكَانُوا يَعْتَدُونَ الله جي ذمي ۽ ماڻھن جي پناھ کانسواءِ جتي لڌا ويندا اُتي مٿن خواري ھنئي ويئي آھي ۽ الله جي ڏمر ھيٺ وري وريا ۽ مٿن محتاجي ھنئي ويئي آھي اِھا سزا ھن سببان آھي جو الله جي آيتن کي نه ٿي مڃيائون ۽ پيغمبرن کي ناحق ٿي ڪُٺائون اِھي عادتون اِنھيءَ سبب آھن جو بي فرماني ڪيائون ۽ حد کان لنگھندڙ ھئا لَيْسُوا سَوَاءً ۗ مِّنْ أَهْلِ الْكِتَابِ أُمَّةٌ قَائِمَةٌ يَتْلُونَ آيَاتِ اللَّهِ آنَاءَ اللَّيْلِ وَهُمْ يَسْجُدُونَ اِھي سڀ ھڪ جھڙا نه آھن ڪتاب وارن مان ھڪڙي اھڙي سڌي ٽولي آھي جو الله جون آيتون رات جي وقتن ۾ پڙھندا آھن ۽ اُھي نماز پڙھندا آھن يُؤْمِنُونَ بِاللَّهِ وَالْيَوْمِ الْآخِرِ وَيَأْمُرُونَ بِالْمَعْرُوفِ وَيَنْهَوْنَ عَنِ الْمُنكَرِ وَيُسَارِعُونَ فِي الْخَيْرَاتِ وَأُولَٰئِكَ مِنَ الصَّالِحِينَ الله ۽ قيامت جي ڏينھن کي مڃيندا آھن ۽ چڱن ڪمن جو حڪم ڪندا آھن ۽ مَدن ڪمن کان جھليندا آھن ۽ چڱاين يعني دين مڃڻ ۾ اڳرائي ڪندا آھن ۽ اِھي ڀلارن مان آھن وَمَا يَفْعَلُوا مِنْ خَيْرٍ فَلَن يُكْفَرُوهُ ۗ وَاللَّهُ عَلِيمٌ بِالْمُتَّقِينَ ۽ جيڪا ڪا چڱائي ڪندا آھن تنھن جي بي قدري پٽيءَ نه ڪبي ۽ الله پرھيزگارن کي ڄاڻندڙ آھي إِنَّ الَّذِينَ كَفَرُوا لَن تُغْنِيَ عَنْهُمْ أَمْوَالُهُمْ وَلَا أَوْلَادُهُم مِّنَ اللَّهِ شَيْئًا ۖ وَأُولَٰئِكَ أَصْحَابُ النَّارِ ۚ هُمْ فِيهَا خَالِدُونَ جن اِنڪار ڪيو تن کان نڪي سندن مال سندن اولاد الله جو ڪُجھ عذاب ٽاري سگھندا ۽ اُھي دوزخي آھن اُھي منجھس سدائين رھڻ وارا آھن مَثَلُ مَا يُنفِقُونَ فِي هَٰذِهِ الْحَيَاةِ الدُّنْيَا كَمَثَلِ رِيحٍ فِيهَا صِرٌّ أَصَابَتْ حَرْثَ قَوْمٍ ظَلَمُوا أَنفُسَهُمْ فَأَهْلَكَتْهُ ۚ وَمَا ظَلَمَهُمُ اللَّهُ وَلَٰكِنْ أَنفُسَهُمْ يَظْلِمُونَ ھن دنيا جي حياتيءَ ۾ جيڪي رياءُ لاءِ خرچيندا آھن تنھن جو مثال اُنھيءَ واءُ جي مثال وانگر آھي جنھن ۾ پارو ھجي جو اُن ٽوليءَ جي پوکه کي پھچي جن پاڻ تي ظلم ڪيو پوءِ اُن سندن پوکه کي ناس ڪري ۽ الله ساڻن ظلم نه ڪيو آھي پر اُھي پاڻ تي ظلم ڪن ٿا يَا أَيُّهَا الَّذِينَ آمَنُوا لَا تَتَّخِذُوا بِطَانَةً مِّن دُونِكُمْ لَا يَأْلُونَكُمْ خَبَالًا وَدُّوا مَا عَنِتُّمْ قَدْ بَدَتِ الْبَغْضَاءُ مِنْ أَفْوَاهِهِمْ وَمَا تُخْفِي صُدُورُهُمْ أَكْبَرُ ۚ قَدْ بَيَّنَّا لَكُمُ الْآيَاتِ ۖ إِن كُنتُمْ تَعْقِلُونَ اِيمان وارؤ پنھنجن مسلمانن کانسواءِ ٻين قومن کي ڳجھو دوست ڪري نه مڃيو جو اوھان جي خرابي ڪرڻ ۾ ڪا گھٽتائي نه ڪندا آھن اوھان جي ڏکه کي دوست رکيائون بيشڪ سندن واتن مان دشمني پڌري ثواب گھرندو تنھن کي اُن مان ڏينداسون ۽ شاڪرن کي چڱو ثواب ستت ڏينداسون وَكَأَيِّن مِّن نَّبِيٍّ قَاتَلَ مَعَهُ رِبِّيُّونَ كَثِيرٌ فَمَا وَهَنُوا لِمَا أَصَابَهُمْ فِي سَبِيلِ اللَّهِ وَمَا ضَعُفُوا وَمَا اسْتَكَانُوا ۗ وَاللَّهُ يُحِبُّ الصَّابِرِينَ ۽ پيغمبرن مان گھڻا ڪافرن سان وڙھيا ساڻن گھڻا خداءِ وارا ھئا پوءِ جيڪي الله جي واٽ ۾ کين پھتو تنھن سببان نڪي سُست ٿيا ۽ نڪي ھيڻا ٿيا ۽ نڪي عاجزي ڪيائون ۽ الله صابرن کي دوست رکندو آھي وَمَا كَانَ قَوْلَهُمْ إِلَّا أَن قَالُوا رَبَّنَا اغْفِرْ لَنَا ذُنُوبَنَا وَإِسْرَافَنَا فِي أَمْرِنَا وَثَبِّتْ أَقْدَامَنَا وَانصُرْنَا عَلَى الْقَوْمِ الْكَافِرِينَ ۽ سندن چوڻ ھن چوڻ کانسواءِ نه ھو ته اسان جا پالڻھار اسان کي اسان جاگناہ بخش ۽ اسان جي ڪم ۾ اسان جي بيجا ھلت بخش ۽ اسان جا پير ثابت رک ۽ اسان کي ڪافرن جي ٽوليءَ تي فتح ڏي فَآتَاهُمُ اللَّهُ ثَوَابَ الدُّنْيَا وَحُسْنَ ثَوَابِ الْآخِرَةِ ۗ وَاللَّهُ يُحِبُّ الْمُحْسِنِينَ پوءِ الله کين دنيا جو ثواب ۽ آخرت جو چڱو ثواب ڏنو ۽ الله ڀلارن کي دوست رکندو آھي يَا أَيُّهَا الَّذِينَ آمَنُوا إِن تُطِيعُوا الَّذِينَ كَفَرُوا يَرُدُّوكُمْ عَلَىٰ أَعْقَابِكُمْ فَتَنقَلِبُوا خَاسِرِينَ اِيمان وارؤ جيڪڏھن اوھين ڪافرن جو چيو مڃيندؤ ته اوھان کي اوھان جي کڙين ڀَرپوءِ تي ڦيرائيندا پوءِ اوھين ڦِري ڇيئي وارا ٿيندؤ بَلِ اللَّهُ مَوْلَاكُمْ ۖ وَهُوَ خَيْرُ النَّاصِرِينَ بلڪ الله اوھان جو سائين آھي ۽ اُھو چڱو مدد ڏيندڙ آھي سَنُلْقِي فِي قُلُوبِ الَّذِينَ كَفَرُوا الرُّعْبَ بِمَا أَشْرَكُوا بِاللَّهِ مَا لَمْ يُنَزِّلْ بِهِ سُلْطَانًا ۖ وَمَأْوَاهُمُ النَّارُ ۚ وَبِئْسَ مَثْوَى الظَّالِمِينَ ڪافرن جي دلين ۾ ھن سببان سگھو دھشت وجھنداسون جو الله سان اھڙيءَ شيءِ کي شريڪ ڪيائون جنھن لاءِ الله ڪا سَنَد نه لاٿي آھي ۽ سندن جاءِ باہ آھي ۽ اُھو ظالمن جو ھنڌ بڇڙو آھي وَلَقَدْ صَدَقَكُمُ اللَّهُ وَعْدَهُ إِذْ تَحُسُّونَهُم بِإِذْنِهِ ۖ حَتَّىٰ إِذَا فَشِلْتُمْ وَتَنَازَعْتُمْ فِي الْأَمْرِ وَعَصَيْتُم مِّن بَعْدِ مَا أَرَاكُم مَّا تُحِبُّونَ ۚ مِنكُم مَّن يُرِيدُ الدُّنْيَا وَمِنكُم مَّن يُرِيدُ الْآخِرَةَ ۚ ثُمَّ صَرَفَكُمْ عَنْهُمْ لِيَبْتَلِيَكُمْ ۖ وَلَقَدْ عَفَا عَنكُمْ ۗ وَاللَّهُ ذُو فَضْلٍ عَلَى الْمُؤْمِنِينَ ۽ بيشڪ الله اوھان سان پنھنجو وعدو سچو ڪيو آھي جڏھن ته اُنھن ڪافرن کي ايستائين سندس حُڪم سان ڪُٺوَ جيستائين بي ھمت ٿيؤ ۽ ڪم پوري ڪرڻ ۾ تڪرار ڪيوَ ۽ جيڪي دوست رکندا ھيؤ سو اوھان کي ڏيکاريائين تنھن کانپوءِ بي فرماني ڪيوَ اوھان مان ڪو اھڙو آھي جو دُنيا کي گھرندو آھي ۽ اوھان مان ڪو اھڙو آھي جو آخرت کي گھرندو آھي وري اوھان کي اُنھن کان ھن لاءِ ڦيرايائين ته اوھان کي پرکي ۽ بيشڪ اوھان کي معافي ڏنائين ۽ الله مؤمنن تي فضل ڪرڻ وارو آھي إِذْ تُصْعِدُونَ وَلَا تَلْوُونَ عَلَىٰ أَحَدٍ وَالرَّسُولُ يَدْعُوكُمْ فِي أُخْرَاكُمْ فَأَثَابَكُمْ غَمًّا بِغَمٍّ لِّكَيْلَا تَحْزَنُوا عَلَىٰ مَا فَاتَكُمْ وَلَا مَا أَصَابَكُمْ ۗ وَاللَّهُ خَبِيرٌ بِمَا تَعْمَلُونَ جڏھن اوھين پري ٿي ڀَڳؤ ۽ ڪنھن ھڪ تي ڪنڌ ڦيري نه ٿي واجھايوَ ۽ پيغمبر اوھان کي اوھان جي پُٺيان سڏيندو رھيو پوءِ خداءِ اوھان کي ڏک تي ڏک جو بدلو ڏنو اِھو ھن لاءِ ته جيڪي اوھان کان ٿي ويو نڪي تنھن تي ۽ جيڪي اوھان کي پھتو نڪي تنھن تي غمگين ٿيو ۽ جيڪي اوھان ڪندا آھيو تنھن جي الله خبر رکندڙ آھي ثُمَّ أَنزَلَ عَلَيْكُم مِّن بَعْدِ الْغَمِّ أَمَنَةً نُّعَاسًا يَغْشَىٰ طَائِفَةً مِّنكُمْ ۖ وَطَائِفَةٌ قَدْ أَهَمَّتْهُمْ أَنفُسُهُمْ يَظُنُّونَ بِاللَّهِ غَيْرَ الْحَقِّ ظَنَّ الْجَاهِلِيَّةِ ۖ يَقُولُونَ هَل لَّنَا مِنَ الْأَمْرِ مِن شَيْءٍ ۗ قُلْ إِنَّ الْأَمْرَ كُلَّهُ لِلَّهِ ۗ يُخْفُونَ فِي أَنفُسِهِم مَّا لَا يُبْدُونَ لَكَ ۖ يَقُولُونَ لَوْ كَانَ لَنَا مِنَ الْأَمْرِ شَيْءٌ مَّا قُتِلْنَا هَاهُنَا ۗ قُل لَّوْ كُنتُمْ فِي بُيُوتِكُمْ لَبَرَزَ الَّذِينَ كُتِبَ عَلَيْهِمُ الْقَتْلُ إِلَىٰ مَضَاجِعِهِمْ ۖ وَلِيَبْتَلِيَ اللَّهُ مَا فِي صُدُورِكُمْ وَلِيُمَحِّصَ مَا فِي قُلُوبِكُمْ ۗ وَاللَّهُ عَلِيمٌ بِذَاتِ الصُّدُورِ وري ڏک کانپوءِ اوھان تي آرام جا اوجھراڪي ھئي لاٿائين جو اوھان مان ھڪ ٽوليءَ کي ڍڪيندي رھي ۽ ٻي ٽولي ھئي جنھن کي سندن نفسن غمگين ڪيو ھو ناحق جاھليت جو گمان الله تي ڀانيائون ٿي چيائون ٿي ته اسان لاءِ اِنھيءَ ڪم مان ڇا ڪجھ ٿيڻو آھي پيغمبر چؤ ته سڀ اختيار الله جو ئي آھي جيڪي توتي پڌرو نه ڪندا آھن سو پنھنجين دلين ۾ لڪائيندا آھن چوندا آھن ته جيڪڏھن اسان جي وس ۾ ڪجھ ڪم ھجي ھا ته اسين ھِتِ نه ڪُسون ھا پيغمبر کين چؤ ته جيڪڏھن اوھين پنھنجن گھرن ۾ ھجو ھا ته جن ڪُسڻ لکيل ھو سي ضرور پنھنجي ڪوس جي جاين ڏانھن نڪرن ھا ۽ ھن لاءِ ته جيڪي اوھان جي سينن ۾ آھي سو الله پرکي ۽ جيڪي اوھان جي دلين ۾ آھي سو کرو ڪري ۽ الله سينن وارو ڀيد ڄاڻندڙ آھي إِنَّ الَّذِينَ تَوَلَّوْا مِنكُمْ يَوْمَ الْتَقَى الْجَمْعَانِ إِنَّمَا اسْتَزَلَّهُمُ الشَّيْطَانُ بِبَعْضِ مَا كَسَبُوا ۖ وَلَقَدْ عَفَا اللَّهُ عَنْهُمْ ۗ إِنَّ اللَّهَ غَفُورٌ حَلِيمٌ اوھان مان جن اُنھيءَ ڏينھن مُنھن موڙيو جو ٻه ٽوليون ھڪ ٻئي جي سامھون ٿيون ته اُنھن کي شيطان سندن ڪجھ ڪِئي جي شامت کان سواءِ نه ٿيڙيو ۽ بيشڪ الله اُنھن کي معافي ڏني ڇوته الله بخشڻھار بُردبار آھي يَا أَيُّهَا الَّذِينَ آمَنُوا لَا تَكُونُوا كَالَّذِينَ كَفَرُوا وَقَالُوا لِإِخْوَانِهِمْ إِذَا ضَرَبُوا فِي الْأَرْضِ أَوْ كَانُوا غُزًّى لَّوْ كَانُوا عِندَنَا مَا مَاتُوا وَمَا قُتِلُوا لِيَجْعَلَ اللَّهُ ذَٰلِكَ حَسْرَةً فِي قُلُوبِهِمْ ۗ وَاللَّهُ يُحْيِي وَيُمِيتُ ۗ وَاللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ ايمان وارؤ اُنھن ڪافرن وانگر نه ٿيو جي پنھنجن ڀائرن لاءِ جڏھن اُھي ملڪ ۾ مسافري ڪن يا لڙائي ڪندڙ ھجن تڏھن چون ٿا ته جيڪڏھن اسان وٽ ھجن ھا ته نه مرن ھا ۽ نڪي ڪُسن ھا ته الله اُھو اَرمان سندن دلين ۾ ڪري ۽ الله جياريندو ۽ ماريندو آھي ۽ جيڪي اوھين ڪندا آھيو سو الله ڏسندڙ آھي وَلَئِن قُتِلْتُمْ فِي سَبِيلِ اللَّهِ أَوْ مُتُّمْ لَمَغْفِرَةٌ مِّنَ اللَّهِ وَرَحْمَةٌ خَيْرٌ مِّمَّا يَجْمَعُونَ ۽ جيڪڏھن اوھين الله جي واٽ ۾ ڪُسي پيؤ يا مري پيؤ ته جيڪي گڏ ڪندا آھن تنھن کان الله جي بخشش ۽ ٻاجھ بلڪل ڀلي آھي وَلَئِن مُّتُّمْ أَوْ قُتِلْتُمْ لَإِلَى اللَّهِ تُحْشَرُونَ ۽ جيڪڏھن اوھين مري وڃو يا ڪُسي پئو ته ضرور الله ڏانھن گڏ ڪيا ويندؤ فَبِمَا رَحْمَةٍ مِّنَ اللَّهِ لِنتَ لَهُمْ ۖ وَلَوْ كُنتَ فَظًّا غَلِيظَ الْقَلْبِ لَانفَضُّوا مِنْ حَوْلِكَ ۖ فَاعْفُ عَنْهُمْ وَاسْتَغْفِرْ لَهُمْ وَشَاوِرْهُمْ فِي الْأَمْرِ ۖ فَإِذَا عَزَمْتَ فَتَوَكَّلْ عَلَى اللَّهِ ۚ إِنَّ اللَّهَ يُحِبُّ الْمُتَوَكِّلِينَ پوءِ پيغمبر الله جي ٻاجھ سببان اُنھن لاءِ نرم ٿيو آھين ۽ جيڪڏھن ڪاوڙيل سخت سخت دل ھجين ھا ته ضرور تنھنجي آسپاس کان ڀڄي اُٿن ھا پوءِ کين معافي ڏي ۽ اُنھن لاءِ بخشش گھُر ۽ ساڻن ڪم ۾ صلاح ڪر پوءِ جڏھن تون پڪو اِرادو ڪرين تڏھن الله تي ڀروسو ڪر ڇوته الله ڀروسي ڪرڻ وارن کي دوست رکندو آھي إِن يَنصُرْكُمُ اللَّهُ فَلَا غَالِبَ لَكُمْ ۖ وَإِن يَخْذُلْكُمْ فَمَن ذَا الَّذِي يَنصُرُكُم مِّن بَعْدِهِ ۗ وَعَلَى اللَّهِ فَلْيَتَوَكَّلِ الْمُؤْمِنُونَ جيڪڏھن الله اوھان کي مدد ڏيندو ته اوھان تي ڪو غالب ٿيڻو نه آھي ۽ جيڪڏھن اوھان کي مدد نه ڏيندو ته اُن کانپوءِ ڪير اوھان کي مدد ڏيندو ۽ مؤمنن کي جڳائي ته الله تي ڀروسو ڪن وَمَا كَانَ لِنَبِيٍّ أَن يَغُلَّ ۚ وَمَن يَغْلُلْ يَأْتِ بِمَا غَلَّ يَوْمَ الْقِيَامَةِ ۚ ثُمَّ تُوَفَّىٰ كُلُّ نَفْسٍ مَّا كَسَبَتْ وَهُمْ لَا يُظْلَمُونَ ۽ ڪنھين پيغمبر کي غنيمت جي مال ۾ خيانت ڪرڻ نه جڳائي ۽ جيڪو خيانت ڪندو سو جيڪا شيء خيانت ڪيائين تنھن سميت قيامت جي ڏينھن ايندو پوءِ سڀڪنھن ماڻھوءَ جيڪي ڪمايو تنھن جو پورو بدلو کيس ڏنو ويندو ۽ اُنھن تي ظلم نه ڪبو أَفَمَنِ اتَّبَعَ رِضْوَانَ اللَّهِ كَمَن بَاءَ بِسَخَطٍ مِّنَ اللَّهِ وَمَأْوَاهُ جَهَنَّمُ ۚ وَبِئْسَ الْمَصِيرُ جيڪو الله جي رضامنديءَ جي پٺيان لڳو سو اُن جھڙو آھي ڇا جيڪو الله جي ڪاوڙ ھيٺ ڦري آيو ۽ سندس جاءِ دوزخ آھي ۽ اُھو ورڻ جو ھنڌ بُڇڙو آھي هُمْ دَرَجَاتٌ عِندَ اللَّهِ ۗ وَاللَّهُ بَصِيرٌ بِمَا يَعْمَلُونَ اُھي الله وٽ وڏن درجن وارا آھن ۽ جيڪي ڪندا آھن سو الله ڏسندڙ آھي لَقَدْ مَنَّ اللَّهُ عَلَى الْمُؤْمِنِينَ إِذْ بَعَثَ فِيهِمْ رَسُولًا مِّنْ أَنفُسِهِمْ يَتْلُو عَلَيْهِمْ آيَاتِهِ وَيُزَكِّيهِمْ وَيُعَلِّمُهُمُ الْكِتَابَ وَالْحِكْمَةَ وَإِن كَانُوا مِن قَبْلُ لَفِي ضَلَالٍ مُّبِينٍ بيشڪ الله مؤمنن تي وڏو احسان ڪيو آھي جڏھن ته انھن ۾ سندن قوم مان پيغمبر موڪليائين جو کين سندس آيتون پڙھي ٻڌائيندو آھي ۽ کين پاڪ ڪندو آھي ۽ کين ڪتاب ۽ دانائي سيکاريندو آھي بيشڪ ھن کان اڳ پڌريءَ ڀل ۾ ڀليل ھؤا أَوَلَمَّا أَصَابَتْكُم مُّصِيبَةٌ قَدْ أَصَبْتُم مِّثْلَيْهَا قُلْتُمْ أَنَّىٰ هَٰذَا ۖ قُلْ هُوَ مِنْ عِندِ أَنفُسِكُمْ ۗ إِنَّ اللَّهَ عَلَىٰ كُلِّ شَيْءٍ قَدِيرٌ ۽ جڏھن اوھانکي جنگ اُحد ۾ ڏک پھتو توڻيڪ پنھنجن دشمنن کي اُن جو ٻيڻو ڏُک بَدۡر ۾ پھچايو ھيو ته چيوَ ٿي ته ھي مصيبت ڪٿان پھتي آھي پيغمبر چؤ ته اُھا اوھان کي پاڻ وٽان پھتي آھي ڇوته الله سڀڪنھن شيء تي وسوارو آھي وَمَا أَصَابَكُمْ يَوْمَ الْتَقَى الْجَمْعَانِ فَبِإِذْنِ اللَّهِ وَلِيَعْلَمَ الْمُؤْمِنِينَ ۽ اُنھيءَ ڏينھن جو ٻه ٽوليون پاڻ ۾ سامھون ٿيون تنھن ڏينھن جيڪي اوھانکي پھتو سو الله جي حُڪم سان ھو ۽ ھن لاءِ ته مؤمنن کي نِکيڙي وَلِيَعْلَمَ الَّذِينَ نَافَقُوا ۚ وَقِيلَ لَهُمْ تَعَالَوْا قَاتِلُوا فِي سَبِيلِ اللَّهِ أَوِ ادْفَعُوا ۖ قَالُوا لَوْ نَعْلَمُ قِتَالًا لَّاتَّبَعْنَاكُمْ ۗ هُمْ لِلْكُفْرِ يَوْمَئِذٍ أَقْرَبُ مِنْهُمْ لِلْإِيمَانِ ۚ يَقُولُونَ بِأَفْوَاهِهِم مَّا لَيْسَ فِي قُلُوبِهِمْ ۗ وَاللَّهُ أَعْلَمُ بِمَا يَكْتُمُونَ ۽ منافقن کي نِکيڙي ۽ کين چيو ويو ته اچو الله جي واٽ ۾ وڙھو يا ڪافرن کي جواب ڪريو چيائون ته جيڪڏھن اسان کي ڪنھن جنگ ٿيڻ جي خبر ھجي ھا ته ضرور اوھان سان ھجون ھا اُھي اُن ڏينھن پاڻ اِيمان کان ڪفر ڏي بلڪل ويجھو ھؤا پنھنجن واتن سان اُھي ڳالھيون چوندا آھن جيڪي سندين دلين ۾ نه آھن ۽ جيڪي لڪائيندا آھن سو الله بلڪل ڄاڻندڙ آھي الَّذِينَ قَالُوا لِإِخْوَانِهِمْ وَقَعَدُوا لَوْ أَطَاعُونَا مَا قُتِلُوا ۗ قُلْ فَادْرَءُوا عَنْ أَنفُسِكُمُ الْمَوْتَ إِن كُنتُمْ صَادِقِينَ ته اِھي اُھي آھن جن پنھنجن ڀائرن کي چيو ۽ ويٺا رھيا ته جيڪڏھن اُھي اسان جو چيو مڃن ھا ته نه ڪُسن ھا پيغمبر کين چؤ ته جيڪڏھن اوھين سچا آھيو ته پاڻ کان موت کي ٽاريو وَلَا تَحْسَبَنَّ الَّذِينَ قُتِلُوا فِي سَبِيلِ اللَّهِ أَمْوَاتًا ۚ بَلْ أَحْيَاءٌ عِندَ رَبِّهِمْ يُرْزَقُونَ ۽ جيڪي الله جي واٽ ۾ ڪُٺا ويا تن کي مئل پٽيءَ نه ڀانءِ بلڪ اُھي جيئرا آھن ۽ پنھنجي پالڻھار وٽ روزي ڏنل آھن فَرِحِينَ بِمَا آتَاهُمُ اللَّهُ مِن فَضْلِهِ وَيَسْتَبْشِرُونَ بِالَّذِينَ لَمْ يَلْحَقُوا بِهِم مِّنْ خَلْفِهِمْ أَلَّا خَوْفٌ عَلَيْهِمْ وَلَا هُمْ يَحْزَنُونَ جيڪي الله اُنھن کي پنھنجي فضل سان ڏنو آھي تنھن حال سان خوش ٿيندڙ آھن ۽ جيڪي سندن پٺيان آھن تن مان جيڪي ساڻن نه گڏيا آھن اُنھن لاءِ ھن ڪري خوش ٿيندا آھن ته کين ڪو ڀؤ نه آھي ۽ نڪي اُھي غمگين رھندا يَسْتَبْشِرُونَ بِنِعْمَةٍ مِّنَ اللَّهِ وَفَضْلٍ وَأَنَّ اللَّهَ لَا يُضِيعُ أَجْرَ الْمُؤْمِنِينَ الله جي نعمت ۽ فضل سان خوش ٿيندا آھن ۽ الله مؤمنن جو اجر نه وڃائيندو آھي الَّذِينَ اسْتَجَابُوا لِلَّهِ وَالرَّسُولِ مِن بَعْدِ مَا أَصَابَهُمُ الْقَرْحُ ۚ لِلَّذِينَ أَحْسَنُوا مِنْهُمْ وَاتَّقَوْا أَجْرٌ عَظِيمٌ جن پاڻ کي ڦٽ پھچڻ کان پوءِ الله ۽ رسول جو حُڪم قبول ڪيو انھن مان جن چڱائي ڪئي ۽ ڊنا تن لاءِ وڏو اجر آھي الَّذِينَ قَالَ لَهُمُ النَّاسُ إِنَّ النَّاسَ قَدْ جَمَعُوا لَكُمْ فَاخْشَوْهُمْ فَزَادَهُمْ إِيمَانًا وَقَالُوا حَسْبُنَا اللَّهُ وَنِعْمَ الْوَكِيلُ جن کي ماڻھن چيو ته ڪافرن ماڻھن اوھان لاءِ لشڪر گڏ ڪيو آھي تنھن ڪري کانئن ڊڄو پوءِ اُن چوڻ ھٿون اُنھن جو ايمان وڌايو ۽ چيائون ته اسان کي الله ڪافي آھي ۽ اُھو چڱو وڪيل ڀرجھلو آھي فَانقَلَبُوا بِنِعْمَةٍ مِّنَ اللَّهِ وَفَضْلٍ لَّمْ يَمْسَسْهُمْ سُوءٌ وَاتَّبَعُوا رِضْوَانَ اللَّهِ ۗ وَاللَّهُ ذُو فَضْلٍ عَظِيمٍ پوءِ اُھي مسلمان الله جي نعمت ۽ فضل سان موٽيا کين ڪا سختي نه پھتي ۽ الله جي راضپي پٺيان لڳا ۽ الله وڏي فضل وارو آھي إِنَّمَا ذَٰلِكُمُ الشَّيْطَانُ يُخَوِّفُ أَوْلِيَاءَهُ فَلَا تَخَافُوهُمْ وَخَافُونِ إِن كُنتُم مُّؤْمِنِينَ اُھو اوھان کي جنگ کان جھليندڙ شيطان کان سواءِ ٻيو نه آھي جو پنھنجن دوستن کي ڊيڄاريندو آھي جيڪڏھن اوھين مؤمن آھيو ته اُنھن ڪافرن کان نه ڊڄو ۽ مون کان ڊڄو وَلَا يَحْزُنكَ الَّذِينَ يُسَارِعُونَ فِي الْكُفْرِ ۚ إِنَّهُمْ لَن يَضُرُّوا اللَّهَ شَيْئًا ۗ يُرِيدُ اللَّهُ أَلَّا يَجْعَلَ لَهُمْ حَظًّا فِي الْآخِرَةِ ۖ وَلَهُمْ عَذَابٌ عَظِيمٌ ۽ پيغمبر جيڪي ڪفر جي سوڀ ۾ اڳرائي ڪندا آھن سي توکي غمگين نه ڪن ڇو ته اُھي الله کي ڪجھ نقصان نه لائيندا الله گھرندو آھي ته اُنھن کي آخرت ۾ ڪجھ بھرو نه ڏي ۽ اُنھن لاءِ وڏو عذاب آھي إِنَّ الَّذِينَ اشْتَرَوُا الْكُفْرَ بِالْإِيمَانِ لَن يَضُرُّوا اللَّهَ شَيْئًا وَلَهُمْ عَذَابٌ أَلِيمٌ جن اِيمان جي بدران ڪفر ڳڌو سي الله کي ڪجھ نقصان نه لائيندا ۽ اُنھن لاءِ ڏکوئيندڙ عڊاب آھي وَلَا يَحْسَبَنَّ الَّذِينَ كَفَرُوا أَنَّمَا نُمْلِي لَهُمْ خَيْرٌ لِّأَنفُسِهِمْ ۚ إِنَّمَا نُمْلِي لَهُمْ لِيَزْدَادُوا إِثْمًا ۚ وَلَهُمْ عَذَابٌ مُّهِينٌ ۽ ڪافر ھيءُ نه ڀانئن ته کين اسان جي مُھلت ڏيڻ اُنھن لاءِ ڀلي آھي اُنھن کي ھن کانسواءِ ڍر نه ڏني سون ته گناھ ۾ وڌن ۽ انھن لاءِ خوار ڪندڙ عذاب آھي مَّا كَانَ اللَّهُ لِيَذَرَ الْمُؤْمِنِينَ عَلَىٰ مَا أَنتُمْ عَلَيْهِ حَتَّىٰ يَمِيزَ الْخَبِيثَ مِنَ الطَّيِّبِ ۗ وَمَا كَانَ اللَّهُ لِيُطْلِعَكُمْ عَلَى الْغَيْبِ وَلَٰكِنَّ اللَّهَ يَجْتَبِي مِن رُّسُلِهِ مَن يَشَاءُ ۖ فَآمِنُوا بِاللَّهِ وَرُسُلِهِ ۚ وَإِن تُؤْمِنُوا وَتَتَّقُوا فَلَكُمْ أَجْرٌ عَظِيمٌ منافقؤ الله کي نه جڳائيندو آھي ته جنھن تي اوھين آھيو تنھن تي مؤمنن کي ڇڏي ڏي تان جو پليت کي پاڪ کان ڌار ڪري ۽ الله کي نه جڳائيندو آھي جو اوھان کي ڳُجھ تي واقف ڪري پر الله پنھنجن پيغمبرن مان جنھن کي گھرندو آھي تنھن کي چونڊيندو آھي پوءِ الله ۽ سندس پيغمبرن کي مڃيو ۽ جيڪڏھن اوھين مڃيندؤ ۽ ڊڄندؤ ته اوھان لاءِ وڏو اجر آھي وَلَا يَحْسَبَنَّ الَّذِينَ يَبْخَلُونَ بِمَا آتَاهُمُ اللَّهُ مِن فَضْلِهِ هُوَ خَيْرًا لَّهُم ۖ بَلْ هُوَ شَرٌّ لَّهُمْ ۖ سَيُطَوَّقُونَ مَا بَخِلُوا بِهِ يَوْمَ الْقِيَامَةِ ۗ وَلِلَّهِ مِيرَاثُ السَّمَاوَاتِ وَالْأَرْضِ ۗ وَاللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌ ۽ جيڪي الله پنھنجي فضل سان کين ڏنو آھي تنھن ۾ جيڪي بخل ڪن سي اِئين پٽيءَ نه ڀانئين ته اُھو بُخل کين ڪو چڱو آھي بلڪ اُھو اُنھن لاءِ بڇڙو آھي جنھن شيء جو بخل ڪيائون سا قيامت جي ڏينھن ڳٽ ڪري وجھبين ۽ آسمانن ۽ زمين جي ميراث الله جي آھي ۽ جيڪي ڪندا آھيو تنھن جي الله خبر رکندڙ آھي لَّقَدْ سَمِعَ اللَّهُ قَوْلَ الَّذِينَ قَالُوا إِنَّ اللَّهَ فَقِيرٌ وَنَحْنُ أَغْنِيَاءُ ۘ سَنَكْتُبُ مَا قَالُوا وَقَتْلَهُمُ الْأَنبِيَاءَ بِغَيْرِ حَقٍّ وَنَقُولُ ذُوقُوا عَذَابَ الْحَرِيقِ بيشڪ الله اُنھن جو چوڻ ٻڌو جن چيو ته الله سڃو آھي ۽ اَسين شاھوڪار آھيون جيڪي چيائون سو لکنداسون ۽ اُنھن جو پيغمبرن کي ناحق ڪھڻ ۽ آخرت ۾ چونداسون ته ساڙيندڙ عذاب جو مزو چکو ذَٰلِكَ بِمَا قَدَّمَتْ أَيْدِيكُمْ وَأَنَّ اللَّهَ لَيْسَ بِظَلَّامٍ لِّلْعَبِيدِ اھا سزا اِنھيءَ سببان آھي جو اوھان جي ھٿن اڳ موڪليو آھي ۽ ھن سببان ته الله ٻانھن تي ظلم ڪندڙ نه آھي الَّذِينَ قَالُوا إِنَّ اللَّهَ عَهِدَ إِلَيْنَا أَلَّا نُؤْمِنَ لِرَسُولٍ حَتَّىٰ يَأْتِيَنَا بِقُرْبَانٍ تَأْكُلُهُ النَّارُ ۗ قُلْ قَدْ جَاءَكُمْ رُسُلٌ مِّن قَبْلِي بِالْبَيِّنَاتِ وَبِالَّذِي قُلْتُمْ فَلِمَ قَتَلْتُمُوهُمْ إِن كُنتُمْ صَادِقِينَ جيڪي چون ٿا ته الله اسان سان انجام ڪيو آھي ته ڪنھن پيغمبر تي ايستائين ايمان نه آڻيون جيستائين ھو اھڙي قرباني اسان وٽ نه آڻي جو اُن قربانيءَ کي باہ کائي پيغمبر کين چئو ته بيشڪ مون کان اڳ ڪيترن پيغمبرن مُعجزا ۽ ٻيو جيڪي چيوَ سو اوھان وٽ آندو پوءِ جيڪڏھن اوھين سچا آھيو ته اُنھن پيغمبرن کي ڇالاءِ ڪٺوَ فَإِن كَذَّبُوكَ فَقَدْ كُذِّبَ رُسُلٌ مِّن قَبْلِكَ جَاءُوا بِالْبَيِّنَاتِ وَالزُّبُرِ وَالْكِتَابِ الْمُنِيرِ پوءِ پيغمبر جيڪڏھن توکي ڪوڙو ڀانيائون ته بيشڪ توکان اڳ ڪيترا پيغمبر ڪوڙا ڀانيا ويا ھؤا جن ڪيتريون حجتون ۽ ڪيترا صحيفا ۽ نوراني ڪتاب وٽن آندا كُلُّ نَفْسٍ ذَائِقَةُ الْمَوْتِ ۗ وَإِنَّمَا تُوَفَّوْنَ أُجُورَكُمْ يَوْمَ الْقِيَامَةِ ۖ فَمَن زُحْزِحَ عَنِ النَّارِ وَأُدْخِلَ الْجَنَّةَ فَقَدْ فَازَ ۗ وَمَا الْحَيَاةُ الدُّنْيَا إِلَّا مَتَاعُ الْغُرُورِ سڀڪو جيءُ موت جو مزو چکندڙ آھي ۽ ھن کان سواءِ ٻيوڪجھ ڪرڻو نه آھي جو اوھان کي قيامت جي ڏينھن پورو اجر ڏبو پوءِ جنھن کي باہ کان پري ڪيو ويو ۽ بھشت ۾ گھيڙيو ويو سو بيشڪ مراد کي پھتو ۽ دنيا جي حياتي ٺڳيءَ جي سامان کانسواءِ نه آھي لَتُبْلَوُنَّ فِي أَمْوَالِكُمْ وَأَنفُسِكُمْ وَلَتَسْمَعُنَّ مِنَ الَّذِينَ أُوتُوا الْكِتَابَ مِن قَبْلِكُمْ وَمِنَ الَّذِينَ أَشْرَكُوا أَذًى كَثِيرًا ۚ وَإِن تَصْبِرُوا وَتَتَّقُوا فَإِنَّ ذَٰلِكَ مِنْ عَزْمِ الْأُمُورِ اوھين پنھنجن مالن ۽ پنھنجين جانين ۾ ضرور پرکبو ۽ جن کي اوھان کان اڳ ڪتاب ڏنو ويو آھي تن کان ۽ مشرڪن کان گھڻيون گاريون ضرور ٻڌندؤ ۽ جيڪڏھن صبر ڪندؤ ۽ الله کان ڊڄندؤ ته اِھا عادت ھمت جي ڪمن مان آھي وَإِذْ أَخَذَ اللَّهُ مِيثَاقَ الَّذِينَ أُوتُوا الْكِتَابَ لَتُبَيِّنُنَّهُ لِلنَّاسِ وَلَا تَكْتُمُونَهُ فَنَبَذُوهُ وَرَاءَ ظُهُورِهِمْ وَاشْتَرَوْا بِهِ ثَمَنًا قَلِيلًا ۖ فَبِئْسَ مَا يَشْتَرُونَ ۽ ياد ڪر جڏھن الله ڪتاب وارن کان ھن لاءِ پڪو انجام ورتو ته اُن ڪتاب کي ماڻھن لاءِ بيان ڪجو ۽ اُن کي نه لڪائجو پوءِ اُن انجام کي پنھنجين پٺين پوئتان اُڇلايائون ۽ اُن جي عيوض ٿورو مُلھ ڳڌائون پوءِ جيڪي ڳنھندا آھن سو بڇڙو آھي لَا تَحْسَبَنَّ الَّذِينَ يَفْرَحُونَ بِمَا أَتَوا وَّيُحِبُّونَ أَن يُحْمَدُوا بِمَا لَمْ يَفْعَلُوا فَلَا تَحْسَبَنَّهُم بِمَفَازَةٍ مِّنَ الْعَذَابِ ۖ وَلَهُمْ عَذَابٌ أَلِيمٌ پيغمبر تون اُنھن کي چڱو نه ڀانءِ جيڪي پنھنجي ڪئي تي خوش ٿيندا آھن ۽ جيڪي نه ڪيائون تنھن تي پنھنجي ساراھڻ کي دوست رکندا آھن پوءِ تون اُنھن کي عذاب کان ڇٽل نه ڀانءِ ۽ اُنھن لاءِ ڏکوئيندڙ عذاب آھي وَلِلَّهِ مُلْكُ السَّمَاوَاتِ وَالْأَرْضِ ۗ وَاللَّهُ عَلَىٰ كُلِّ شَيْءٍ قَدِيرٌ ۽ آسمانن ۽ زمين جي بادشاھي الله جي آھي ۽ الله سڀڪنھن شيءِ تي وسوارو آھي إِنَّ فِي خَلْقِ السَّمَاوَاتِ وَالْأَرْضِ وَاخْتِلَافِ اللَّيْلِ وَالنَّهَارِ لَآيَاتٍ لِّأُولِي الْأَلْبَابِ آسمانن ۽ زمين جي بڻاوت ۾ ۽ رات ۽ ڏينھن جي ڦير گھير ۾ اُنھن سمجھ وارن لاءِ ضرور نشانيون آھن الَّذِينَ يَذْكُرُونَ اللَّهَ قِيَامًا وَقُعُودًا وَعَلَىٰ جُنُوبِهِمْ وَيَتَفَكَّرُونَ فِي خَلْقِ السَّمَاوَاتِ وَالْأَرْضِ رَبَّنَا مَا خَلَقْتَ هَٰذَا بَاطِلًا سُبْحَانَكَ فَقِنَا عَذَابَ النَّارِ جيڪي بيٺي ۽ ويٺي ۽ پنھنجن پاسن ڀر الله کي ياد ڪندا آھن ۽ آسمانن ۽ زمين جي بڻاوت ۾ فڪر ڪندا آھن ۽ چوندا آھن ته اسانجا پالڻھار ھنن کي بي رٿو نه بڻايو اَٿيئي توکي پاڪ ڄاڻون ٿا تنھنڪري اسان کي باہ جي عذاب کان بچاءِ رَبَّنَا إِنَّكَ مَن تُدْخِلِ النَّارَ فَقَدْ أَخْزَيْتَهُ ۖ وَمَا لِلظَّالِمِينَ مِنْ أَنصَارٍ اسانجا پالڻھار جنھن کي تو باہ ۾ گھيڙيو تنھن کي بيشڪ خوار ڪيئي ۽ ظالمن جو ڪو مددگار نه آھي رَّبَّنَا إِنَّنَا سَمِعْنَا مُنَادِيًا يُنَادِي لِلْإِيمَانِ أَنْ آمِنُوا بِرَبِّكُمْ فَآمَنَّا ۚ رَبَّنَا فَاغْفِرْ لَنَا ذُنُوبَنَا وَكَفِّرْ عَنَّا سَيِّئَاتِنَا وَتَوَفَّنَا مَعَ الْأَبْرَارِ اسانجا پالڻھار اسان سڏيندڙ جي سڏ کي ٻڌو جو ايمان ڏانھن سڏيندو آھي ته پنھنجي پالڻھار تي ايمان آڻيو پوءِ مڃيوسون اسانجا پالڻھار اسان کي اسانجا گناہ بخش ۽ اسان کان اسان جون مدايون ميٽ ۽ اسان کي ڀلارن سان گڏ مار رَبَّنَا وَآتِنَا مَا وَعَدتَّنَا عَلَىٰ رُسُلِكَ وَلَا تُخْزِنَا يَوْمَ الْقِيَامَةِ ۗ إِنَّكَ لَا تُخْلِفُ الْمِيعَادَ اسانجا پالڻھار اسان کي پنھنجن پيغمبرن جي زبان سان جيڪي انجام ڏنو اٿيئي سو پار ۽ قيامت جي ڏينھن اسان کي خوار نه ڪج ڇوته تون وعدي خلافي نه ڪندو آھين فَاسْتَجَابَ لَهُمْ رَبُّهُمْ أَنِّي لَا أُضِيعُ عَمَلَ عَامِلٍ مِّنكُم مِّن ذَكَرٍ أَوْ أُنثَىٰ ۖ بَعْضُكُم مِّن بَعْضٍ ۖ فَالَّذِينَ هَاجَرُوا وَأُخْرِجُوا مِن دِيَارِهِمْ وَأُوذُوا فِي سَبِيلِي وَقَاتَلُوا وَقُتِلُوا لَأُكَفِّرَنَّ عَنْهُمْ سَيِّئَاتِهِمْ وَلَأُدْخِلَنَّهُمْ جَنَّاتٍ تَجْرِي مِن تَحْتِهَا الْأَنْهَارُ ثَوَابًا مِّنْ عِندِ اللَّهِ ۗ وَاللَّهُ عِندَهُ حُسْنُ الثَّوَابِ پوءِ سندن پالڻھار اُنھن جي دعا قبول ڪئي ته آءٌ اوھان مان عمل ڪندڙ مڙس يا زال جو عمل نه وڃائيندس اوھان مان ڪي ڪن مان پيدا ٿيل آھن يعني سڀ ھڪ جنس آھيو پوءِ جن وطن ڇڏيو ۽ پنھنجن گھرن مان ڪڍيا ويا ۽ مُنھنجي واٽ ۾ ڏکويا ويا ۽ وڙھيا ۽ ڪُٺا ويا تن کان سندن مدايون ضرور ميٽيندس ۽ اُنھن کي بھشت جي اھڙن باغن ۾ ضرور داخل ڪندس جن جي ھيٺان واھيون وھنديون آھن اِھو بدلو الله وٽان آھي ۽ الله وٽ چڱو بدلو آھي لَا يَغُرَّنَّكَ تَقَلُّبُ الَّذِينَ كَفَرُوا فِي الْبِلَادِ پيغمبر ملڪن ۾ ڪافرن جي اَچ وڃ توکي نه ٺڳي مَتَاعٌ قَلِيلٌ ثُمَّ مَأْوَاهُمْ جَهَنَّمُ ۚ وَبِئْسَ الْمِهَادُ اِھو فائدو ٿورو آھي وري سندن جاءِ دوزخ آھي ۽ اُھو بڇڙو ھنڌ آھي لَٰكِنِ الَّذِينَ اتَّقَوْا رَبَّهُمْ لَهُمْ جَنَّاتٌ تَجْرِي مِن تَحْتِهَا الْأَنْهَارُ خَالِدِينَ فِيهَا نُزُلًا مِّنْ عِندِ اللَّهِ ۗ وَمَا عِندَ اللَّهِ خَيْرٌ لِّلْأَبْرَارِ پر جيڪي پنھنجي پالڻھار کان ڊنا تن لاءِ باغ آھن جن جي ھيٺان واھيون وھنديون آھن اُن ۾ سدائين رھڻ وارا آھن اِھا مھماني الله وٽان آھي ۽ جيڪي الله وٽ آھي سو ڀلارن لاءِ چڱو آھي وَإِنَّ مِنْ أَهْلِ الْكِتَابِ لَمَن يُؤْمِنُ بِاللَّهِ وَمَا أُنزِلَ إِلَيْكُمْ وَمَا أُنزِلَ إِلَيْهِمْ خَاشِعِينَ لِلَّهِ لَا يَشْتَرُونَ بِآيَاتِ اللَّهِ ثَمَنًا قَلِيلًا ۗ أُولَٰئِكَ لَهُمْ أَجْرُهُمْ عِندَ رَبِّهِمْ ۗ إِنَّ اللَّهَ سَرِيعُ الْحِسَابِ ۽ ڪتاب وارن مان ڪي اُھي آھن جيڪي الله کي ۽ جيڪي توڏانھن قرآن لاٿو ويو تنھن کي ۽ جيڪي توريت اُنھن ڏانھن لاٿو ويو تنھن کي الله جي عاجزي ڪندڙ ٿي مڃيندا آھن الله جي آيتن کي ٿوري مُلھ سان نه وڪڻندا آھن اِھي اُھي آھن جن لاءِ سندن پالڻھار وٽ سندن اجر آھي ڇوته الله سگھو حساب وٺندڙ آھي يَا أَيُّهَا الَّذِينَ آمَنُوا اصْبِرُوا وَصَابِرُوا وَرَابِطُوا وَاتَّقُوا اللَّهَ لَعَلَّكُمْ تُفْلِحُونَ ايمان وارؤ صبر ڪريو ۽ جھاد لاءِ بيھڪ وارا ٿيو ۽ مُحڪم رھو ۽ الله کان ڊڄو ته مانَ اوھين ڇٽو توحيد رباني بديع الدين شاه الراشدي
