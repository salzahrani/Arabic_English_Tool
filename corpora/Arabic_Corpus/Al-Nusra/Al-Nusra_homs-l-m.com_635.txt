ދިވެހި
Sorah Yusuf (Joseph ) - Verses Number 111
الر ۚ تِلْكَ آيَاتُ الْكِتَابِ الْمُبِينِ ( 1 )
ا ل ر (މި އަކުރުތަކުގެ مراد ދެނެވޮޑިގެންވަނީ اللَّه އެވެ.) މިއީ ފާޅުވެ ބަޔާންވެގެންވާ، ފޮތުގެ آية ތަކެވެ.
إِنَّا أَنزَلْنَاهُ قُرْآنًا عَرَبِيًّا لَّعَلَّكُمْ تَعْقِلُونَ ( 2 )
ހަމަކަށަވަރުން، ތިމަންރަސްކަލާނގެ އެ ފޮތް ބާވައިލެއްވީ، عرب ބަހުންއޮތް قرآن އެއް ކަމުގައެވެ. އެއީ ތިޔަބައިމީހުން ވިސްނައި ފިކުރުކުރުމަށްޓަކައެވެ.
نَحْنُ نَقُصُّ عَلَيْكَ أَحْسَنَ الْقَصَصِ بِمَا أَوْحَيْنَا إِلَيْكَ هَٰذَا الْقُرْآنَ وَإِن كُنتَ مِن قَبْلِهِ لَمِنَ الْغَافِلِينَ ( 3 )
އެންމެ ރިވެތި ވާހަކަތައް، ތިމަންރަސްކަލާނގެ ކަލޭގެފާނަށް ކިޔައިދެއްވަމެވެ. (އެއީ) ތިމަންރަސްކަލާނގެ، ކަލޭގެފާނަށް މި قرآن وحى ކުރެއްވި ކުރެއްވުމުންނެވެ. އޭގެކުރިން، ކަލޭގެފާނު އެ ވާހަކަތައް ނުދަންނަ މީހުންގެ ތެރެއިން ވީނަމަވެހެވެ.
إِذْ قَالَ يُوسُفُ لِأَبِيهِ يَا أَبَتِ إِنِّي رَأَيْتُ أَحَدَ عَشَرَ كَوْكَبًا وَالشَّمْسَ وَالْقَمَرَ رَأَيْتُهُمْ لِي سَاجِدِينَ ( 4 )
يوسف ގެފާނު، އެކަލޭގެފާނުގެ ބައްޕާއަށް ވިދާޅުވިހިނދު، ހަނދުމަކުރާށެވެ! އޭ ބައްޕާއެވެ! ހަމަކަށަވަރުން، އެގާރަ ތަރިޔާއި، އިރާއި، ހަނދު، ތިމަންކަލޭގެފާނަށް ހުވަފެނުން ފެނިއްޖެއެވެ. ފެނުނީ، އެތަކެތި ތިމަންކަލޭގެފާނަށް سجدة ކުރާތީއެވެ.
قَالَ يَا بُنَيَّ لَا تَقْصُصْ رُؤْيَاكَ عَلَىٰ إِخْوَتِكَ فَيَكِيدُوا لَكَ كَيْدًا ۖ إِنَّ الشَّيْطَانَ لِلْإِنسَانِ عَدُوٌّ مُّبِينٌ ( 5 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އޭ ދަރިފުޅާއެވެ! ތިބާޔަށް ފެނުނު ހުވަފެން ތިބާގެ ބޭބޭމެންނަށް ކިޔައިނުދޭށެވެ! ފަހެ، އެއުރެން ތިބާޔަށް ނުބައި ރޭވުމެއް ރާވައިފާނެތެވެ. ހަމަކަށަވަރުން، شيطان އާއަކީ، އިންސާނާއަށްހުރި، ފާޅުވެގެންވާ عداوة ތެރިއެކެވެ.
وَكَذَٰلِكَ يَجْتَبِيكَ رَبُّكَ وَيُعَلِّمُكَ مِن تَأْوِيلِ الْأَحَادِيثِ وَيُتِمُّ نِعْمَتَهُ عَلَيْكَ وَعَلَىٰ آلِ يَعْقُوبَ كَمَا أَتَمَّهَا عَلَىٰ أَبَوَيْكَ مِن قَبْلُ إِبْرَاهِيمَ وَإِسْحَاقَ ۚ إِنَّ رَبَّكَ عَلِيمٌ حَكِيمٌ ( 6 )
އަދި ހަމައެފަދައިން، ތިބާގެ ވެރިރަސްކަލާނގެ، ތިބާ خيار ކުރައްވައި، ހުވަފެނުގެ ވާހަކަތައް، މާނަކުރުން ތިބާޔަށް ދަސްކޮށްދެއްވައި، އަދި ތިބާޔަށާއި، يعقوب ގެ ދަރިކޮޅަށް، އެކަލާނގެ نعمة ތައް ފުރިހަމަކޮށްދެއްވާނެތެވެ. (އެއީ) މީގެކުރިން، ތިބާގެ ދެކާފައިން ކަމަށްވާ، إبراهيم ގެފާނާއި، إسحاق ގެފާނަށް އެ نعمة ތައް ފުރިހަމަކޮށްދެއްވި ފަދައިންނެވެ. ހަމަކަށަވަރުން، ތިބާގެ ވެރިރަސްކަލާނގެއީ، މޮޅަށް ދެނެވޮޑިގެންވާ، حكيم ވަންތަ ރަސްކަލާނގެއެވެ.
لَّقَدْ كَانَ فِي يُوسُفَ وَإِخْوَتِهِ آيَاتٌ لِّلسَّائِلِينَ ( 7 )
يوسف ގެފާނާއި، އެކަލޭގެފާނުގެ أخ ންގެ ވާހަކައިގައި، سؤال ކުރާ މީހުންނަށް، ހެކިތަކެއްވާކަން ކަށަވަރެވެ.
إِذْ قَالُوا لَيُوسُفُ وَأَخُوهُ أَحَبُّ إِلَىٰ أَبِينَا مِنَّا وَنَحْنُ عُصْبَةٌ إِنَّ أَبَانَا لَفِي ضَلَالٍ مُّبِينٍ ( 8 )
އެ أخ ންތައް، ބުނިހިނދު، ހަނދުމަކުރާށެވެ! ތިމަންމެންނީ، ގަދަފަދަ جماعة އެއްކަމުގައި ވާހިނދުވެސް، يوسف ގެފާނާއި، އޭނާގެ أخ (بنيامين) ތިމަންމެންގެ ބައްޕާގެ އަރިހުގައި ލޮބުވެތިކަން މާބޮޑެވެ. ހަމަކަށަވަރުން، ތިމަންމެންގެ ބައްޕާވަނީ، ފާޅުވެގެންވާ މަގުއޮޅުމެއްގައެވެ.
اقْتُلُوا يُوسُفَ أَوِ اطْرَحُوهُ أَرْضًا يَخْلُ لَكُمْ وَجْهُ أَبِيكُمْ وَتَكُونُوا مِن بَعْدِهِ قَوْمًا صَالِحِينَ ( 9 )
ތިޔަބައިމީހުން يوسف ގެފާނު قتل ކޮށްލާށެވެ! ނުވަތަ، ދުރު ބިމަކަށް އެކަލޭގެފާނު އެއްލައިލާށެވެ! އޭރުން ތިޔަބައިމީހުންގެ ބައްޕާގެ ހެެޔޮކަން ތިޔަބައިމީހުންނަށް އެކަނި ލިބޭނެތެވެ. އަދި އެއަށްފަހު ތިޔަބައިމީހުން (توبة ވެ) صالح ރަނގަޅު ބައެއްކަމުގައި ވާނީއެވެ.
قَالَ قَائِلٌ مِّنْهُمْ لَا تَقْتُلُوا يُوسُفَ وَأَلْقُوهُ فِي غَيَابَتِ الْجُبِّ يَلْتَقِطْهُ بَعْضُ السَّيَّارَةِ إِن كُنتُمْ فَاعِلِينَ ( 10 )
އެއުރެންގެ ތެރެއިން ބުނާ މީހަކު ބުންޏެވެ. ތިޔަބައިމީހުން يوسف ގެފާނު قتل ނުކުރާށެވެ! ތިޔަބައިމީހުން ކަމެއްކުރަންޏާ (އަނދިރި) ވަޅެއްގެ އަޑިޔަށް އެކަލޭގެފާނު އެއްލައިލާށެވެ! އޭރުން، ދަތުުރުކުރަމުންދާ ބަޔަކު، އެކަލޭގެފާނު ހޮވައިގެން ގެންގޮސްފާނެތެވެ.
قَالُوا يَا أَبَانَا مَا لَكَ لَا تَأْمَنَّا عَلَىٰ يُوسُفَ وَإِنَّا لَهُ لَنَاصِحُونَ ( 11 )
އެއުރެން ދެންނެވޫއެވެ. އޭ އަހުރެމެންގެ ބައްޕާއެވެ! يوسف ގެފާނާމެދު، ކަލޭގެފާނު ތިމަންމެންނަށް އިތުބާރު ނުކުރައްވަނީ ކީއްވެތޯއެވެ. ހަމަކަށަވަރުން، ތިމަންމެންނީ، އެކަލޭގެފާނަށް ހެޔޮއެދޭ ބަޔަކީމުއެވެ.
أَرْسِلْهُ مَعَنَا غَدًا يَرْتَعْ وَيَلْعَبْ وَإِنَّا لَهُ لَحَافِظُونَ ( 12 )
ކައިބޮއި އުފާކޮށް، ކުޅެހެދުމަށްޓަކައި، ތިމަންމެންނާއެކުގައި، މާދަމާ އެކަލޭގެފާނު ފޮނުއްވާށެވެ! ހަމަކަށަވަރުން، ތިމަންމެން އެކަލޭގެފާނު ބަލަހައްޓައި ރައްކާތެރިކުރާހުށީމުއެވެ.
قَالَ إِنِّي لَيَحْزُنُنِي أَن تَذْهَبُوا بِهِ وَأَخَافُ أَن يَأْكُلَهُ الذِّئْبُ وَأَنتُمْ عَنْهُ غَافِلُونَ ( 13 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އޭނާ ގޮވައިގެން ތިޔަބައިމީހުން ދިއުމަކީ، ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނަށް ހިތާމަ ލިބޭކަމެކެވެ. އަދި ތިޔަބައިމީހުން އޭނާގެ މަތިން غافل ވެގެންތިއްބައި، މިނިކާވަގު (އައިސް) އޭނާ ކައިފާނެތީ ތިމަންކަލޭގެފާނު ބިރުފުޅު ގަނެތެވެ.
قَالُوا لَئِنْ أَكَلَهُ الذِّئْبُ وَنَحْنُ عُصْبَةٌ إِنَّا إِذًا لَّخَاسِرُونَ ( 14 )
އެއުރެން ދެންނެވޫއެވެ. ތިމަންމެންނީ ބާރުހުރި ބަޔެއްކަމަށްވާހިނދު، މިނިކާވަގު (އައިސް) އޭނާ ކައިފިނަމަ، ހަމަކަށަވަރުން، ތިމަންމެންވާހުށީ (ކަމެއްނުކުރެވޭ) ގެއްލެނިވިގެންވާ ބަޔަކު ކަމުގައެވެ.
فَلَمَّا ذَهَبُوا بِهِ وَأَجْمَعُوا أَن يَجْعَلُوهُ فِي غَيَابَتِ الْجُبِّ ۚ وَأَوْحَيْنَا إِلَيْهِ لَتُنَبِّئَنَّهُم بِأَمْرِهِمْ هَٰذَا وَهُمْ لَا يَشْعُرُونَ ( 15 )
ފަހެ، އެއުރެން އެކަލޭގެފާނު ގޮވައިގެން ދިޔަހިނދު، އަނދިރި ވަޅަކަށް އެކަލޭގެފާނު ލުމަށް އެންމެން އެއްބަސްވޫއެވެ. ތިމަންރަސްކަލާނގެ އެކަލޭގެފާނަށް وحى ކުރެއްވީމެވެ. އެއުރެންގެ މިކަންތަކުގެ ވާހަކަ، އެއުރެންނަށް ނޭނގިތިއްބައި، ހަމަކަށަވަރުން، ކަލޭގެފާނު އެއުރެންނަށް (ފަހުން) ކިޔައިދެއްވާނެތެވެ.
وَجَاءُوا أَبَاهُمْ عِشَاءً يَبْكُونَ ( 16 )
އަދި عشاء ވަގުތުގައި، އެއުރެންގެ ބައްޕާގެ އަރިހަށް ރޮމުން އެއުރެން އައޫއެވެ.
قَالُوا يَا أَبَانَا إِنَّا ذَهَبْنَا نَسْتَبِقُ وَتَرَكْنَا يُوسُفَ عِندَ مَتَاعِنَا فَأَكَلَهُ الذِّئْبُ ۖ وَمَا أَنتَ بِمُؤْمِنٍ لَّنَا وَلَوْ كُنَّا صَادِقِينَ ( 17 )
އެއުރެން ދެންނެވޫއެވެ. އޭ އަހުރެމެންގެ ބައްޕާއެވެ! ތިމަންމެންގެ ސާމާނުތަކުގެ ކައިރީގައި يوسف ގެފާނު ބަހައްޓާފައި، ތިމަންމެން ވާދަކުރުމަށް ދިޔައީމުއެވެ. ދެން މިނިކާވަގެއްއައިސް، އެކަލޭގެފާނު ކައިިފިއެވެ. ތިމަންމެން ތެދުވެރިންކަމުގައިވިޔަސް، ކަލޭގެފާނު ތިމަންމެންގެ ބަސް ތެދެއް ނުކުރައްވާނެތެވެ.
وَجَاءُوا عَلَىٰ قَمِيصِهِ بِدَمٍ كَذِبٍ ۚ قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرًا ۖ فَصَبْرٌ جَمِيلٌ ۖ وَاللَّهُ الْمُسْتَعَانُ عَلَىٰ مَا تَصِفُونَ ( 18 )
އަދި އެއުރެން އެކަލޭގެފާނުގެ ގަމީސްކޮޅުގައި ދޮގު ލެޔެއް އުނގުޅައިގެން ގެނައޫއެވެ. (يعقوب ގެފާނު) ވިދާޅުވިއެވެ. އެހެންނަކުންނޫނެވެ. ތިޔަބައިމީހުންގެ އަމިއްލަ نفس ތައް (ނުބައި) ކަމެއްކުރުމަށް، ތިޔަބައިމީހުންނަށް ތަންދިނީއެވެ. ފަހެ، (ތިމަންކަލޭގެފާނަށް އޮތީ) ރިވެތިވެގެންވާ ކެތްކުރުމެކެވެ. އަދި ތިޔަބައިމީހުން، ސިފަކޮށް ބުނާ އެއްޗެއްގެ މައްޗަށް (ކެތްކުރުމުގައި) އެހީދެއްވާ ފަރާތަކީ اللَّه އެވެ.
وَجَاءَتْ سَيَّارَةٌ فَأَرْسَلُوا وَارِدَهُمْ فَأَدْلَىٰ دَلْوَهُ ۖ قَالَ يَا بُشْرَىٰ هَٰذَا غُلَامٌ ۚ وَأَسَرُّوهُ بِضَاعَةً ۚ وَاللَّهُ عَلِيمٌ بِمَا يَعْمَلُونَ ( 19 )
އެތަނަށް ދަތުރުކުރާ ބަޔަކު އައޫއެވެ. އަދި އެއުރެންނަށް ފެން ގެނެސްދޭމީހާ (ވަޅުކައިރިއަށް) ފޮނުވޫއެވެ. އަދި އޭނާ ބާލިދީ ވަޅަށް ތިރިކޮށްފައި (يوسف ގެފާނު ވަޅުން ނަގާފައި) ބުންޏެވެ. އޭ ހުރި އުފަލެއްގެ ބޮޑުކަމާއެވެ! މިހުރީ ކުޑަ ފިރިހެން ކުއްޖެކެވެ. ދެން ވިޔަފާރި މުދަލެއްގެ ގޮތުގައި، އެކަލޭގެފާނު ފޮރުވައި، ރައްކާކޮށްފޫއެވެ. اللَّه އީ، އެމީހުން ކުރާ ކަންތައް މޮޅަށް ދެނެވޮޑިގެންވާ ރަސްކަލާނގެއެވެ.
وَشَرَوْهُ بِثَمَنٍ بَخْسٍ دَرَاهِمَ مَعْدُودَةٍ وَكَانُوا فِيهِ مِنَ الزَّاهِدِينَ ( 20 )
އަދި ވަރަށް ކުޑަ އަގަކަށް، ގުނައިލެވޭ މަދު درهم ކޮޅަކަށް، އެއުރެން އެކަލޭގެފާނު ވިއްކާލައިފޫއެވެ. އަދި އެކަލޭގެފާނާމެދު، އެއުރެންވީ (މަދު އެތިކޮޅަކަށް ރުހޭ) زاهد ންގެ ތެރެއިންނެވެ.
وَقَالَ الَّذِي اشْتَرَاهُ مِن مِّصْرَ لِامْرَأَتِهِ أَكْرِمِي مَثْوَاهُ عَسَىٰ أَن يَنفَعَنَا أَوْ نَتَّخِذَهُ وَلَدًا ۚ وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِي الْأَرْضِ وَلِنُعَلِّمَهُ مِن تَأْوِيلِ الْأَحَادِيثِ ۚ وَاللَّهُ غَالِبٌ عَلَىٰ أَمْرِهِ وَلَٰكِنَّ أَكْثَرَ النَّاسِ لَا يَعْلَمُونَ ( 21 )
مصر ން އެކަލޭގެފާނު ގަތް މީހާ، އޭނާގެ އަންހެނުންނަށް ބުންޏެވެ. ކަނބުލޭގެ، އެކަލޭގެފާނުގެ ހުންނެވުން ރަނގަޅުގޮތަކަށް ހަމަޖައްސާށެވެ! އެއީ އެކަލޭގެފާނު، ތިމަންމެންނަށް މަންފާއެއް ކުރޭތޯއެވެ. ނުވަތަ ތިމަންމެން އެކަލޭގެފާނު ދަރިއެއްކަމުގައި ހަދާނީއެވެ. އެފަދައިން، ތިމަންރަސްކަލާނގެ يوسف ގެފާނު (مصر ގެ) ބިމުގައި ބާރުވެރި ކުރެއްވީމެވެ. އަދި ހުވަފެނުގެ ވާހަކަތައް މާނަކުރެއްވުން އެކަލޭގެފާނަށް ދަސްކޮށްދެއްވުމަށްޓަކައެވެ. اللَّه އީ، އެކަލާނގެ ކަންތައްތައް ހިންގެވުމުގައި، ބާރުވެރިވަންތަ ރަސްކަލާނގެއެވެ. އެހެނެއްކަމަކު، މީސްތަކުންކުރެ ގިނަމީހުން (އެކަމެއް) ނުދަނެތެވެ.
وَلَمَّا بَلَغَ أَشُدَّهُ آتَيْنَاهُ حُكْمًا وَعِلْمًا ۚ وَكَذَٰلِكَ نَجْزِي الْمُحْسِنِينَ ( 22 )
އެކަލޭގެފާނު ފުރާ ފުރިހަމަވީހިނދު، ތިމަންރަސްކަލާނގެ، އެކަލޭގެފާނަށް حكمة އާއި، علم ދެއްވީމެވެ. إحسان ތެރިންނަށް ތިމަންރަސްކަލާނގެ ޖަޒާދެއްވަނީ، އެފަދައިންނެވެ.
وَرَاوَدَتْهُ الَّتِي هُوَ فِي بَيْتِهَا عَن نَّفْسِهِ وَغَلَّقَتِ الْأَبْوَابَ وَقَالَتْ هَيْتَ لَكَ ۚ قَالَ مَعَاذَ اللَّهِ ۖ إِنَّهُ رَبِّي أَحْسَنَ مَثْوَايَ ۖ إِنَّهُ لَا يُفْلِحُ الظَّالِمُونَ ( 23 )
އަދި އެކަލޭގެފާނު އެކަނބުލޭގެ ގޭގައި ހުންނެވި ކަނބުލޭގެ، އެކަލޭގެފާނުގެ އަރިހުން ނުބައި ކަމަކަށް އެދުނެވެ. އަދި ދޮރުލައްޕައި، އެކަނބުލޭގެ ދެންނެވިއެވެ. ކަލޭގެފާނު (ތިމަންނާގެ ގާތަށް) ވަޑައިގަންނަވާށެވެ! އެކަލޭގެފާނު ވިދާޅުވިއެވެ. [معاذ اللَّه] އެވެ. (އެބަހީ: اللَّه ގެ ރައްކާތެރިކަމަށް އެދެމެވެ. خيانة ތެރިވުމަށް ކަނބުލޭގެ ތިޔަ އެދޭ މީހާ) އެއީ ހަމަކަށަވަރުން، ތިމަންގެ ސާހިބު މީހާއެވެ. އޭނާ ތިމަންކަލޭގެފާނު ބަހައްޓާފައިވަނީ، މާތް ރަނގަޅުގޮތަށެވެ. ހަމަކަށަވަރުން، އަނިޔާވެރިން ނުދިނޭނެތެވެ.
وَلَقَدْ هَمَّتْ بِهِ ۖ وَهَمَّ بِهَا لَوْلَا أَن رَّأَىٰ بُرْهَانَ رَبِّهِ ۚ كَذَٰلِكَ لِنَصْرِفَ عَنْهُ السُّوءَ وَالْفَحْشَاءَ ۚ إِنَّهُ مِنْ عِبَادِنَا الْمُخْلَصِينَ ( 24 )
އަދި ހަމަކަށަވަރުން (އެދުމާއެކު) އެކަނބުލޭގެ އެކަލޭގެފާނާ ކުރިމަތިލިއެވެ. އަދި އެކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ دليل ފެނިވަޑައިނުގެންނެވިނަަމަ، އެކަލޭގެފާނުގެ ހިތްޕުޅުވެސް އެކަނބުލޭގެ ފަރާތަށް ލެނބުނީހެވެ. އެފަދައިން (އެ دليل ދެއްކެވީ) ތިމަންރަސްކަލާނގެ އެކަލޭގެފާނުގެ ކިބައިން ނުބައި عمل އާއި، ހުތުރުކަންބޮޑު ފާފަ ދުރުކުރެއްވުމަށްޓަކައެވެ. ހަމަކަށަވަރުން، އެކަލޭގެފާނު ވަނީ، ތިމަންރަސްކަލާނގެ خيار ކުރެއްވި އަޅުންގެ ތެރެއިންނެވެ.
وَاسْتَبَقَا الْبَابَ وَقَدَّتْ قَمِيصَهُ مِن دُبُرٍ وَأَلْفَيَا سَيِّدَهَا لَدَى الْبَابِ ۚ قَالَتْ مَا جَزَاءُ مَنْ أَرَادَ بِأَهْلِكَ سُوءًا إِلَّا أَن يُسْجَنَ أَوْ عَذَابٌ أَلِيمٌ ( 25 )
އަދި އެ ދެމީހުންމެ ދޮރާ ދިމާއަށް އަވަސްވެގެންފޫއެވެ. އަދި އެކަނބުލޭގެ އެކަލޭގެފާނުގެ ގަމީސްކޮޅު ފުރަގަހުން (ހިފައި) ވީދައިލިއެވެ. އަދި ދޮރުކައިރިން އެކަނބުލޭގެ ފިރިމީހާ އެ ދެމީހުންނަށް ފެނުނެވެ. އެކަނބުލޭގެ ބުންޏެވެ. ތިބާގެ އަނބިމީހާއަށް ނުބައިކަމެއްކުރަން އުޅުނުމީހަކަށް، އެމީހާ ޖަލުގައި ހައްޔަރުކުރުން، ނުވަތަ ވޭންދެނިވި عقوبات ދިނުން ފިޔަވައި އޮތީ ކޮން ޖަޒާއެއް ހެއްޔެވެ؟
قَالَ هِيَ رَاوَدَتْنِي عَن نَّفْسِي ۚ وَشَهِدَ شَاهِدٌ مِّنْ أَهْلِهَا إِن كَانَ قَمِيصُهُ قُدَّ مِن قُبُلٍ فَصَدَقَتْ وَهُوَ مِنَ الْكَاذِبِينَ ( 26 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ނުބައިކަންތަކަށް ތިމަންކަލޭގެފާނުގެ ކިބައިން އެދުނީ، އެކަނބުލޭގެއެވެ. އަދި (އެކަމާމެދު) އެކަނބުލޭގެ ތިމާގެ މީހުންކުރެ ހެކިވެރިޔަކު ހެކިދީ ބުންޏެވެ. އެކަލޭގެފާނުގެ ގަމީސްކޮޅު ވީދިފައިވަނީ، ކުރިމަތިންނަމަ، އެކަނބުލޭގެ އެބުނީ ތެދެވެ. އަދި އެކަލޭގެފާނު ވަނީ ދޮގުވެރިންގެ ތެރެއިންނެވެ.
وَإِن كَانَ قَمِيصُهُ قُدَّ مِن دُبُرٍ فَكَذَبَتْ وَهُوَ مِنَ الصَّادِقِينَ ( 27 )
އަދި އެކަލޭގެފާނުގެ ގަމީސްކޮޅު ވީދިފައިވަނީ، ފުރަގަހުންނަމަ، އެކަނބުލޭގެ އެހެދީ ދޮގެވެ. އަދި އެކަލޭގެފާނު އެވަނީ ތެދުވެރިންގެ ތެެރެއިންނެވެ.
فَلَمَّا رَأَىٰ قَمِيصَهُ قُدَّ مِن دُبُرٍ قَالَ إِنَّهُ مِن كَيْدِكُنَّ ۖ إِنَّ كَيْدَكُنَّ عَظِيمٌ ( 28 )
ފަހެ، އެކަލޭގެފާނުގެ ގަމީސްކޮޅު ފުރަގަހުން ވީދިފައިވާތަން ފެނުމުން، އޭނާ (އެބަހީ: އެކަނބުލޭގެ ފިރިމީހާ) ބުންޏެވެ. ހަމަކަށަވަރުން، އެއީ ތިޔަކަނބަލުންގެ މަކަރުވެރިކަމުގެ ތެރެއިން ކަމެކެވެ. ހަމަކަށަވަރުން، ތިޔަކަނބަލުންގެ މަކަރުވެރިކަމަކީ، ބޮޑުވެގެންވާ ކަމެކެވެ.
يُوسُفُ أَعْرِضْ عَنْ هَٰذَا ۚ وَاسْتَغْفِرِي لِذَنبِكِ ۖ إِنَّكِ كُنتِ مِنَ الْخَاطِئِينَ ( 29 )
يوسف ގެފާނެވެ! މިކަންތަކުން ދުރުހެލިވެވަޑައިގަންނަވާށެވެ! އަދި ކަނބުލޭގެ ކުރި ކުށަށް މާފު އެދޭށެވެ! ހަމަކަށަވަރުން، ކަނބުލޭގެވަނީ، ކުށްވެރިންގެ ތެރެއިންނެވެ.
وَقَالَ نِسْوَةٌ فِي الْمَدِينَةِ امْرَأَتُ الْعَزِيزِ تُرَاوِدُ فَتَاهَا عَن نَّفْسِهِ ۖ قَدْ شَغَفَهَا حُبًّا ۖ إِنَّا لَنَرَاهَا فِي ضَلَالٍ مُّبِينٍ ( 30 )
(މިކަންތަކުގެ خبر ފެތުރިގެންދިޔުމުން) ރަށުތެރޭގެ އަންހެނުންގެ ބަޔަކު ބުނެފޫއެވެ. العزيز ގެ އަންހެނުން، އެކަނބުލޭގެ އަޅުކުއްޖާގެ ކިބައިން ނުބައިކަންތަކަށް އެދެނީއެވެ. ހަމަކަށަވަރުން، އެކަނބުލޭގެ ހިތްވަނީ، އެކުއްޖާގެ ލޯބިން ފުރިފައެވެ. ހަމަކަށަވަރުން، އެކަނބުލޭގެ ވާކަމަށް ތިމަންމެންދެކެނީ ބަޔާންވެގެންވާ ފުރެދުމެއްގައެވެ.
فَلَمَّا سَمِعَتْ بِمَكْرِهِنَّ أَرْسَلَتْ إِلَيْهِنَّ وَأَعْتَدَتْ لَهُنَّ مُتَّكَأً وَآتَتْ كُلَّ وَاحِدَةٍ مِّنْهُنَّ سِكِّينًا وَقَالَتِ اخْرُجْ عَلَيْهِنَّ ۖ فَلَمَّا رَأَيْنَهُ أَكْبَرْنَهُ وَقَطَّعْنَ أَيْدِيَهُنَّ وَقُلْنَ حَاشَ لِلَّهِ مَا هَٰذَا بَشَرًا إِنْ هَٰذَا إِلَّا مَلَكٌ كَرِيمٌ ( 31 )
ފަހެ، އެ އަންހެނުން ދެއްކި (މަލާމާތުގެ) މަކަރުވެރި ވާހަކަ އެކަނބުލޭގެއަށް އިވުމުން، އެއުރެން ބަލައިފޮނުވިއެވެ. އަދި ލެނގިގެންތިބޭނެ ތަނެއް ތައްޔާރުކުރިއެވެ. އަދި އެއިންކޮންމެ މީހަކަށް ފިޔޮއްސެއް ދިނެވެ. އަދި يوسف ގެފާނަށް ދެންނެވިއެވެ. އެއުރެންގެ ތެރެޔަށް ނުކުމެވަޑައިގަންނަވާށެވެ! ދެން، އެ އަންހެނުން އެކަލޭގެފާނު ދުށްހިނދު، އެއުރެން އެކަލޭގެފާނުގެ ރީތިކަމާމެދު حيران ވޫއެވެ. އަދި އެއުރެން އަތްތައް ކޮށަމުން ދިޔައޫއެވެ. އަދި ބުނެފޫއެވެ. [حاش للَّه] އެވެ. (اللَّه ގެ ހުސްطاهر ވަންތަ ކަމާއެވެ!) މިއީއަކީ آدم ގެދަރިއެއްވާ ނޫނެވެ. މިއީ ހަމަކަށަވަރުން، މާތްވެގެންވާ ملائكة އަކު ކަމުގައި ނޫނީ ނުވެތެވެ.
قَالَتْ فَذَٰلِكُنَّ الَّذِي لُمْتُنَّنِي فِيهِ ۖ وَلَقَدْ رَاوَدتُّهُ عَن نَّفْسِهِ فَاسْتَعْصَمَ ۖ وَلَئِن لَّمْ يَفْعَلْ مَا آمُرُهُ لَيُسْجَنَنَّ وَلَيَكُونًا مِّنَ الصَّاغِرِينَ ( 32 )
އެކަނބުލޭގެ ބުންޏެވެ. ތިޔަބައިމީހުން ތިމަންނާއަށް މަލާމާތްކޮށް ވާހަކަދެއްކީ، އޭނާއާ މެދުގައެވެ. ހަމަކަށަވަރުން، ތިމަން އޭނާގެ ކިބައިން ނުބައިކަންތަކަށް އެދުނީއެވެ. ފަހެ، އެހިނދު އޭނާ އެކަމުން ރައްކާތެރިވެގަތެވެ. އަދި ތިމަން އޭނާއަށް أمر ކުރާ ކަންތައް ނުކުޅަނަމަ، ޖަލުގައި އޭނާ ހައްޔަރު ކުރެވޭނެކަން ކަށަވަރެވެ. އަދި އޭނާ ވާހުށީ، ނިކަމެތިވެގެންވާ މީހުންގެ ތެރެއިންނެވެ.
قَالَ رَبِّ السِّجْنُ أَحَبُّ إِلَيَّ مِمَّا يَدْعُونَنِي إِلَيْهِ ۖ وَإِلَّا تَصْرِفْ عَنِّي كَيْدَهُنَّ أَصْبُ إِلَيْهِنَّ وَأَكُن مِّنَ الْجَاهِلِينَ ( 33 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އަޅުގެ ވެރިރަސްކަލާކޮ! ތިމަންއަޅާ ކުރުމަށް އެއުރެން އެ އެދޭކަންތަކަށްވުރެ، ޖަލުގައި ހައްޔަރުވުންވެސް، ތިމަންއަޅާއަށް ލޮބުވެތިކަން މާބޮޑެވެ. އަދި އިބަރަސްކަލާނގެ އެއުރެންގެ މަކަރުވެރިކަން ތިމަން އަޅާގެ ކިބައިން ދުރުނުކުރެއްވިނަމަ، ތިމަންއަޅާ އެއުރެންނަށް ލެނބިގެން ހިނގައިދާނެތެވެ. އަދި جاهل ންގެ ތެރެއިން ތިމަންއަޅާ ވެދާނެތެވެ.
فَاسْتَجَابَ لَهُ رَبُّهُ فَصَرَفَ عَنْهُ كَيْدَهُنَّ ۚ إِنَّهُ هُوَ السَّمِيعُ الْعَلِيمُ ( 34 )
ފަހެ، އެކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ އެކަލޭގެފާނުގެ دعاء އިޖާބަކުރައްވައި، އެއުރެންގެ މަކަރުވެރިކަން އެކަލޭގެފާނުގެ ކިބައިން ދުރުކުރެއްވިއެވެ. ހަމަކަށަވަރުން، އެކަލާނގެއީ، މޮޅަށް އައްސަވާވޮޑިގެންވާ، މޮޅަށް ދެނެވޮޑިގެންވާ ރަސްކަލާނގެއެވެ.
ثُمَّ بَدَا لَهُم مِّن بَعْدِ مَا رَأَوُا الْآيَاتِ لَيَسْجُنُنَّهُ حَتَّىٰ حِينٍ ( 35 )
ދެން (އެކަލޭގެފާނު ބަރީއަވެގެންވާކަމުގެ) دليل ތައް އެއުރެންނަށް ފެނިދިއުމަށްފަހުވެސް، ހަމަކަށަވަރުން، އެކަލޭގެފާނު ދުވަސްކޮޅަކަށް ޖަލުގައި ހައްޔަރުކުރުމަށް އެއުރެންނަށް ފެނުނެވެ.
وَدَخَلَ مَعَهُ السِّجْنَ فَتَيَانِ ۖ قَالَ أَحَدُهُمَا إِنِّي أَرَانِي أَعْصِرُ خَمْرًا ۖ وَقَالَ الْآخَرُ إِنِّي أَرَانِي أَحْمِلُ فَوْقَ رَأْسِي خُبْزًا تَأْكُلُ الطَّيْرُ مِنْهُ ۖ نَبِّئْنَا بِتَأْوِيلِهِ ۖ إِنَّا نَرَاكَ مِنَ الْمُحْسِنِينَ ( 36 )
އަދި (ރަސްގެފާނުގެ خادم ންގެ ތެރެއިންވާ) ދެކުއްޖަކުވެސް އެކަލޭގެފާނުގެ އަރިހުގައި ޖަލަށްލެވުނެވެ. އޭގެތެރެއިން އެކަކު ދެންނެވިއެވެ. ހަމަކަށަވަރުން، ރާހެދުމަށް، ތިމަން މޭބިސްކަދުރުން ދިޔަފެލާތީ ތިމަންނާއަށް ހުވަފެނުން ފެނުނެވެ. އަދި އަނެކަކު ދެންނެވިއެވެ. ހަމަކަށަވަރުން، ތިމަންގެ ބޮލުގައި ރޮށިތަކެއް ބަހައްޓައިގެން ދަނިކޮށް، ދޫނި އެއިން ކާތީ ތިމަންނާއަށް ހުވަފެނުން ފެނުނެވެ. މިކަމުގެ މާނަ ތިމަންމެންނަށް ކަލޭގެފާނު ކިޔައިދެއްވާށެވެ! ހަމަކަށަވަރުން، ތިމަންމެން ކަލޭގެފާނު ދެކެނީ، إحسان ތެރިންގެ ތެރެއިންވާ ބޭކަލަކު ކަމުގައެވެ.
قَالَ لَا يَأْتِيكُمَا طَعَامٌ تُرْزَقَانِهِ إِلَّا نَبَّأْتُكُمَا بِتَأْوِيلِهِ قَبْلَ أَن يَأْتِيَكُمَا ۚ ذَٰلِكُمَا مِمَّا عَلَّمَنِي رَبِّي ۚ إِنِّي تَرَكْتُ مِلَّةَ قَوْمٍ لَّا يُؤْمِنُونَ بِاللَّهِ وَهُم بِالْآخِرَةِ هُمْ كَافِرُونَ ( 37 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަ ދެމީހުންނަށް ލިބޭ ކާނާއެއް ތިޔަ ދެމީހުންނަށް އައުމުގެ ކުރިން، ތިމަންކަލޭގެފާނު ތިޔަ ދެމީހުންނަށް އެއީ ކޮންއެއްޗެއްކަން އަންގައިދެއްވާހުށީމެވެ. އެއީ ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ ތިމަންކަލޭގެފާނަށް އުނގަންނައިދެއްވި ކަންތަކުގެ ތެރެއިން ކަމެކެވެ. اللَّه އަށް إيمان ނުވާ ބަޔެއްގެ دين ހަމަކަށަވަރުން، ތިމަން ދޫކޮށްލައިފީމެވެ. އަދި އެއުރެންނީ، ހަމަ، آخرة އަށް كافر ވެގެންވާ ބަޔެކެވެ.
وَاتَّبَعْتُ مِلَّةَ آبَائِي إِبْرَاهِيمَ وَإِسْحَاقَ وَيَعْقُوبَ ۚ مَا كَانَ لَنَا أَن نُّشْرِكَ بِاللَّهِ مِن شَيْءٍ ۚ ذَٰلِكَ مِن فَضْلِ اللَّهِ عَلَيْنَا وَعَلَى النَّاسِ وَلَٰكِنَّ أَكْثَرَ النَّاسِ لَا يَشْكُرُونَ ( 38 )
އަދި ތިމަންކަލޭގެފާނުގެ ކާބަފައިކަލުނަވި إبراهيم ގެފާނާއި، إسحاق ގެފާނާއި، يعقوب ގެފާނުގެ دين އަށް ތިމަންކަލޭގެފާނު تبع ވެވަޑައިގަތީމެވެ. اللَّه އަށް އެއްވެސް އެއްޗެއް شريك ކުރުމަކީ ތިމަންމެންނަށް އޮތްކަމެއްނޫނެވެ. އެއީ ތިމަންމެންނަށާއި، މީސްތަކުންނަށް اللَّه ދެއްވި فضل ވަންތަކަމުގެ ތެރެއިންވާ ކަމެކެވެ. އެހެނެއްކަމަކު، މީސްތަކުންކުރެ ގިނަމީހުން (އެކަލާނގެއަށް) شكر އެއް ނުކުރެތެވެ.
يَا صَاحِبَيِ السِّجْنِ أَأَرْبَابٌ مُّتَفَرِّقُونَ خَيْرٌ أَمِ اللَّهُ الْوَاحِدُ الْقَهَّارُ ( 39 )
އޭ ޖަލުގެ ދެ އެކުވެރިންނޭވެ! ވަކިވަކިވެގެންވާ ކަލާނގެ ތަކެއްހޭ ރަނގަޅީ، ނުވަތަ އެއްކައުވަންތަ، ބާރުވެރިވަންތަ اللَّه ހެއްޔެވެ؟
مَا تَعْبُدُونَ مِن دُونِهِ إِلَّا أَسْمَاءً سَمَّيْتُمُوهَا أَنتُمْ وَآبَاؤُكُم مَّا أَنزَلَ اللَّهُ بِهَا مِن سُلْطَانٍ ۚ إِنِ الْحُكْمُ إِلَّا لِلَّهِ ۚ أَمَرَ أَلَّا تَعْبُدُوا إِلَّا إِيَّاهُ ۚ ذَٰلِكَ الدِّينُ الْقَيِّمُ وَلَٰكِنَّ أَكْثَرَ النَّاسِ لَا يَعْلَمُونَ ( 40 )
ތިޔަބައިމީހުންނާއި، ތިޔަބައިމީހުންގެ ކާބަފައިން އެ އެއްޗިއްސަށް ނަންދީފައިވާ އެއްޗެހިތަކަކަށްމެނުވީ، اللَّه ފިޔަވައި ތިޔަބައިމީހުން އަޅުކަމެއް ނުކުރަމުއެވެ. އެ އެއްޗަކަށް (އަޅުކަންކުރުމަށް) އެއްވެސް حجّة އެއް اللَّه ބާވައެއްނުލައްވަތެވެ. اللَّه އަށް މެނުވީ حكم ކުރެއްވުމެއްނެތެވެ. އެކަލާނގެއަށް ފިޔަވައި ތިޔަބައިމީހުން އަޅުކަން ނުކުރުމަށް އެކަލާނގެ ވަނީ أمر ކުރައްވާފައެވެ. އެއީ حق ތެދު دين އެވެ. އެހެނެއްކަމަކު، މީސްތަކުންކުރެ ގިނަމީހުން (އެކަމެއް) ނުދަނެތެވެ.
يَا صَاحِبَيِ السِّجْنِ أَمَّا أَحَدُكُمَا فَيَسْقِي رَبَّهُ خَمْرًا ۖ وَأَمَّا الْآخَرُ فَيُصْلَبُ فَتَأْكُلُ الطَّيْرُ مِن رَّأْسِهِ ۚ قُضِيَ الْأَمْرُ الَّذِي فِيهِ تَسْتَفْتِيَانِ ( 41 )
އޭ ޖަލުގެ ދެ އެކުވެރިންނޭވެ! ތިޔަ ދެމީހުންކުރެ އެކަކު ދަންނާށެވެ! ފަހެ، އޭނާގެ ސާހިބުމީހާއަށް އޭނާ ރާ ދޭނެތެވެ. އަދި އަނެކަކު ދަންނާށެވެ! ފަހެ، އޭނާ صليب އަށް އެރުވޭނެތެވެ. ދެން އޭނާގެ ބޮލުން ދޫނި ކާނެތެވެ. ތިޔަ ދެމީހުން އޮޅުންފިލުވަން އުޅޭކަންތައްވަނީ (އެގޮތަށް) ނިޔާކުރައްވާފައެވެ.
وَقَالَ لِلَّذِي ظَنَّ أَنَّهُ نَاجٍ مِّنْهُمَا اذْكُرْنِي عِندَ رَبِّكَ فَأَنسَاهُ الشَّيْطَانُ ذِكْرَ رَبِّهِ فَلَبِثَ فِي السِّجْنِ بِضْعَ سِنِينَ ( 42 )
އެ ދެމީހުންކުރެ ނަޖާވާނެކަމަށް ހީކުރެއްވި މީހާއަށް އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިމަންކަލޭގެފާނުގެ ވާހަކަ، ތިބާގެ ސާހިބުމީހާގެ ގާތުގައި ކިޔައިދޭށެވެ! ފަހެ، އޭނާގެ ސާހިބުމީހާގެ ގާތުގައި އެވާހަކަދައްކަން شيطان އާ އޭނާ ހަނދާން ނެއްތިއެވެ. ފަހެ (އެކަމުގެ ސަބަބުން) އެކަލޭގެފާނު ޖަލުގައި (އިތުރު) އެތައްއަހަރަކު ހުންނެވިއެވެ.
وَقَالَ الْمَلِكُ إِنِّي أَرَىٰ سَبْعَ بَقَرَاتٍ سِمَانٍ يَأْكُلُهُنَّ سَبْعٌ عِجَافٌ وَسَبْعَ سُنبُلَاتٍ خُضْرٍ وَأُخَرَ يَابِسَاتٍ ۖ يَا أَيُّهَا الْمَلَأُ أَفْتُونِي فِي رُؤْيَايَ إِن كُنتُمْ لِلرُّؤْيَا تَعْبُرُونَ ( 43 )
ރަސްގެފާނު ވިދާޅުވިއެވެ. ހަމަކަށަވަރުން، ތިމަންނާއަށް (ހުވަފެނުގައި) ފަލަ ހަތް ގެރި ފެނެއެވެ. (ފެންނަނީ) ހިކި ހަތް ގެރި، އެ ގެރިތައް ކަނިކޮންނެވެ. އަދި ރޯ ހަތް ކުރާއާއި، ހިކި ހަތް ކުރާ ފެނެއެވެ. އޭ ބޮޑުންނޭވެ! ތިޔަބައިމީހުން ހުވަފެނުގެ މާނަ ކިޔައިދޭ ބަޔަކު ކަމުގައިވާނަމަ، އަހުރެންގެ މި ހުވަފެނާ މެދުގައި، އަހުރެންނަށް އޮޅުންފިލުވައިދޭށެވެ!
قَالُوا أَضْغَاثُ أَحْلَامٍ ۖ وَمَا نَحْنُ بِتَأْوِيلِ الْأَحْلَامِ بِعَالِمِينَ ( 44 )
އެއުރެން ދެންނެވޫއެވެ. (ތިޔައީ) ހުވަފެނުގެ ހާސަރެކެވެ. އަދި (އެކަހަލަ) ހުވަފެން މާނަކުރަން ދަންނަ ބަޔަކު ކަމުގައި ތިމަންމެން ނުވަމުއެވެ.
وَقَالَ الَّذِي نَجَا مِنْهُمَا وَادَّكَرَ بَعْدَ أُمَّةٍ أَنَا أُنَبِّئُكُم بِتَأْوِيلِهِ فَأَرْسِلُونِ ( 45 )
އަދި އެ ދެމީހުންކުރެ (ޖަލުން) ސަލާމަތްވީމީހާ އެތައްދުވަހެއް ފަހުން (يوسف ގެފާނު މަތިިން) ހަނދާންވެގެން ބުންޏެވެ. އެ ހުވަފެނުގެ މާނަ ތިޔަބައިމީހުންނަށް ތިމަން ކިޔައިދީފާހުށީމެވެ. ފަހެ، ތިޔަބައިމީހުން (يوسف ގެފާނުގެ އަރިހަށް) ތިމަން ފޮނުވާށެވެ!
يُوسُفُ أَيُّهَا الصِّدِّيقُ أَفْتِنَا فِي سَبْعِ بَقَرَاتٍ سِمَانٍ يَأْكُلُهُنَّ سَبْعٌ عِجَافٌ وَسَبْعِ سُنبُلَاتٍ خُضْرٍ وَأُخَرَ يَابِسَاتٍ لَّعَلِّي أَرْجِعُ إِلَى النَّاسِ لَعَلَّهُمْ يَعْلَمُونَ ( 46 )
(އޭނާ ދެންނެވިއެވެ.) އޭ يوسف ގެފާނެވެ! އޭ ތެދުވެރިޔާއެވެ! ހިކި ހަތް ގެރި (ވެގެން) ކާ، ފަލަ ހަތް ގެރިއާއި، ރޯ ހަތް ކުރާއާއި، ހިކި ހަތް ކުރާއާމެދު (އެބަހީ: އެ ހުވަފެނާމެދު) އަހުރެމެންނަށް އޮޅުންފިލުއްވައި ދެއްވާށެވެ! (އެއީ) ތިމަން އެއުރެން ގާތަށް އެނބުރިގޮސް ކިޔައިދިނުމަށްޓަކައެވެ. (އެއީ) އެއުރެންނަށް (އެކަން) އެނގުމަށްޓަކައެވެ.
قَالَ تَزْرَعُونَ سَبْعَ سِنِينَ دَأَبًا فَمَا حَصَدتُّمْ فَذَرُوهُ فِي سُنبُلِهِ إِلَّا قَلِيلًا مِّمَّا تَأْكُلُونَ ( 47 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ވިދިވިދިގެން ހަތް އަހަރުވަންދެން ތިޔަބައިމީހުން ގޮވާންކުރާ ހުށިކަމެވެ! އަދި ތިޔަބައިމީހުން ކަނޑާ ގޮވާމެއް، ކުރާގަނޑުގައި ބަހައްޓާށެވެ! ކެއުމަށް ބޭނުންކުރާ މަދު އެތިކޮޅެއް މެނުވީއެވެ.
ثُمَّ يَأْتِي مِن بَعْدِ ذَٰلِكَ سَبْعٌ شِدَادٌ يَأْكُلْنَ مَا قَدَّمْتُمْ لَهُنَّ إِلَّا قَلِيلًا مِّمَّا تُحْصِنُونَ ( 48 )
ދެން އެއަށްފަހު، ތަދުމަޑު ހަތް އަހަރު އަންނާނެތެވެ. ތިޔަބައިމީހުން ރައްކާކުރި ތަކެތިން މަދުއެތިކޮޅެއް މެނުވީ، އެ އަހަރުތަކަށް ބަހައްޓާފައިހުރި ތަކެތި ތިޔަބައިމީހުން ކައިހުސްކުރާނޫއެވެ.
ثُمَّ يَأْتِي مِن بَعْدِ ذَٰلِكَ عَامٌ فِيهِ يُغَاثُ النَّاسُ وَفِيهِ يَعْصِرُونَ ( 49 )
ދެން އެއަށްފަހު، މީސްތަކުންނަށް ވާރޭފެން ގިނައިން ލިބޭނޭ އަހަރެއް އަންނާނެތެވެ. އަދި އެ އަހަރުގައި އެއުރެން (ދިޔަފެލާ ތަކެތިން) ދިޔަފެލާނެތެވެ.
وَقَالَ الْمَلِكُ ائْتُونِي بِهِ ۖ فَلَمَّا جَاءَهُ الرَّسُولُ قَالَ ارْجِعْ إِلَىٰ رَبِّكَ فَاسْأَلْهُ مَا بَالُ النِّسْوَةِ اللَّاتِي قَطَّعْنَ أَيْدِيَهُنَّ ۚ إِنَّ رَبِّي بِكَيْدِهِنَّ عَلِيمٌ ( 50 )
ރަސްގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުން ތިމަންގެ ގާތަށް އޭނާ ގެންނާށެވެ! ފަހެ، ރަސްގެފާނު ފޮނުއްވި މީހާ އެކަލޭގެފާނުގެ އަރިހަށް އައުމުން، ވިދާޅުވިއެވެ. ތިބާގެ ސާހިބުމީހާގެ ގާތަށް ތިބާ އެނބުރި ހިނގައިދާށެވެ! ފަހެ، އެއުރެންގެ އަތްތައް ކޮށައިލި އަންހެނުންތަކަށް ވީގޮތަކާމެދު، އޭނާއާ ސުވާލުކޮށްބަލާށެވެ! ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ އެއުރެންގެ މަކަރުވެރިކަން މޮޅަށް ދެނެވޮޑިގެންވެތެވެ.
قَالَ مَا خَطْبُكُنَّ إِذْ رَاوَدتُّنَّ يُوسُفَ عَن نَّفْسِهِ ۚ قُلْنَ حَاشَ لِلَّهِ مَا عَلِمْنَا عَلَيْهِ مِن سُوءٍ ۚ قَالَتِ امْرَأَتُ الْعَزِيزِ الْآنَ حَصْحَصَ الْحَقُّ أَنَا رَاوَدتُّهُ عَن نَّفْسِهِ وَإِنَّهُ لَمِنَ الصَّادِقِينَ ( 51 )
އޭނާ (އެބަހީ: ރަސްގެފާނު) ވިދާޅުވިއެވެ. ތިޔަކަނބަލުން يوسف ގެފާނުގެ ކިބައިން ނުބައިކަމަކަށް އެދުނުހިނދު، ކަންވީގޮތަކީ ކޮބައިހެއްޔެވެ؟ އެއުރެން ދެންނެވޫއެވެ. حاش للَّه އެވެ! (اللَّه ގެ ހުސްطاهر ވަންތަކަމާއެވެ!) އެކަލޭގެފާނުގެ ކިބައިގައި އެއްވެސް ނުބައިކަމެއްވާ ކަމަކަށް ތިމަންމެން ދެނެއެއްނެތީމުއެވެ. العزيز ގެ އަންހެނުން ބުންޏެވެ. މިހާރު حق ތެދު ފާޅުވެއްޖެއެވެ. އެކަލޭގެފާނުގެ ކިބައިން ނުބައިކަމަކަށް އެދުނީ، ތިމަންނާއެވެ. އަދި ހަމަކަށަވަރުން، އެކަލޭގެފާނު ވަނީ، ތެދުވެރިންގެ ތެރެއިންނެވެ.
ذَٰلِكَ لِيَعْلَمَ أَنِّي لَمْ أَخُنْهُ بِالْغَيْبِ وَأَنَّ اللَّهَ لَا يَهْدِي كَيْدَ الْخَائِنِينَ ( 52 )
(يوسف ގެފާނު ވިދާޅުވިއެވެ.) އެއީ ތިމަން غيب ގައި އޭނާއަށް (އެބަހީ: العزيز އަށް) خيانة ތެރި ނުވާކަމާއި، އަދި خيانة ތެރިންގެ މަކަރުވެރި ރޭވުން اللَّه ފަހިކޮށް ނުދެއްވާނެކަން، އޭނާއަށް އެނގުމަށްޓަކައެވެ.
وَمَا أُبَرِّئُ نَفْسِي ۚ إِنَّ النَّفْسَ لَأَمَّارَةٌ بِالسُّوءِ إِلَّا مَا رَحِمَ رَبِّي ۚ إِنَّ رَبِّي غَفُورٌ رَّحِيمٌ ( 53 )
އަދި ތިމަންކަލޭގެފާނުގެ نفس ވެސް (ތުހުމަތުން) ބަރީއައެއް ނުކުރައްވަމެވެ. ހަމަކަށަވަރުން، نفس އަކީ، ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ رحمة ލައްވައިފިނަމަ، މެނުވީ، ނުބައިކަންތަކަށް އަބަދުވެސް އަމުރުކުރާ އެއްޗެއްކަން ކަށަވަރެވެ. ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެއީ، ގިނަގިނައިން ފާފަފުއްސަވާ، رحيم ވަންތަ ރަސްކަލާނގެއެވެ.
وَقَالَ الْمَلِكُ ائْتُونِي بِهِ أَسْتَخْلِصْهُ لِنَفْسِي ۖ فَلَمَّا كَلَّمَهُ قَالَ إِنَّكَ الْيَوْمَ لَدَيْنَا مَكِينٌ أَمِينٌ ( 54 )
ރަސްގެފާނު ވިދާޅުވިއެވެ. އެކަލޭގެފާނު ތިމަންގެ خدمة އަށް خاص ކުރުމަށްޓަކައި، އެކަލޭގެފާނު ތިމަންގެ ގާތަށް ގެންނާށެވެ! ފަހެ، އެކަލޭގެފާނާ ވާހަކަ ދައްކަވާފައި، އެ ރަސްގެފާނު ވިދާޅުވިއެވެ. ހަމަކަށަވަރުން، ކަލޭގެފާނީ، މިއަދު ތިމަންގެ ގާތުގައި އިތުބާރުކުރެވޭ أمانة ތެރިޔަކީމުއެވެ.
قَالَ اجْعَلْنِي عَلَىٰ خَزَائِنِ الْأَرْضِ ۖ إِنِّي حَفِيظٌ عَلِيمٌ ( 55 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. (مصر ގެ) ބިމުގެ خزانة ތަކުގެ ވެރިޔަކަށް ތިމަން ލައްވާށެވެ! ހަމަކަށަވަރުން، ތިމަންނާއީ (އެއަށް) ރައްކާތެރިވާ (އެއާބެހޭ ކަންތައްތައް) ރަނގަޅަށް ދަންނަ މީހަކީމެވެ.
وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِي الْأَرْضِ يَتَبَوَّأُ مِنْهَا حَيْثُ يَشَاءُ ۚ نُصِيبُ بِرَحْمَتِنَا مَن نَّشَاءُ ۖ وَلَا نُضِيعُ أَجْرَ الْمُحْسِنِينَ ( 56 )
އެފަދައިން އެކަލޭގެފާނު ބޭނުންފުޅު ތަނެއްގައި ހުންނެވޭނެގޮތުން ތިމަންރަސްކަލާނގެ އެކަލޭގެފާނު (مصر ގެ) ބިމުގައި ބާރުވެރި ކުރެއްވީމެވެ. ތިމަންރަސްކަލާނގެ އިރާދަކުރައްވާ ބަޔަކަށް، ތިމަންރަސްކަލާނގެ رحمة ލައްވަމެވެ. އަދި إحسان ތެރިންގެ ދަރުމަ، ތިމަންރަސްކަލާނގެ ގެއްލެނި ނުކުރައްވާހުށީމެވެ.
وَلَأَجْرُ الْآخِرَةِ خَيْرٌ لِّلَّذِينَ آمَنُوا وَكَانُوا يَتَّقُونَ ( 57 )
އަދި ހަމަކަށަަވަރުން، إيمان ވެ، تقوى ވެރިވެއުޅުނު މީހުންނަށް، آخرة ގެ ދަރުމައިގެ ހެޔޮކަން މާބޮޑެވެ.
وَجَاءَ إِخْوَةُ يُوسُفَ فَدَخَلُوا عَلَيْهِ فَعَرَفَهُمْ وَهُمْ لَهُ مُنكِرُونَ ( 58 )
އަދި يوسف ގެފާނުގެ أخ ން އަތުވެއްޖައޫއެވެ. ދެން އެކަލޭގެފާނުގެ އަރިހަށް އެއުރެން ވަދެއްޖައޫއެވެ. އަދި އެކަލޭގެފާނަށް އެއުރެން އެނގިވަޑައިގަތެވެ. ފަހެ، އެއުރެންނަކަށް އެކަލޭގެފާނަކީ ކާކުކަމެއް ނޭނގެތެވެ.
وَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ قَالَ ائْتُونِي بِأَخٍ لَّكُم مِّنْ أَبِيكُمْ ۚ أَلَا تَرَوْنَ أَنِّي أُوفِي الْكَيْلَ وَأَنَا خَيْرُ الْمُنزِلِينَ ( 59 )
އެއުރެން ބޭނުންވެގެން އައި ސާމާނު ތައްޔާރުކޮށްދެއްވިހިނދު، އެކަލޭގެފާނު ވިދާޅުވިއެވެ. (ތިޔަބައިމީހުން ދޫކޮށްފައި އައި) ތިޔަބައިމީހުންނާ އެއްބަފާ أخ އެއް ތިމަންކަލޭގެފާނުގެ އަރިހަށް ގެންނާށެވެ! ތިމަންކަލޭގެފާނު ހަމަޔަށް މިނެދެއްވާކަމާއި، ތިމަންކަލޭގެފާނީ، އެންމެހެޔޮގޮތުގައި މެހުމާނުންނަށް ކަމޭހިތާ މީހަކުކަން ތިޔަބައިމީހުން ނުދެކެމުހެއްޔެވެ؟
فَإِن لَّمْ تَأْتُونِي بِهِ فَلَا كَيْلَ لَكُمْ عِندِي وَلَا تَقْرَبُونِ ( 60 )
ފަހެ، ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނުގެ އަރިހަށް އޭނާ ނުގެނައިނަމަ، ތިމަންކަލޭގެފާނުގެ އަރިހުން ތިޔަބައިމީހުންނަކަށް މިނެދިނުމެއް ނެތެވެ. އަދި ތިޔަބައިމީހުން، ތިމަންކަލޭގެފާނުގެ އަރިހަށް ނާންނާށެވެ!
قَالُوا سَنُرَاوِدُ عَنْهُ أَبَاهُ وَإِنَّا لَفَاعِلُونَ ( 61 )
އެއުރެން ދެންނެވޫއެވެ. އޭނާގެ ބައްޕާ އަރިހުން އޭނާ ގެނެވޭތޯ ތިމަންމެން ގަދަމަސައްކަތް ކުރާނަމެވެ. އަދި ހަމަކަށަވަރުން، ތިމަންމެން (އެކަން) ކުރާހުށީމުއެވެ.
وَقَالَ لِفِتْيَانِهِ اجْعَلُوا بِضَاعَتَهُمْ فِي رِحَالِهِمْ لَعَلَّهُمْ يَعْرِفُونَهَا إِذَا انقَلَبُوا إِلَىٰ أَهْلِهِمْ لَعَلَّهُمْ يَرْجِعُونَ ( 62 )
އެކަލޭގެފާނު ގެންގުޅުއްވާ ކުދިންނަށް ވިދާޅުވިއެވެ. އެއުރެން (އަގަކަށް) ގެނައި މުދާ، އެއުރެންގެ ތަކެތީގެ ތެރެޔަށްލާށެވެ! (އެއީ) އެއުރެންގެ أهل ންގެ ގާތަށް އެނބުރިދާހިނދު، އެކަން އެއުރެންނަށް އެނގުމަށްޓަކައެވެ. (އަދި ތިމަންކަލޭގެފާނުގެ އަރިހަށް) އެއުރެން އެނބުރި އާދޭތޯއެވެ.
فَلَمَّا رَجَعُوا إِلَىٰ أَبِيهِمْ قَالُوا يَا أَبَانَا مُنِعَ مِنَّا الْكَيْلُ فَأَرْسِلْ مَعَنَا أَخَانَا نَكْتَلْ وَإِنَّا لَهُ لَحَافِظُونَ ( 63 )
ފަހެ، އެއުރެންގެ ބައްޕާގެ އަރިހަށް އެއުރެން އެނބުރި ދިޔަހިނދު އެއުރެން ދެންނެވޫއެވެ. އޭ އަހުރެމެންގެ ބައްޕާއެވެ! (ތިމަންމެންގެ ކޮއްކޮ ތިމަންމެންނާއެކު ނުފޮނުއްވައިފިނަމަ، ދެން) ތިމަންމެންނަށް މިނެދިނުން މަނާވީއެވެ. ފަހެ، ތިމަންމެންނަށް މިނެދޭތޯ ތިމަންމެންނާއެކު ތިމަންމެންގެ ކޮއްކޮ ފޮނުއްވާށެވެ! އަދި ހަމަކަށަވަރުން، ތިމަންމެން އެކަލޭގެފާނު ބަލަހައްޓައި ރައްކާތެރިކުރާ ހުށީމުއެވެ.
قَالَ هَلْ آمَنُكُمْ عَلَيْهِ إِلَّا كَمَا أَمِنتُكُمْ عَلَىٰ أَخِيهِ مِن قَبْلُ ۖ فَاللَّهُ خَيْرٌ حَافِظًا ۖ وَهُوَ أَرْحَمُ الرَّاحِمِينَ ( 64 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. މީގެކުރިން އޭނާގެ أخ (يوسف ގެފާނު) އާމެދު ތިޔަބައިމީހުންނަށް އިތުބާރުކުރި ފަދައިން މެނުވީ، އޭނާއާމެދު ތިޔަބައިމީހުންގެ މައްޗަށް ތިމަން އިތުބާރުކުރާނޭ ހެއްޔެވެ؟ ފަހެ، اللَّه އީ، ރައްކާތެރިކުރައްވާ، އެންމެ ހެޔޮ ރަސްކަލާނގެއެވެ. އަދި އެކަލާނގެއީ رحمة ލައްވާ، އެންމެ رحمة ވަންތަ ރަސްކަލާނގެއެވެ.
وَلَمَّا فَتَحُوا مَتَاعَهُمْ وَجَدُوا بِضَاعَتَهُمْ رُدَّتْ إِلَيْهِمْ ۖ قَالُوا يَا أَبَانَا مَا نَبْغِي ۖ هَٰذِهِ بِضَاعَتُنَا رُدَّتْ إِلَيْنَا ۖ وَنَمِيرُ أَهْلَنَا وَنَحْفَظُ أَخَانَا وَنَزْدَادُ كَيْلَ بَعِيرٍ ۖ ذَٰلِكَ كَيْلٌ يَسِيرٌ ( 65 )
އަދި އެއުރެންގެ ސާމާނު އެއުރެން ހުޅުވިހިނދު، އެއުރެންގެ މުދާ (އޭގެތެރޭގައި) އެއުރެންނަށް ރައްދުކުރެވިފައިވަނިކޮށް ފެނިފައި، އެއުރެން ދެންނެވޫއެވެ. އޭ އަހުރެމެންގެ ބައްޕާއެވެ! (މިއަށްވުރެމޮޅަށް) ތިމަންމެން އެދޭނީ ކޮންކަމެއްހެއްޔެވެ؟ މިހިރީ ތިމަންމެންގެ މުދާ ތިމަންމެންނަށް ރައްދުކުރެވިފައެވެ. އަދި ތިމަންމެންގެ أهل ންނަށް (އިތުރަށް) ކާބޯތަކެތި ގެނެސްދީ، އަދި ތިމަންމެންގެ ކޮއްކޮ ރައްކާތެރި ކުރާހުށީމުއެވެ. އަދި جمل އަކަށް އެރުވޭތަކެތި އިތުރުކުރާނޫއެވެ. އެއީ މިނެދެވޭ މަދު އެތިކޮޅެކެވެ.
قَالَ لَنْ أُرْسِلَهُ مَعَكُمْ حَتَّىٰ تُؤْتُونِ مَوْثِقًا مِّنَ اللَّهِ لَتَأْتُنَّنِي بِهِ إِلَّا أَن يُحَاطَ بِكُمْ ۖ فَلَمَّا آتَوْهُ مَوْثِقَهُمْ قَالَ اللَّهُ عَلَىٰ مَا نَقُولُ وَكِيلٌ ( 66 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންގެ މައްޗަށް عداوة ތެރިންގެ ބަޔަކު ގަދަވެއްޖެހިނދަކު މެނުވީ، އޭނާ އަނބުރާ ތިމަންކަލޭގެފާނުގެ އަރިހަށް ގެނެސްދޭންކަމަށް، اللَّه ގެ حضرة ންވާ عهد އެއް ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނަށް ގެނެސްދީފުމަށް ދާނދެން ތިޔަބައިމީހުންނާއެކު އޭނާއެއް ނުފޮނުއްވާ ހުށީމެވެ. ފަހެ، އެއުރެންގެ ކަށަވަރުވެގެންވާ عهد ގެ ބަސް ގެނެސްގަތްހިނދު، އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިމަންމެން މިބުނާ އެއްޗަކަށް، اللَّه ހެކިވެވޮޑިގެންވާނެތެވެ.
وَقَالَ يَا بَنِيَّ لَا تَدْخُلُوا مِن بَابٍ وَاحِدٍ وَادْخُلُوا مِنْ أَبْوَابٍ مُّتَفَرِّقَةٍ ۖ وَمَا أُغْنِي عَنكُم مِّنَ اللَّهِ مِن شَيْءٍ ۖ إِنِ الْحُكْمُ إِلَّا لِلَّهِ ۖ عَلَيْهِ تَوَكَّلْتُ ۖ وَعَلَيْهِ فَلْيَتَوَكَّلِ الْمُتَوَكِّلُونَ ( 67 )
އަދި އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އޭ އަހުރެންގެ ދަރިންނޭވެ! ތިޔަބައިމީހުން (އެތަނަށްގޮސް) އެއްދޮރޯއްޓަކުން ނުވަންނާށެވެ! އަދި ތަފާތު ދޮރޯށިތަކަކުން ވަންނާށެވެ! އަދި اللَّه ގެ حضرة ން ތިޔަބައިމީހުންނަށް އެއްވެސްކަމެއް ތިމަންނާއަކަށް ފުއްދައެއް ނުދެވޭނެތެވެ. اللَّه އަށް މެނުވީ حكم ކުރެއްވުމެއް ނެތެވެ. ތިމަންކަލޭގެފާނު وكيل ކުރައްވަނީ، އެކަލާނގެއަށެވެ. އަދި وكيل ކުރާމީހުން ހަމައެކަނި އެކަލާނގެއަށް وكيل ކުރާހުށިކަމެވެ!
وَلَمَّا دَخَلُوا مِنْ حَيْثُ أَمَرَهُمْ أَبُوهُم مَّا كَانَ يُغْنِي عَنْهُم مِّنَ اللَّهِ مِن شَيْءٍ إِلَّا حَاجَةً فِي نَفْسِ يَعْقُوبَ قَضَاهَا ۚ وَإِنَّهُ لَذُو عِلْمٍ لِّمَا عَلَّمْنَاهُ وَلَٰكِنَّ أَكْثَرَ النَّاسِ لَا يَعْلَمُونَ ( 68 )
އަދި އެއުރެންގެ ބައްޕާ أمر ކުރެއްވިގޮތަށް، އެއުރެންގޮސް ވަންހިނދު އެކަލޭގެފާނަކަށް اللَّه ގެ حضرة ން އެއްވެސްކަމެއް އެއުރެންނަކަށް ފުއްދައެއް ނުދެއްވުނެވެ. ހަމައެކަނި، يعقوب ގެފާނުގެ ހިތްޕުޅުގައިއޮތް ބޭނުމެއް މެނުވީ ނުފުދެތެވެ. (އެއީ يعقوب ގެފާނު ވިދާޅުވިގޮތަށް އެއުރެން އެކަން ކުރުމެވެ.) އަދި ހަމަކަށަވަރުން، އެކަލޭގެފާނީ، ތިމަންރަސްކަލާނގެ އެކަލޭގެފާނަށް އުނގަންނައިދެއްވި އެއްޗެއް ރަނގަޅަށް ދެނެހުންނެވި ބޭކަލެކެވެ. އެހެނެއްކަމަކު، މީސްތަކުންކުރެ ގިނަމީހުން (އެކަމެއް) ނުދަނެތެވެ.
وَلَمَّا دَخَلُوا عَلَىٰ يُوسُفَ آوَىٰ إِلَيْهِ أَخَاهُ ۖ قَالَ إِنِّي أَنَا أَخُوكَ فَلَا تَبْتَئِسْ بِمَا كَانُوا يَعْمَلُونَ ( 69 )
އަދި يوسف ގެފާނުގެ އަރިހަށް އެއުރެން ވަނުމުން، އެކަލޭގެފާނުގެ أخ އެކަލޭގެފާނުގެ އަރިހަށް ގެންދަވައި ވިދާޅުވިއެވެ. ހަމަކަށަވަރުން، ތިމަންނާއީ، ތިބާގެ أخ އާއެވެ. ފަހެ، އެއުރެން ކޮށްއުޅުނު ކަންތަކާމެދު ތިބާ ހިތާމަނުކުރާށެވެ!
فَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ جَعَلَ السِّقَايَةَ فِي رَحْلِ أَخِيهِ ثُمَّ أَذَّنَ مُؤَذِّنٌ أَيَّتُهَا الْعِيرُ إِنَّكُمْ لَسَارِقُونَ ( 70 )
ފަހެ، އެކަލޭގެފާނު އެއުރެންގެ ސާމާނު ތައްޔާރުކޮށްދެއްވިހިނދު (މިނެހެދުމަށް ގެންގުޅޭ) سقاية، އޭނާގެ أخ ގެ ތަކެތީގެ ތެރެއަށް ލެއްވިއެވެ. ދެން ގޮވައި އަންގާ މީހަކު ގޮވިއެވެ. އޭ قافلة ގެ މީހުންނޭވެ! ހަމަކަށަވަރުން، ތިޔަބައިމީހުންނީ، ވައްކަންކުރާ ބަޔެކެވެ.
قَالُوا وَأَقْبَلُوا عَلَيْهِم مَّاذَا تَفْقِدُونَ ( 71 )
އެއުރެން އެއުރެންނާ ކުރިމަތިލައި އައިސް ބުނެފޫއެވެ. ތިޔަބައިމީހުން ގެއްލިގެންތިޔަ އުޅެނީ ކޮންއެއްޗެއް ހެއްޔެވެ؟
قَالُوا نَفْقِدُ صُوَاعَ الْمَلِكِ وَلِمَن جَاءَ بِهِ حِمْلُ بَعِيرٍ وَأَنَا بِهِ زَعِيمٌ ( 72 )
އެއުރެން ބުނެފޫއެވެ. ތިމަންމެން ގެއްލިގެން މިއުޅެނީ، ރަސްގެފާނުގެ ނާޅިއެވެ. އަދި އެ ގެނެސްދިން މީހަކަށް އޮތް އިނާމަކީ، جمل އަކަށް އެރުވޭހާ ތަކެއްޗެވެ. އެކަމަށް އަހަރެން ޖާމިނުވެގެންވަމެވެ.
قَالُوا تَاللَّهِ لَقَدْ عَلِمْتُم مَّا جِئْنَا لِنُفْسِدَ فِي الْأَرْضِ وَمَا كُنَّا سَارِقِينَ ( 73 )
އެއުރެން ދެންނެވޫއެވެ. اللَّه ގަންދީ ބުނަމުއެވެ. ތިމަންމެންއައީ ބިމުގައި فساد ކުރުމަށް ނޫންކަން ތިޔަބައިމީހުންނަށް އެނގިފައިވާނެކަން ކަށަވަރެވެ. އަދި ވައްކަންކުރާ ބަޔަކު ކަމުގައި ތިމަންމެން ނުވަމުއެވެ.
قَالُوا فَمَا جَزَاؤُهُ إِن كُنتُمْ كَاذِبِينَ ( 74 )
އެއުރެން ބުނެފޫއެވެ. ތިޔަބައިމީހުން ދޮގުހަދާކަމަށްވާނަމަ، ފަހެ، އެކަމުގެ (އެބަހީ: މިނާއެތި ވަގަށްނެގުމުގެ) ޖަޒާއަކީ ކޮބައިހެއްޔެވެ؟
قَالُوا جَزَاؤُهُ مَن وُجِدَ فِي رَحْلِهِ فَهُوَ جَزَاؤُهُ ۚ كَذَٰلِكَ نَجْزِي الظَّالِمِينَ ( 75 )
އެއުރެން ބުނެފޫއެވެ. އެކަމުގެ ޖަޒާއަކީ، އެމީހެއްގެ ތަކެތީގެ ތެރެއިން އެ މިނާއެތި ފެނުނު މީހެކެވެ. ފަހެ، އޭނާ އެއީ އެކަމުގެ ޖަޒާއެވެ. (އެބަހީ: ޖަޒާދިނުމަށްޓަކައި އެމީހަކު ހިފެހެއްޓުމެވެ.) އަނިޔާވެރިންނަށް ތިމަންރަސްކަލާނގެ ޖަޒާދެއްވަނީ، އެފަދައިންނެވެ.
فَبَدَأَ بِأَوْعِيَتِهِمْ قَبْلَ وِعَاءِ أَخِيهِ ثُمَّ اسْتَخْرَجَهَا مِن وِعَاءِ أَخِيهِ ۚ كَذَٰلِكَ كِدْنَا لِيُوسُفَ ۖ مَا كَانَ لِيَأْخُذَ أَخَاهُ فِي دِينِ الْمَلِكِ إِلَّا أَن يَشَاءَ اللَّهُ ۚ نَرْفَعُ دَرَجَاتٍ مَّن نَّشَاءُ ۗ وَفَوْقَ كُلِّ ذِي عِلْمٍ عَلِيمٌ ( 76 )
ފަހެ، އެކަލޭގެފާނުގެ أخ އާގެ ތަކެތި ހޯދުމުގެ ކުރިން، އެއުރެންގެ ތަކެތި ހޯދަންފެށިއެވެ. ދެން އެކަލޭގެފާނުގެ أخ އާގެ ތަކެތީގެ ތެރެއިން، އެ މިނާއެތި ނެރުނެވެ. އެގޮތަށް يوسف ގެފާނަށްޓަކައި، ތިމަންރަސްކަލާނގެ (އެކަންތައް) ރޭއްވެވީމެވެ. اللَّه އިރާދަކުރެއްވުމާނުލައި، ރަސްގެފާނުގެ އާދަޔާ އެއްގޮތަށް އެކަލޭގެފާނަކަށް، އޭނާގެ أخ އާ އަތަކު ނުލެއްވުނީހެވެ. ތިމަންރަސްކަލާނގެ އިރާދަކުރައްވާ ބަޔަކު، ދަރަޖަ މަތިވެރިކުރައްވަމެވެ. އަދި ކޮންމެ علٰٰم ވެރިޔެއްގެ މައްޗަށް (ހުރިހާކަމެއް) މޮޅަށް ދެނެވޮޑިގެންވާ ކަލާނގެ ވޮޑިގެންވެއެވެ.
قَالُوا إِن يَسْرِقْ فَقَدْ سَرَقَ أَخٌ لَّهُ مِن قَبْلُ ۚ فَأَسَرَّهَا يُوسُفُ فِي نَفْسِهِ وَلَمْ يُبْدِهَا لَهُمْ ۚ قَالَ أَنتُمْ شَرٌّ مَّكَانًا ۖ وَاللَّهُ أَعْلَمُ بِمَا تَصِفُونَ ( 77 )
އެއުރެން ބުނެފޫއެވެ. އޭނާ ވައްކަންކުރިކަމަށްވާނަމަ، މީގެކުރިން އޭނާގެ أخ އަކުވެސްވަނީ ވައްކަންކޮށްފައެވެ. ފަހެ، يوسف ގެފާނު، އެކަލޭގެފާނުގެ ހިތްޕުޅުގައި އެ ބަސް ސިއްރުކުރެއްވިއެވެ. އަދި އެއުރެންނަށް އެކަން ފާޅެއްނުކުރައްވަތެވެ. އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންގެ ދަރަޖަ އެއަށްވުރެ މާނުބައެވެ. އަދި ތިޔަބައިމީހުން ސިފަކޮށް ބުނާއެއްޗެއް اللَّه ދެނެވޮޑިގަތުން ބޮޑެވެ.
قَالُوا يَا أَيُّهَا الْعَزِيزُ إِنَّ لَهُ أَبًا شَيْخًا كَبِيرًا فَخُذْ أَحَدَنَا مَكَانَهُ ۖ إِنَّا نَرَاكَ مِنَ الْمُحْسِنِينَ ( 78 )
އެއުރެން ދެންނެވޫއެވެ. އޭ العزيز އެވެ! ހަމަކަށަވަރުން، އޭނާގެ قدر ވެރި މުސްކުޅި ބައްޕައެއްވެއެވެ. ފަހެ، ތިމަންމެންގެ ތެރެއިން އެކަކު އޭނާގެ ބަދަލުގައި ހިފަހައްޓަވާށެވެ! ހަމަކަށަވަރުން، ތިމަންމެން ކަލޭގެފާނު ދެކެނީ، إحسان ތެރިންގެ ތެރެއިންނެވެ.
قَالَ مَعَاذَ اللَّهِ أَن نَّأْخُذَ إِلَّا مَن وَجَدْنَا مَتَاعَنَا عِندَهُ إِنَّا إِذًا لَّظَالِمُونَ ( 79 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. [معاذ اللَّه] އެވެ. (އެބަހީ: اللَّه ގެ ރައްކާތެރިކަމަށް އެދެމެވެ.) އެމީހެއްގެ އަތުގައި ތިމަންމެންގެ ތަކެތި ހުއްޓައި ފެނުނު މީހަކު މެނުވީ، ތިމަންމެން ހިފެހެއްޓުމުން ދުރުވެ، ރައްކާތެރިވުން އެދެމެވެ. އެހެންވެއްޖެނަމަ، ހަމަކަށަވަރުން، ތިމަންމެންވާހުށީ އަނިޔާވެރިން ކަމުގައެވެ.
فَلَمَّا اسْتَيْأَسُوا مِنْهُ خَلَصُوا نَجِيًّا ۖ قَالَ كَبِيرُهُمْ أَلَمْ تَعْلَمُوا أَنَّ أَبَاكُمْ قَدْ أَخَذَ عَلَيْكُم مَّوْثِقًا مِّنَ اللَّهِ وَمِن قَبْلُ مَا فَرَّطتُمْ فِي يُوسُفَ ۖ فَلَنْ أَبْرَحَ الْأَرْضَ حَتَّىٰ يَأْذَنَ لِي أَبِي أَوْ يَحْكُمَ اللَّهُ لِي ۖ وَهُوَ خَيْرُ الْحَاكِمِينَ ( 80 )
ފަހެ، އެއުރެން އެކަލޭގެފާނުގެ ކިބައިން اميد ކެނޑުނުހިނދު، އެއުރެން އެކަހެރިވެ ސައްލާކުރޫއެވެ. އެއުރެންގެ ބޮޑަކަށްހުރި މީހާ ބުންޏެވެ. ތިޔަބައިމީހުންގެ މައްޗަށް ތިޔަބައިމީހުންގެ ބައްޕާ، اللَّه ގެ حضرة ން عهد ކަށަވަރު ހިއްޕަވާފައިވާކަމާއި، އޭގެކުރިން يوسف ގެފާނާމެދު ތިޔަބައިމީހުން ފަރުވާކުޑަކޮށްފައިވާކަން، ތިޔަބައިމީހުންނަކަށް ނޭނގޭހެއްޔެވެ؟ ފަހެ، ތިމަންގެ ބައްޕާ ތިމަންނައަށް إذن ދެއްވައިފުމަށްދާނދެން، ނުވަތަ اللَّه ތިމަންގެ މައްޗަށް ނިޔާކުރައްވައިފުމަށްދާނދެން ތިމަން (مصر ގެ) ބިން ދޫކޮށެއް ނުލާހުށީމެވެ. އަދި އެކަލާނގެއީ حكم ކުރައްވާ އެންމެ ހެޔޮ ރަސްކަލާނގެއެވެ.
ارْجِعُوا إِلَىٰ أَبِيكُمْ فَقُولُوا يَا أَبَانَا إِنَّ ابْنَكَ سَرَقَ وَمَا شَهِدْنَا إِلَّا بِمَا عَلِمْنَا وَمَا كُنَّا لِلْغَيْبِ حَافِظِينَ ( 81 )
ތިޔަބައިމީހުން އެނބުރި ތިޔަބައިމީހުންގެ ބައްޕާ އަރިހަށްގޮސް ދަންނަވާށެވެ! އޭ އަހުރެމެންގެ ބައްޕާއެވެ! ހަމަކަށަވަރުން، ކަލޭގެފާނުގެ ދަރިކަލުން ވައްކަންކޮށްފިއެވެ. އަދި ތިމަންމެންނަށް އެނގިފައިވާކަމަކަށް މެނުވީ، ތިމަންމެން ހެތްކެއް ނުވަމުއެވެ. އަދި غيب ގައިވާ ކަންތައް ދަންނަބަޔަކު ކަމުގައި ތިމަންމެން ނުވަމުއެވެ.
وَاسْأَلِ الْقَرْيَةَ الَّتِي كُنَّا فِيهَا وَالْعِيرَ الَّتِي أَقْبَلْنَا فِيهَا ۖ وَإِنَّا لَصَادِقُونَ ( 82 )
އަދި ތިމަންމެންގޮސް ތިބި ރަށުގެ މީހުންނާ، އަދި ތިމަންމެން އޭގައި ދަތުރުކޮށްފައި އައި قافلة ގެ މީހުންނާ سؤال ކުރައްވާށެވެ! އަދި ހަމަކަށަވަރުން، ތިމަންމެންނީ ތެދުބުނާ ބަޔަކީމުއެވެ.
قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرًا ۖ فَصَبْرٌ جَمِيلٌ ۖ عَسَى اللَّهُ أَن يَأْتِيَنِي بِهِمْ جَمِيعًا ۚ إِنَّهُ هُوَ الْعَلِيمُ الْحَكِيمُ ( 83 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އަދި ކިއެއްތަ! ތިޔަބައިމީހުންގެ نفس ތައް ތިޔަބައިމީހުންނަށް (ނުބައި) ކަމެއް ޒީނަތްތެރިކޮށްދިނީއެވެ. ފަހެ، ދެން އޮތީ ރިވެތިވެގެންވާ ކެތްކުރުމެކެވެ. اللَّه އެ އެންމެން ތިމަންކަލޭގެފާނުގެ އަރިހަށް ގެނެސްދެއްވައިފާށިއެވެ! ހަމަކަށަވަރުން، އެކަލާނގެއީ، މޮޅަށް ދެނެވޮޑިގެންވާ، حكيم ވަންތަ ރަސްކަލާނގެއެވެ.
وَتَوَلَّىٰ عَنْهُمْ وَقَالَ يَا أَسَفَىٰ عَلَىٰ يُوسُفَ وَابْيَضَّتْ عَيْنَاهُ مِنَ الْحُزْنِ فَهُوَ كَظِيمٌ ( 84 )
އަދި އެކަލޭގެފާނު އެއުރެންނާ އެއްކިބާވެ ވަޑައިގަތެވެ. އަދި ވިދާޅުވިއެވެ. يوسف ގެފާނުގެ މައްޗަށް އަހުރެން ކުރާ ހިތާމައެއްގެ ބޮޑުކަމާއެވެ! އަދި ހިތާމައިގެ ސަބަބުން، އެކަލޭގެފާނުގެ ލޯފުޅު ހުދުވެގެންދިޔައެވެ. ފަހެ، އެކަލޭގެފާނީ، ހިތާމަފާޅު ނުކުރައްވައި، ހިފަހައްޓަވާ ބޭކަލެކެވެ.
قَالُوا تَاللَّهِ تَفْتَأُ تَذْكُرُ يُوسُفَ حَتَّىٰ تَكُونَ حَرَضًا أَوْ تَكُونَ مِنَ الْهَالِكِينَ ( 85 )
އެއުުރެން ދެންނެވޫއެވެ. اللَّه ގަންދީ ހުވާކުރަމެވެ. ކަލޭގެފާނު ބަލިحال ބޮޑުވެއްޖެއުމަށްދާނދެން، ނުވަތަ މަރުވެފައިވާ މީހުންގެ ތެރެއިން ހިމެނިއްޖައުމަށްދާނދެން، يوسف ގެފާނުމަތިން ހަނދުމަކުރެއްވުމުގައި ކަލޭގެފާނު ފިލާވަޑައިނުގަންނަވަމުއެވެ.
قَالَ إِنَّمَا أَشْكُو بَثِّي وَحُزْنِي إِلَى اللَّهِ وَأَعْلَمُ مِنَ اللَّهِ مَا لَا تَعْلَمُونَ ( 86 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިމަންކަލޭގެފާނުގެ މޮޅިޔާއި، ހިތާމައިގެ شكوى ދަންނަވާކަން ކަށަވަރީ، ހަމައެކަނި اللَّه އަށެވެ. އަދި ތިޔަބައިމީހުންނަށް ނޭނގޭ ކަންތައްތަކެއް، اللَّه ގެ حضرة ން ތިމަންކަލޭގެފާނަށް އެނގިވަޑައިގަނެއެވެ.
يَا بَنِيَّ اذْهَبُوا فَتَحَسَّسُوا مِن يُوسُفَ وَأَخِيهِ وَلَا تَيْأَسُوا مِن رَّوْحِ اللَّهِ ۖ إِنَّهُ لَا يَيْأَسُ مِن رَّوْحِ اللَّهِ إِلَّا الْقَوْمُ الْكَافِرُونَ ( 87 )
އޭ އަހުރެންގެ ދަރިންނޭވެ! يوسف ގެފާނާއި، އޭނާގެ أخ އާގެ خبر އެއްވޭތޯ ތިޔަބައިމީހުން ދާށެވެ! އަދި اللَّه ގެ رحمة ން ތިޔަބައިމީހުން اميد ކަނޑައިނުލާށެވެ! (ހަަމަކަށަވަރުން، كافر ވިމީހުން މެނުވީ، اللَّه ގެ رحمة ން اميد ކަނޑައެއް ނުލާނެތެވެ.)
فَلَمَّا دَخَلُوا عَلَيْهِ قَالُوا يَا أَيُّهَا الْعَزِيزُ مَسَّنَا وَأَهْلَنَا الضُّرُّ وَجِئْنَا بِبِضَاعَةٍ مُّزْجَاةٍ فَأَوْفِ لَنَا الْكَيْلَ وَتَصَدَّقْ عَلَيْنَا ۖ إِنَّ اللَّهَ يَجْزِي الْمُتَصَدِّقِينَ ( 88 )
ފަހެ، އެއުރެން އެކަލޭގެފާނުގެ އަރިހަށް ވަންހިނދު، އެއުރެން ދެންނެވޫއެވެ. އޭ العزيز އެވެ! ތިމަންމެންނަށާއި، ތިމަންމެންގެ أهل ވެރިންނަށް ތަދުމަޑުކަން ޖެހިއްޖެއެވެ. ތިމަންމެންއައީ، ދެރަ މުދަލަކާއިގެންނެވެ. ފަހެ، ތިމަންމެންނަށް ފުރިހަމަގޮތުގައި މިނެދެއްވައި، ތިމަންމެންނަށް صدقات ކުރައްވާށެވެ! ހަމަކަށަވަރުން، صدقات ކުރާ މީހުންނަށް اللَّه (ހެޔޮ) ޖަޒާދެއްވާނެތެވެ.
قَالَ هَلْ عَلِمْتُم مَّا فَعَلْتُم بِيُوسُفَ وَأَخِيهِ إِذْ أَنتُمْ جَاهِلُونَ ( 89 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންނަށް ނޭނގިތިބި حال، يوسف ގެފާނަށާއި، އެކަލޭގެފާނުގެ أخ އާއަށް ތިޔަބައިމީހުން ކުރިކަންތައް ތިޔަބައިމީހުން ދެނެތިބީމުހެއްޔެވެ؟
قَالُوا أَإِنَّكَ لَأَنتَ يُوسُفُ ۖ قَالَ أَنَا يُوسُفُ وَهَٰذَا أَخِي ۖ قَدْ مَنَّ اللَّهُ عَلَيْنَا ۖ إِنَّهُ مَن يَتَّقِ وَيَصْبِرْ فَإِنَّ اللَّهَ لَا يُضِيعُ أَجْرَ الْمُحْسِنِينَ ( 90 )
އެއުރެން ދެންނެވޫއެވެ. ހަމަކަށަވަރުން، ތިޔައީ يوسف ގެފާނު ތޯއެވެ؟ އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އަހުރެންނަކީ، يوسف ގެފާނީމެވެ. އަދި މިއީ ތިމަންކަލޭގެފާނުގެ أخ އާއެވެ. ހަމަކަށަވަރުން، اللَّه ތިމަންމެންގެ މައްޗަށް ދަރުމަވަންތަވެވޮޑިގެންފިއެވެ. ހަމަކަށަވަރުން، تقوى ވެރިވެ ކެތްތެރިވާމީހާ (ދަންނާށެވެ!) ފަހެ، ހަމަކަށަވަރުން، إحسان ތެރިންގެ ދަރުމަ، اللَّه ގެއްލެނި ނުކުރައްވާނެތެވެ.
قَالُوا تَاللَّهِ لَقَدْ آثَرَكَ اللَّهُ عَلَيْنَا وَإِن كُنَّا لَخَاطِئِينَ ( 91 )
އެއުރެން ދެންނެވޫއެވެ. اللَّه ގަންދީ ބުނަމުއެވެ. ހަމަކަށަވަރުން، اللَّه ތިމަންމެންގެ މައްޗަށް ކަލޭގެފާނު މަތިވެރިކުރައްވައި، މާތްކުރެއްވިއެވެ. އަދި ހަމަކަށަވަރުން، ތިމަންމެންވަނީ، ކުށްވެރިން ކަމުގައެވެ.
قَالَ لَا تَثْرِيبَ عَلَيْكُمُ الْيَوْمَ ۖ يَغْفِرُ اللَّهُ لَكُمْ ۖ وَهُوَ أَرْحَمُ الرَّاحِمِينَ ( 92 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. މިއަދު ތިޔަބައިމީހުންގެ މައްޗަކަށް މަލާމާތެއް ދެރަބަހެއް ނެތެވެ. ތިޔަބައިމީހުންނަށް اللَّه ފާފަފުއްސަވާށިއެވެ! އަދި އެކަލާނގެއީ، رحمة ލައްވާ އެންމެ رحمة ވަންތަ ރަސްކަލާނގެއެވެ.
اذْهَبُوا بِقَمِيصِي هَٰذَا فَأَلْقُوهُ عَلَىٰ وَجْهِ أَبِي يَأْتِ بَصِيرًا وَأْتُونِي بِأَهْلِكُمْ أَجْمَعِينَ ( 93 )
ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނުގެ މި ގަމީސްކޮޅާއިގެން ދާށެވެ! އަދި ތިމަންކަލޭގެފާނުގެ ބައްޕާގެ މޫނުފުޅުމައްޗަށް އެ ގަމީސްކޮޅު އެއްލާށެވެ! އޭރުން ލޮލަށް ފެންނަމީހަކު ކަމުގައި އެކަލޭގެފާނު އަލުން ވެވަޑައިގެންފާނެތެވެ. އަދި ތިޔަބައިމީހުންގެ أهل ވެރިންގެ އެންމެން ގޮވައިގެން ތިމަންކަލޭގެފާނުގެ އަރިހަށް އަންނާށެވެ!
وَلَمَّا فَصَلَتِ الْعِيرُ قَالَ أَبُوهُمْ إِنِّي لَأَجِدُ رِيحَ يُوسُفَ ۖ لَوْلَا أَن تُفَنِّدُونِ ( 94 )
އެ قافلة (ރަށުން) ވަކިވިހިނދު، އެއުރެންގެ ބައްޕާ ވިދާޅުވިއެވެ. ހަމަކަށަވަރުން، ތިމަންކަލޭގެފާނަށް يوسف ގެފާނުގެ ވަސް އެބަދުވައެވެ. ތިޔަބައިމީހުން ތިމަންކަލޭގެފާނު މޮޔައަކަށް ނަހަދާނެނަމައެވެ.
قَالُوا تَاللَّهِ إِنَّكَ لَفِي ضَلَالِكَ الْقَدِيمِ ( 95 )
އެއުރެން ދެންނެވޫއެވެ. اللَّه ގަންދީ ބުނަމުއެވެ. ހަމަކަށަވަރުން، ކަލޭގެފާނުވަނީ، ކުރިންވެސް ކަލޭގެފާނު ހުންނެވި ކުށް خيال ގައިކަން ކަށަވަރެވެ.
فَلَمَّا أَن جَاءَ الْبَشِيرُ أَلْقَاهُ عَلَىٰ وَجْهِهِ فَارْتَدَّ بَصِيرًا ۖ قَالَ أَلَمْ أَقُل لَّكُمْ إِنِّي أَعْلَمُ مِنَ اللَّهِ مَا لَا تَعْلَمُونَ ( 96 )
ފަހެ، އުފާވެރިކަމުގެ خبر ދޭމީހާ އައިސް އެކަލޭގެފާނުގެ މޫނުފުޅުމައްޗަށް އެ ގަަމީސްކޮޅު އެއްލިއެވެ. ފަހެ، އެހިނދު ލޮލަށް ފެންނަމީހަކު ކަމުގައި އަލުން އެކަލޭގެފާނު ވެވަޑައިގެންފިއެވެ. އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންނަށް ނޭނގޭ ކަންތައްތަކެއް ހަމަކަށަވަރުން، اللَّه ގެ حضرة ން ތިމަންކަލޭގެފާނަށް އެނގިވަޑައިގަންނަވާކަމަށް، ތިމަންކަލޭގެފާނު ވިދާޅެއް ނުވަންހެއްޔެވެ؟
قَالُوا يَا أَبَانَا اسْتَغْفِرْ لَنَا ذُنُوبَنَا إِنَّا كُنَّا خَاطِئِينَ ( 97 )
އެއުރެން ދެންނެވޫއެވެ. އޭ އަހުރެމެންގެ ބައްޕާއެވެ! ތިމަންމެންނަށް ތިމަންމެންގެ ފާފަތައް ފުއްސެވުމަށް، ކަލޭގެފާނު دعاء ކޮށްދެއްވާށެވެ! ހަމަކަށަވަރުން، ތިމަންމެންވީ ކުށްވެރިން ކަމުގައެވެ.
قَالَ سَوْفَ أَسْتَغْفِرُ لَكُمْ رَبِّي ۖ إِنَّهُ هُوَ الْغَفُورُ الرَّحِيمُ ( 98 )
އެކަލޭގެފާނު ވިދާޅުވިއެވެ. ތިޔަބައިމީހުންނަށް ފާފަފުއްސެވުމަށްއެދި ތިމަންކަލޭގެފާނުގެ ވެރިރަސްކަލާނގެ حضرة އަށް دعاء ކުރައްވާހުށީމެވެ. ހަމަކަށަވަރުން، އެކަލާނގެއީ، ގިނަގިނައިން ފާފަފުއްސަވާ، رحيم ވަންތަ ރަސްކަލާނގެއެވެ.
فَلَمَّا دَخَلُوا عَلَىٰ يُوسُفَ آوَىٰ إِلَيْهِ أَبَوَيْهِ وَقَالَ ادْخُلُوا مِصْرَ إِن شَاءَ اللَّهُ آمِنِينَ ( 99 )
ފަހެ، އެއުރެން (އެންމެން) يوسف ގެފާނުގެ އަރިހަށް (ގޮސް) ވަންހިނދު، އެކަލޭގެފާނުގެ މައިންބަފައިން އެކަލޭގެފާނުގެ އަރިހަށް ލަފައިލެއްވިއެެވެ. އަދި ވިދާޅުވިއެވެ. اللَّه ގެ އިރާދަފުޅާއެކު، އަމާންކަން ލިބިގެންވާ حال ގައި، مصر އަށް ވެދެވަޑައިގަންނަވާށެވެ!
وَرَفَعَ أَبَوَيْهِ عَلَى الْعَرْشِ وَخَرُّوا لَهُ سُجَّدًا ۖ وَقَالَ يَا أَبَتِ هَٰذَا تَأْوِيلُ رُؤْيَايَ مِن قَبْلُ قَدْ جَعَلَهَا رَبِّي حَقًّا ۖ وَقَدْ أَحْسَنَ بِي إِذْ أَخْرَجَنِي مِنَ السِّجْنِ وَجَاءَ بِكُم مِّنَ الْبَدْوِ مِن بَعْدِ أَن نَّزَغَ الشَّيْطَانُ بَيْنِي وَبَيْنَ إِخْوَتِي ۚ إِنَّ رَبِّي لَطِيفٌ لِّمَا يَشَاءُ ۚ إِنَّهُ هُوَ الْعَلِيمُ الْحَكِيمُ ( 100 )
އަދި يوسف ގެފާނު އެކަލޭގެފާނުގެ މައިބަފައިކަލުން تخت އަށް އުފުއްލެވިއެވެ. އަދި އެ އެންމެން އެކަލޭގެފާނަށް ކުރުނީސްކޮށް، سجدة އަށް ތިރިވެގަތޫއެވެ. އަދި އެކަލޭގެފާނު ވިދާޅުވިއެވެ. އޭ ބައްޕާއެވެ! ކުރިން ތިމަންނާއަށް ފެނިވަޑައިގަތް ހުވަފެނުގެ މާނައަކީ، މިއީއެވެ. ހަމަކަށަވަރުން، ތިމަންގެ ވެރިރަސްކަލާނގެ، އެ ހުވަފެން حق ތެދެއް ކަމުގައި ލެއްވިއެވެ. އަދި ތިމަން ޖަލުން ނެރުއްވުމާއި، ތިމަންނާއާއި، ތިމަންގެ أخ ންނާ ދެމެދު شيطان އާ ނުބައިކޮށްފުމަށް ފަހުގައި، بدوى ންގެ ދިރިއުޅުމުން ތިޔަބައިމީހުން ގެންނެވި ކަމާއި، މިހުރިހާ ކަމެއްގައި ތިމަންނާއަށް އެކަލާނގެ ރަނގަޅުގޮތް މެދުވެރިކުރެއްވިއެވެ. ހަމަކަށަވަރުން، ތިމަންގެ ވެރިރަސްކަލާނގެއީ، އެކަލާނގެ އިރާދަކުރައްވާ ކަމަކަށް އޯގާވަންތަ ރަސްކަލާނގެއެވެ. ހަމަކަށަވަރުން، އެކަލާނގެއީ، މޮޅަށް ދެނެވޮޑިގެންވާ، حكيم ވަންތަ ރަސްކަލާނގެއެވެ.
رَبِّ قَدْ آتَيْتَنِي مِنَ الْمُلْكِ وَعَلَّمْتَنِي مِن تَأْوِيلِ الْأَحَادِيثِ ۚ فَاطِرَ السَّمَاوَاتِ وَالْأَرْضِ أَنتَ وَلِيِّي فِي الدُّنْيَا وَالْآخِرَةِ ۖ تَوَفَّنِي مُسْلِمًا وَأَلْحِقْنِي بِالصَّالِحِينَ ( 101 )
އަޅުގެ ވެރިރަސްކަލާކޮ! ހަމަކަށަވަރުން، އިބަރަސްކަލާނގެ ތިމަންއަޅާއަށް ވެރިކަން ދެއްވީމުއެވެ. އަދި ހުވަފެންތައް މާނަކުރަން ދަސްކޮށްދެއްވީމުއެވެ. އިސްވެދިޔަ مثال އަކާނުލައި، އުޑުތަކާއި، ބިން ހެއްދެވި ރަސްކަލާކޮ! ދުނިޔެއާއި، آخرة ގައި މިއަޅާގެ އެހީތެރިއަކީ، އިބަރަސްކަލާނގެއެވެ. مسلم އެއްގެ ގޮތުގައި މިއަޅާ މަރުގަންނަވާނދޭވެ! އަދި صالح ންގެ ތެރެއިން މިއަޅާ ލައްވާނދޭވެ!
ذَٰلِكَ مِنْ أَنبَاءِ الْغَيْبِ نُوحِيهِ إِلَيْكَ ۖ وَمَا كُنتَ لَدَيْهِمْ إِذْ أَجْمَعُوا أَمْرَهُمْ وَهُمْ يَمْكُرُونَ ( 102 )
އެއީ غيب ގެ خبر ތަކުގެ ތެރެއިން، ކަލޭގެފާނަށް ތިމަންރަސްކަލާނގެ وحى ކުރައްވާ خبر ތަކެވެ. އަދި އެއުރެން މަކަރުވެރި ރޭވުންތައް ރާވާ حال، އެއުރެންގެ ކަންތައްތަކަށް އެއްބަސްވިއިރަކު، އެއުރެންގެ ގާތުގައި ކަލޭގެފާނު ވަޑައިގެން ނުވަމުއެވެ.
وَمَا أَكْثَرُ النَّاسِ وَلَوْ حَرَصْتَ بِمُؤْمِنِينَ ( 103 )
އަދި ކަލޭގެފާނު ކިތަންމެވަރަކަށް شوق ވެރިވެވަޑައިގަތަސް، މީސްތަކުންކުރެ ގިނަމީހުން إيمان ވާ ބަޔަކު ކަމުގައެއް ނުވެތެވެ.
وَمَا تَسْأَلُهُمْ عَلَيْهِ مِنْ أَجْرٍ ۚ إِنْ هُوَ إِلَّا ذِكْرٌ لِّلْعَالَمِينَ ( 104 )
އަދި އެ قرآن އަށްޓަކައި، ކަލޭގެފާނު އެއުރެންގެ ކިބައިން އުޖޫރައަކަށް އެދިވަޑައެއް ނުގަންނަވަމުއެވެ. عالم ތަކަށް ހަނދާންކޮށްދިނުމެއްކަމުގައި މެނުވީ، އެ قرآن ނުވެތެވެ.
وَكَأَيِّن مِّنْ آيَةٍ فِي السَّمَاوَاتِ وَالْأَرْضِ يَمُرُّونَ عَلَيْهَا وَهُمْ عَنْهَا مُعْرِضُونَ ( 105 )
އުޑުތަކާއި، ބިމުގައިވާ ހެކިތަކުގެ ތެރެއިން، އެއުރެންނަށް ފެނި، ކުރިމަތިވާ، އަދި އެ ހެކިތަކުން އެއުރެން އެއްކިބާވެ، އެއާމެދު ފިކުރުނުކުރާ، ކިތައްކިތައް ހެކި ވޭހެއްޔެވެ؟
وَمَا يُؤْمِنُ أَكْثَرُهُم بِاللَّهِ إِلَّا وَهُم مُّشْرِكُونَ ( 106 )
އަދި އެއުރެންކުރެ ގިނަމީހުން، شريك ކުރާ ބަޔަކުކަމުގައިވާ حال ގައި މެނުވީ، اللَّه އަށް إيمان އެއް ނުވެތެވެ.
أَفَأَمِنُوا أَن تَأْتِيَهُمْ غَاشِيَةٌ مِّنْ عَذَابِ اللَّهِ أَوْ تَأْتِيَهُمُ السَّاعَةُ بَغْتَةً وَهُمْ لَا يَشْعُرُونَ ( 107 )
ފަހެ، اللَّه ގެ عذاب ގެ ތެރެއިން ކޯފާއެއް އެއުރެންގެ މައްޗަށް އަތުވެދާނެކަމުގެ ބިރެއް، ނުވަތަ އެއުރެން ނުދެނެތިއްބައި، ކުއްލިއަކަށް قيامة ގެ ވަގުތު އެއުރެންނަށް އަތުވެދާނޭކަމުގެ ބިރެއްނެތި، އެއުރެން އަމާންކަން ލިބިގެން ތިބީހެއްޔެވެ؟
قُلْ هَٰذِهِ سَبِيلِي أَدْعُو إِلَى اللَّهِ ۚ عَلَىٰ بَصِيرَةٍ أَنَا وَمَنِ اتَّبَعَنِي ۖ وَسُبْحَانَ اللَّهِ وَمَا أَنَا مِنَ الْمُشْرِكِينَ ( 108 )
ކަލޭގެފާނު ވިދާޅުވާށެވެ! މިއީ ތިމަންކަލޭގެފާނުގެ މަގެވެ. ތިމަންކަލޭގެފާނާއި، ތިމަންކަލޭގެފާނަށް تبع ވި މީހުން، اللَّه ގެ ފަރާތަށް ގޮވައިލައްވަނީ، އަލިވެގެންވާ دليل އާއެކުގައެވެ. އަދި اللَّه ގެ ހުސްطاهر ވަންތަކަމާއެވެ! އަދި شريك ކުރާ މީހުންގެ ތެރެއިން ތިމަންކަލޭގެފާނު ނުވަމެވެ.
وَمَا أَرْسَلْنَا مِن قَبْلِكَ إِلَّا رِجَالًا نُّوحِي إِلَيْهِم مِّنْ أَهْلِ الْقُرَىٰ ۗ أَفَلَمْ يَسِيرُوا فِي الْأَرْضِ فَيَنظُرُوا كَيْفَ كَانَ عَاقِبَةُ الَّذِينَ مِن قَبْلِهِمْ ۗ وَلَدَارُ الْآخِرَةِ خَيْرٌ لِّلَّذِينَ اتَّقَوْا ۗ أَفَلَا تَعْقِلُونَ ( 109 )
އަދި ކަލޭގެފާނުގެ ކުރިންވެސް، ރަށްތަކުގެ أهل ވެރިންގެ ތެރެއިން އެބަޔަކަށް ތިމަންރަސްކަލާނގެ وحى އިއްވަވައި، رسول ކަމާއިގެން ފިރިހެނުން މެނުވީ ނުފޮނުއްވަމެވެ. ފަހެ، ބިމުގައި ދަތުރުފަތުރުކޮށް އެއުރެންގެ ކުރިން އުޅުނު މީހުންގެ ނިމުންވީ ކޮންފަދައަކުންކަން، އެއުރެން ނުބަލަނީ ހެއްޔެވެ؟ އަދި ހަމަކަށަވަރުން، تقوى ވެރިވި މީހުންނަށް، آخرة ގެ ގޮވަތި، ހެޔޮކަން ބޮޑެވެ. ފަހެ، ތިޔަބައިމީހުންނަކަށް ނުވިސްނޭހެއްޔެވެ؟
حَتَّىٰ إِذَا اسْتَيْأَسَ الرُّسُلُ وَظَنُّوا أَنَّهُمْ قَدْ كُذِبُوا جَاءَهُمْ نَصْرُنَا فَنُجِّيَ مَن نَّشَاءُ ۖ وَلَا يُرَدُّ بَأْسُنَا عَنِ الْقَوْمِ الْمُجْرِمِينَ ( 110 )
(މެދުވެރިވާ ދުއްތުރާތަކުގެ ސަބަބުން) رسول ންގެ اميد ކެނޑިި، އެބޭކަލުން ދޮގުކުރެވުނީކަމަށް އެބޭކަލުންނަށް ހީވެދާހިނދު، ތިމަންރަސްކަލާނގެ نصر، އެބޭކަލުންނަށް އަތުވެދެއެވެ. ފަހެ، ތިމަންރަސްކަލާނގެ އިރާދަކުރައްވާ ބަޔަކު ސަލާމަތްކުރައްވަމެވެ. އަދި ކުށްވެރިވި މީހުންގެ ކިބައިން، ތިމަންރަސްކަލާނގެ عذاب އަނބުރާ ރައްދުނުކުރެވޭ ހުށްޓެވެ.
لَقَدْ كَانَ فِي قَصَصِهِمْ عِبْرَةٌ لِّأُولِي الْأَلْبَابِ ۗ مَا كَانَ حَدِيثًا يُفْتَرَىٰ وَلَٰكِن تَصْدِيقَ الَّذِي بَيْنَ يَدَيْهِ وَتَفْصِيلَ كُلِّ شَيْءٍ وَهُدًى وَرَحْمَةً لِّقَوْمٍ يُؤْمِنُونَ ( 111 )
ހަމަކަށަވަރުން، އެއުރެންގެ ވާހަކަތަކުގައި، ބުއްދީގެ أهل ވެރިންނަށް عبرة އެއް ވާކަން ކަށަވަރެވެ. އެ قرآن އަކީ، ހަދައިގެން އުފެއްދޭ ވާހަކަތަކެއްނޫނެވެ. އެހެނެއްކަމަކު، އެއީ އޭގެކުރިން ބާވައިލެއްވި ފޮތްތައް ތެދުކުރާ، އަދި ހުރިހާކަމެއް تفصيل ކޮށްދޭ، އަދި إيمان ވާ ބަޔަކަށް هداية އެއް ކަމުގައްޔާއި، رحمة އެއް ކަމުގައިވާ ފޮތެވެ.
